night_coven_reforms = {
	trigger = { religion = night_coven }

	can_buy_idea = {
		is_at_war = no
		NOT = { num_of_rebel_controlled_provinces = 1 }
		stability = 1
		if = {
			limit = { ai = no }
			NOT = { overextension_percentage = 0.01 }
		}
		if = {
			limit = {
				NOT = {
					OR = {
						has_country_flag = night_coven_reform_prog_1
						has_country_flag = night_coven_reform_prog_2
						has_country_flag = night_coven_reform_prog_3
						has_country_flag = night_coven_reform_prog_4
					}
				}
			}
			num_of_owned_provinces_with = {
				value = 10
				NOT = { region = yezel_mora_region }
			}
		}
		else_if = {
			limit = { has_country_flag = night_coven_reform_prog_1 }
			num_of_owned_provinces_with = {
				value = 20
				NOT = { region = yezel_mora_region }
			}
		}
		else_if = {
			limit = { has_country_flag = night_coven_reform_prog_2 }
			num_of_owned_provinces_with = {
				value = 30
				NOT = { region = yezel_mora_region }
			}
		}
		else_if = {
			limit = { has_country_flag = night_coven_reform_prog_3 }
			num_of_owned_provinces_with = {
				value = 40
				NOT = { region = yezel_mora_region }
			}
		}
		else_if = {
			limit = { has_country_flag = night_coven_reform_prog_4 }
			if = {
				limit = { ai = yes }
				always = yes
			}
			else = {
				num_of_owned_provinces_with = {
					value = 50
					NOT = { region = yezel_mora_region }
				}
			}
		}
	}
	
	night_coven_reform_1 = {
		infantry_power = 0.05
	}
	night_coven_reform_2 = {
		war_exhaustion = -0.03
	}
	night_coven_reform_3 = {
		available_province_loot = 0.50
	}
	night_coven_reform_4 = {
		reinforce_speed = 0.10
	}
	night_coven_reform_5 = {
		spy_offence = 0.15
	}
	ai_will_do = {
		factor = 1
	}
}

gods_of_the_taychend_reform_1 = {
	trigger = {
		OR = {
			religion = gods_of_the_taychend
			religion = oren_nayiru
		}
	}

	can_buy_idea = {
		if = {
			limit = { 
				OR = {
					has_country_flag = free_taychend_reform_flag 
					has_country_flag = G64_skip_first_two_reforms_flag
				}
			}
			always = yes
		}
		else = {
			is_at_war = no
			stability = 1
			NOT = { overextension_percentage = 0.01 }
			NOT = { num_of_rebel_armies = 1 }
			total_development = 100
			calc_true_if = {
				all_subject_country = { NOT = { liberty_desire = 50 } }
				amount = 2
			}
		}
	}

	gods_of_the_taychend_reform_1 = {
		stability_cost_modifier = -0.1
	}
	
	ai_will_do = {
		factor = 1
	}
}
