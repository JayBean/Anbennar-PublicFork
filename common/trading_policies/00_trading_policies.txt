# First policy below is the default if no other selected

maximize_profit = {
	can_select = {
		FROM = {
			has_trader = ROOT
		}
		NOT = {
			has_country_flag = cannot_use_maximize_profit
		}
	}

	trade_power = {
		duration = -1
		power_modifier = 0.05
		key = maximize_profits
	}
	
	center_of_reformation = no
	button_gfx = GFX_Trading_Policy_Max_Profit
}

hostile_trading = {
	can_select = {
		FROM = {
			has_trader = ROOT
		}
	}

	# Also applied versus countries with the node as their home node
	countries_with_merchant_modifier = {
		spy_offence = 0.25
	}
	
	button_gfx = GFX_Trading_Policy_Hostile_Trading
}

improve_inland_routes = {
	can_select = {
		FROM = {
			has_trader = ROOT
		}
		if = {
			limit = { 
				NOT = { 
					OR = {
						has_government_attribute = free_improve_inland_routes 
						has_country_flag = can_always_use_improve_inland_routes
					
					}
				} 
			}
			FROM = {
				trade_share = {
					country = ROOT
					share = 50
				}
			}
		}
	}

	can_maintain = {
		FROM = {
			has_trader = ROOT
		}
		if = {
			limit = { 
				NOT = { 
					OR = {
						has_government_attribute = free_improve_inland_routes 
						has_country_flag = can_always_use_improve_inland_routes
					
					}
				} 
			}
			FROM = {
				trade_share = {
					country = ROOT
					share = 40
				}
			}
		}
	}
	
	node_province_modifier = {
		siege_ability = 0.10
		artillery_levels_available_vs_fort = 1
	}	
	button_gfx = GFX_Trading_Policy_Improve_Routes
}
