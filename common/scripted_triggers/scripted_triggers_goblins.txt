		trade_goods = mithril

		trade_goods = coal

factory_farm_check = {
	AND = {
		NOT = {
			has_province_modifier = allclan_factory_farm_food
			has_province_modifier = allclan_factory_farm_cash_crops
			has_province_modifier = allclan_factory_farm_spidersilk
		}
		has_valid_factory_farm_good = yes
	}
}
