harpy_move_pop_aspect = {
	sprite = "GFX_dissolve_monasteries_icon"
	cost = 100
	
	trigger = {
		NOT = { has_country_flag = open_harpy_take_pop_menu }
		OR = {
			any_owned_province = {
				NOT = { is_capital_of = ROOT has_harpy_majority_trigger = yes }
				development = 4
			}
			any_subject_country = {
				any_owned_province = {
					NOT = { has_harpy_majority_trigger = yes }

				}

		}

	effect = {
		country_event = {
			id = the_hunt_events.2
			days = 0
		}
		hidden_effect = { set_country_flag = open_harpy_take_pop_menu }
	}
	
	ai_will_do = {
		factor = 10
		modifier = {
			factor = 0
			is_at_war = yes
		}
	}
}
