federation_cohesion_triggers = {
	hidden = yes
	potential = {
		NOT = { has_global_flag = lake_federation_disbanded }
		NOT = { has_global_flag = lake_federation_formed }
	}

	trigger = {
		has_global_flag = lake_cohesion_1
	}

	#Province scope
	on_activation = {
		if = {
			limit = { owner = { has_country_modifier = lake_federation_member } }
			owner = { country_event = { id = splintering.1 } }
		}
		else = {
			random_country = {
				limit = { has_country_modifier = lake_federation_member }
				country_event = { id = splintering.1 }
			}
		}
	}

	on_deactivation = {
		if = {
			limit = {
				NOT = {
					has_global_flag = lake_cohesion_5
					has_global_flag = lake_cohesion_4
					has_global_flag = lake_cohesion_3
					has_global_flag = lake_cohesion_2
				}
			}
			every_country = {
				limit = { has_country_modifier = lake_federation_member }
				country_event = { id = lake.39 }
			}
		}
	}
}

# Legion maintenance modifier setup - for Castanor
castanor_legions_1_maintenance_triggered = {
    potential = {
        B32 = { exists = yes }
    }
	hidden = yes

    trigger = {
        B32 = {
            any_hired_mercenary_company = {
                template = merc_castanorian_legion_1
            }
        }
    }

    on_activation = {
		# First time hire manpower fix
		if = {
			limit = { B32 = { has_country_flag = merc_castanorian_legion_1_first_hire } }
			B32 = {
				clr_country_flag = merc_castanorian_legion_1_first_hire
				random_hired_mercenary_company = {
					limit = { template = merc_castanorian_legion_1 }
					add_company_manpower = -1
					add_company_manpower = 0.1
				}
			}
		}
		if = {
			limit = { B32 = { has_country_modifier = castanor_legion_doctrine_full_citizen_army } }
			B32 = {
				add_country_modifier = {
					name = castanor_legions_citizen_army_1_maintenance_mod
					duration = -1
				}
			}
		}
		else = {
			B32 = {
				add_country_modifier = {
					name = castanor_legions_1_maintenance_mod
					duration = -1
				}
			}
		}
		B32 = { remove_country_modifier = castanor_legions_citizen_army_1_maintenance_mod_inactive }
		B32 = { remove_country_modifier = castanor_legions_1_maintenance_mod_inactive }
		B32 = { add_army_professionalism = 0.05 }
    }

    on_deactivation = {
		if = {
			limit = { B32 = { has_country_modifier = castanor_legion_doctrine_full_citizen_army } }
			B32 = {
				add_country_modifier = {
					name = castanor_legions_citizen_army_1_maintenance_mod_inactive
					duration = -1
				}
			}
		}
		else = {
			B32 = {
				add_country_modifier = {
					name = castanor_legions_1_maintenance_mod_inactive
					duration = -1
				}
			}
		}
		B32 = { remove_country_modifier = castanor_legions_citizen_army_1_maintenance_mod }
		B32 = { remove_country_modifier = castanor_legions_1_maintenance_mod }
    }
}

rahen_repairing_infrastructure_monthly_tick = {
	potential = {
		OR = {
			has_province_modifier = rahen_severely_neglected_infrastructure
			has_province_modifier = rahen_neglected_infrastructure
		}
	}
	trigger = {
		OR = {
			AND = {
				is_month = 0
				NOT = { is_month = 1 }
			}
			AND = {
				is_month = 2
				NOT = { is_month = 3 }
			}
			AND = {
				is_month = 4
				NOT = { is_month = 5 }
			}
			AND = {
				is_month = 6
				NOT = { is_month = 7 }
			}
			AND = {
				is_month = 8
				NOT = { is_month = 9 }
			}
			AND = {
				is_month = 10
				NOT = { is_month = 11 }
			}
		}
	}
	on_activation = {
		if = {
			limit = { infantry_in_province  = 2 }
			change_variable = {
				which = rahen_months_to_repair
				value = -1
			}
			if = {
				limit = { NOT = { check_variable = { which = rahen_months_to_repair value = 1 } } }
				if = {
					limit = { has_province_modifier = rahen_severely_neglected_infrastructure }
					remove_province_modifier = rahen_severely_neglected_infrastructure
					add_permanent_province_modifier = {
						name = rahen_neglected_infrastructure
						duration = -1
					}
					set_variable = { which = rahen_months_to_repair value = 6 }
				}
				else = {
					remove_province_modifier = rahen_neglected_infrastructure
					remove_province_triggered_modifier = rahen_repairing_infrastructure_monthly_tick
					remove_province_triggered_modifier = rahen_repairing_infrastructure
				}
			}
		}
	}
	hidden = yes
	on_deactivation = {
		if = {
			limit = { infantry_in_province  = 2 has_siege = no }
			change_variable = {
				which = rahen_months_to_repair
				value = -1
			}
			if = {
				limit = { NOT = { check_variable = { which = rahen_months_to_repair value = 1 } } }
				if = {
					limit = { has_province_modifier = rahen_severely_neglected_infrastructure }
					remove_province_modifier = rahen_severely_neglected_infrastructure
					add_permanent_province_modifier = {
						name = rahen_neglected_infrastructure
						duration = -1
					}
				}
				else = {
					remove_province_modifier = rahen_neglected_infrastructure
					remove_province_triggered_modifier = rahen_repairing_infrastructure_monthly_tick
					remove_province_triggered_modifier = rahen_repairing_infrastructure
				}
			}
		}
	}
}
