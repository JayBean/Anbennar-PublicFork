

#Standard Agendas
estate_boar_command_hire_advisor = {
	can_select = {
		NOT = {
			employed_advisor = {
				category = MIL
			}
		}
		NOT = { 
			AND = {
				command_has_potential_campaign = yes
				command_has_available_boar_campaign = yes
			}
		}
	}
	selection_weight = {
		factor = 1
	}
	task_requirements = {
		OR = {
			commandant = 1
			recruitmaster = 1
		}
		is_in_deficit = no
	}
    task_completed_effect = {
		on_completed_agenda_effect = {
			estate = estate_boar_command
		}
		add_estate_loyalty = {
			estate = estate_boar_command
			loyalty = 10
		}
    }
            failing_effect = {
		on_failed_agenda_effect = yes
		clr_auto_complete_flag = yes
		on_failed_agenda_effect = yes
		clr_auto_complete_flag = yes
        add_estate_loyalty_modifier = {
            estate = estate_boar_command
			desc = EST_VAL_AGENDA_DENIED
			loyalty = -5
			duration = 7300
        }
    }
}

#Restore legitimacy
estate_wolf_command_restore_legitimacy = {
	can_select = {
		NOT = { legitimacy_equivalent = 65 }
		government = monarchy
		NOT = { has_ruler_modifier = legitimacy_defended }
		NOT = { has_regency = yes }
		faction_in_power = hob_wolf_command
		NOT = { 
			AND = {
				command_has_potential_campaign = yes
				command_has_available_wolf_campaign = yes
			}
		}
	}
	selection_weight = {
		factor = 2
		modifier = {
			factor = 1.5
			NOT = { legitimacy_equivalent = 50 }
		}
		modifier = {
			factor = 1.5
			NOT = { legitimacy_equivalent = 35 }
		}
	}
	fail_if = {
		OR = {
			NOT = { government = monarchy }
			has_regency = yes
		}
	}
	task_requirements = {
		legitimacy_equivalent = 90
	}
    task_completed_effect = {
		on_completed_agenda_effect = {
			estate = estate_wolf_command
		}
		add_estate_loyalty = {
			estate = estate_wolf_command
			loyalty = 10
		}
		add_ruler_modifier = {
			name = legitimacy_defended
		}
    }
        failing_effect = {
		on_failed_agenda_effect = yes
		clr_auto_complete_flag = yes
        add_estate_loyalty_modifier = {
            estate = estate_wolf_command
			desc = EST_VAL_AGENDA_DENIED
			loyalty = -5
			duration = 7300
        }
    }
}

estate_wolf_command_develop_x = { #Tax
	selection_weight = {
		factor = 1
	}
	can_select = {
		capital_scope = {
			is_state_core = root
		}
		NOT = { 
			AND = {
				command_has_potential_campaign = yes
				command_has_available_wolf_campaign = yes
			}
		}
	}
	pre_effect = {
		random_list = {
			1 = {
				random_owned_province = {
					limit = {
						is_state_core = root
						area_for_scope_province = {
							is_capital_of = root
						}
					}
					save_event_target_as = agenda_province
				}
			}
			2 = {
				trigger = {
					any_owned_province = {
						is_state_core = root
						OR = {
							AND = {
								religion = root
								OR = {
									culture = root
									has_owner_accepted_culture = yes
								}
							}
						}
						region_for_scope_province = {
							is_capital_of = root
						}
					}
				}
				random_owned_province = {
					limit = {
						is_state_core = root
						OR = {
							AND = {
								religion = root
								OR = {
									culture = root
									has_owner_accepted_culture = yes
								}
							}
						}
						region_for_scope_province = {
							is_capital_of = root
						}
					}
					save_event_target_as = agenda_province
				}
			}
			1 = {
				random_owned_province = {
					limit = {
						is_state_core = root
					}
					save_event_target_as = agenda_province
				}
			}
		}
		event_target:agenda_province = {
			if = {
				limit = {
					check_variable = {
						which = province_adm_var
						value = 1
					}
				}
				set_variable = {
					which = province_adm_var
					value = 0
				}
			}
			export_to_variable = {
				which = province_adm_var
				value = trigger_value:base_tax
			}
			change_variable = {
				which = province_adm_var
				value = 2
			}
		}
	}
	provinces_to_highlight = {
		province_id = event_target:agenda_province
	}
	fail_if = {
		event_target:agenda_province = {
			NOT = { owned_by = root }
		}
	}
    task_requirements = {
		event_target:agenda_province = {
			base_tax = "province_adm_var"
		}
    }
    task_completed_effect = {
		on_completed_agenda_effect = {
			estate = estate_wolf_command
		}
		add_estate_loyalty = {
			estate = estate_wolf_command
			loyalty = 10
		}
		add_prestige = 10
    }
        failing_effect = {
		on_failed_agenda_effect = yes
		clr_auto_complete_flag = yes
		event_target:agenda_province = {
			if = {
				limit = { owned_by = root }
				add_named_unrest = {
					name = local_wolfborn_displeased
					value = 5
				}
			}
		}
        add_estate_loyalty_modifier = {
            estate = estate_wolf_command
			desc = EST_VAL_AGENDA_DENIED
			loyalty = -5
			duration = 7300
        }
    }
}

					}
					remove_province_modifier = agenda_cultural_conversion_help
				}
			}
		}

		on_failed_agenda_effect = yes
		clr_auto_complete_flag = yes
        add_estate_loyalty_modifier = {
            estate = estate_elephant_command
			desc = EST_VAL_AGENDA_DENIED
