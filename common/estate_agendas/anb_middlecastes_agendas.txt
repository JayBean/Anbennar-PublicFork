		primitives = no

		}

		NOT = { has_country_modifier = naval_enthusiasm_mission }

		any_rival_country = {
			NOT = { war_with = root }
			NOT = { root = { navy_size = prev } }

				who = root
				value = 1.05
			}
			NOT = {
				navy_balance = {
					who = root
