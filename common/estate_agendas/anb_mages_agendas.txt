

#This one or the next one will always be valid
estate_mages_fire_advisor = {
	can_select = {
		employed_advisor = {
			category = ADM
		}
		NOT = {
			employed_advisor = {
				type = court_mage
			}
		}
	}
	selection_weight = {
		factor = 1
		modifier = {
			factor = 0.5
			always = yes
		}
		modifier = {
			factor = 0.5
			employed_advisor = {
				type = inquisitor
			}
		}
		modifier = {
			factor = 0.5
			employed_advisor = {
				type = natural_scientist
			}
		}
		modifier = {
			factor = 0.5
			NOT = {
				estate_influence = {
					estate = estate_mages
					influence = 50
				}
			}
		}
	}
	immediate_effect = {
		hidden_effect = {
			set_country_flag = need_to_fire_adm_advisor
		}
	}
	on_invalid = {
		clr_country_flag = need_to_fire_adm_advisor
	}
	task_requirements = {
		custom_trigger_tooltip = {
			tooltip = fire_adm_advisor_tooltip
			has_country_flag = just_fired_adm_advisor
		}
	}
    task_completed_effect = {
		on_completed_agenda_effect = {
			estate = estate_mages
		}
		add_estate_loyalty = {
			estate = estate_mages
			loyalty = 10
		}
		create_advisor = court_mage
		
		clr_country_flag = just_fired_adm_advisor
		clr_country_flag = need_to_fire_adm_advisor
	}
	invalid_trigger = {
		NOT = {
			employed_advisor = {
				category = ADM
			}
		}
		NOT = { has_country_flag = just_fired_adm_advisor }
	}
        failing_effect = {
		on_failed_agenda_effect = yes
		clr_auto_complete_flag = yes
        add_estate_loyalty_modifier = {
            estate = estate_mages
			desc = EST_VAL_AGENDA_DENIED
			loyalty = -5
			duration = 7300
        }
		clr_country_flag = need_to_fire_adm_advisor
    }
}

			}
		}
		modifier = {
			factor = 2

			NOT = { has_institution = enlightenment }
		}
	}
	pre_effect = {
		random_owned_province = {
			limit = {
				is_state_core = root
				NOT = { has_building = university }
				development = 10
			}
			save_event_target_as = agenda_province
		}
		random_owned_province = {
			limit = {
				is_state_core = root
				NOT = { has_building = university }
				development = 15
			}
			save_event_target_as = agenda_province
		}
		random_owned_province = {
			limit = {
				is_state_core = root
				NOT = { has_building = university }
				development = 20
			}
			save_event_target_as = agenda_province
		}
		random_owned_province = {
			limit = {
				is_state_core = root
				NOT = { has_building = university }
				is_capital = yes
			}
			save_event_target_as = agenda_province
		}
	}
	provinces_to_highlight = {
		province_id = event_target:agenda_province
	}
	fail_if = {
		event_target:agenda_province = {
			NOT = { owned_by = root }
		}

		event_target:agenda_province = {
			has_building = university
		}

		on_completed_agenda_effect = {
			estate = estate_mages

estate_mages_conquer_damestear = {
	can_select = {
		is_free_or_tributary_trigger = yes
		is_at_war = no
		any_neighbor_country = {
			#num_of_cities = 6
			is_free_or_tributary_trigger = yes
			NOT = { is_subject_of = root }
			NOT = { overlord_of = root }
			NOT = { alliance_with = root }
			NOT = { ROOT = { truce_with = prev } }
			NOT = { 
				has_opinion = {
					who = root
					value = 25
				}
			}
			any_owned_province = { #No need for this agenda if covered by a mission
				trade_goods = damestear
			}
			ROOT = {
				knows_country = prev
			}
			OR = {
				is_part_of_hre = no
				NOT = { hre_reform_passed = landfriede }
				ROOT = { is_part_of_hre = no }
			}
		}
	}
	selection_weight = {
		factor = 1
		modifier = {
			factor = 0.5
			has_non_generic_missions = yes
		}
		modifier = {
			factor = 1.5
			NOT = { damestear = 1 }
		}
		modifier = {
			factor = 1.5
			NOT = { damestear = 3 }
		}
		modifier = {
			factor = 0.25
			any_country = {
				coalition_target = root
			}
		}
	}
	pre_effect = {
		random_neighbor_country = {
			limit = {
				#num_of_cities = 6
				is_free_or_tributary_trigger = yes
				NOT = { is_subject_of = root }
				NOT = { overlord_of = root }
				NOT = { alliance_with = root }
				NOT = { ROOT = { truce_with = prev } }
				NOT = { 
					has_opinion = {
						who = root
						value = 25
					}
				}
				any_owned_province = { #No need for this agenda if covered by a mission
					trade_goods = damestear
				}
				ROOT = {
					knows_country = prev
				}
				OR = {
					is_part_of_hre = no
					NOT = { hre_reform_passed = landfriede }
					ROOT = { is_part_of_hre = no }
				}
			}
			random_owned_province = {
				limit = {
					trade_goods = damestear
				}
				save_event_target_as = agenda_province
			}
			random_owned_province = {
				limit = {
					trade_goods = damestear

					any_neighbor_province = {
						owned_by = root
					}
				}
				save_event_target_as = agenda_province
			}
		}
	}
	immediate_effect = {
		event_target:agenda_province = {
			add_claim = root
		}
	}
	on_invalid = {
		if = {
			limit = {
				event_target:agenda_province = {
					is_claim = root
					NOT = { is_permanent_claim = root }
				}
			}
			event_target:agenda_province = {
				remove_claim = root
			}
		}
		if = {
			limit = {
				event_target:agenda_province = {
					NOT = { trade_goods = damestear }
				}
			}
			event_target:agenda_province = {
				remove_claim = root
			}
		}
	}
	provinces_to_highlight = {
		area_for_scope_province = {
			province_id = event_target:agenda_province
		}
	}
	task_requirements = {
		event_target:agenda_province = {
			country_or_non_sovereign_subject_holds = root
		}
	}
	fail_if = {
		event_target:agenda_province = {
			NOT = { is_claim = root }
			NOT = { is_core = root }
			NOT = { owned_by = root }
		}
	}
	task_completed_effect = {
		add_estate_loyalty = {
			estate = estate_mages
			loyalty = 20
		}
		add_adm_power = 50
	}
        failing_effect = {
		on_failed_agenda_effect = yes
		clr_auto_complete_flag = yes
        add_estate_loyalty_modifier = {
            estate = estate_mages
			desc = EST_VAL_AGENDA_DENIED
			loyalty = -5
			duration = 7300
        }
		if = {
			limit = {
				event_target:agenda_province = {
					is_claim = root
					NOT = { is_permanent_claim = root }
				}
			}
			event_target:agenda_province = {
					remove_claim = root
			}
		}
    }
}

#Fort on Desired Tradegood
estate_mages_build_fort_building_on_damestear = {
	can_select = {
		any_owned_province = {
			trade_goods = damestear
			OR = {
				AND = {
					NOT = { has_building = fort_15th }
					NOT = { has_building = fort_16th }
					NOT = { has_building = fort_17th }
					NOT = { has_building = fort_18th }
					num_free_building_slots = 1
					any_neighbor_province = {
						owner = {
							OR = {
								is_rival = root
								root = { is_rival = prev }
								AND = {
									is_subject_other_than_tributary_trigger = yes
									overlord = {
										OR = {
											is_rival = root
											root = { is_rival = prev }
										}
									}
								}
							}
						}
					}
					NOT = {
						any_neighbor_province = {
							owned_by = root
							OR = {
								has_building = fort_15th
								has_building = fort_16th
								has_building = fort_17th
								has_building = fort_18th
							}
						}
					}
				}
				AND = {
					NOT = { has_building = fort_15th }
					NOT = { has_building = fort_16th }
					NOT = { has_building = fort_17th }
					NOT = { has_building = fort_18th }
					num_free_building_slots = 1
					any_neighbor_province = {
						owned_by = root
						any_neighbor_province = {
							owner = {
								OR = {
									is_rival = root
									root = { is_rival = prev }
									AND = {
										is_subject_other_than_tributary_trigger = yes
										overlord = {
											OR = {
												is_rival = root
												root = { is_rival = prev }
											}
										}
									}
								}
							}
							NOT = {
								any_neighbor_province = {
									owned_by = root
									OR = {
										has_building = fort_15th
										has_building = fort_16th
										has_building = fort_17th
										has_building = fort_18th
									}
								}
							}
						}
					}
				}
				AND = {
					has_building = fort_15th
					ROOT = { mil_tech = 14 }
				}
				AND = {
					has_building = fort_16th
					ROOT = { mil_tech = 19 }
				}
				AND = {
					has_building = fort_17th
					ROOT = { mil_tech = 24 }
				}
			}
			NOT = { has_building = fort_18th }
		}
	}
	selection_weight = {
		factor = 1
		modifier = {
			factor = 1.5
			any_neighbor_country = {
				OR = {
					is_rival = root
					root = { is_rival = prev }
				}
				army_size = root
			}
		}
	}
	pre_effect = {
		set_variable = {
			which = estate_mages_build_fort_building_in_y_var
			value = 0
		}
		random_owned_province = {
			limit = {
				trade_goods = damestear

			is_subject_of_type = daimyo_xiaken_vassal
			is_subject_of_type = tributary_state
		}

			estate_loyalty = {
				estate = estate_mages
				loyalty = 60
			}
			NOT = { war_with = ROOT } # not currently at war
			NOT = { is_rival = ROOT } # not an enemy
			NOT = { ROOT = { is_rival = PREV } } # not a rival
			has_opinion = { who = ROOT value = -50 } # not awful relations
			NOT = { alliance_with = ROOT } # not already allied
			NOT = { # target is within 100 distance
				border_distance = {
					who = ROOT
					distance = 100
				}
			}
			OR = { 
				AND = { # Case 1: Country is not in the Raj
					OR = { # target is not a subject or is a tributary (is not in the Raj)
						is_subject = no
						is_subject_of_type = tributary_state
					}
					ROOT = { # Country is not in the Raj
						AND = { 
							NOT = {	is_subject_of_type = great_daimyo_vassal }
							NOT = { is_subject_of_type = daimyo_vassal }
							NOT = { is_subject_of_type = daimyo_xiaken_vassal }
						}
					}
				}
				AND = { # Case 2: Country is in the Raj
					ROOT = { # Country is in the Raj
						OR = {
							is_subject_of_type = great_daimyo_vassal
							is_subject_of_type = daimyo_vassal
							is_subject_of_type = daimyo_xiaken_vassal
						}
					}
					OR = { # target is also in the Raj
						is_subject_of_type = great_daimyo_vassal
						is_subject_of_type = daimyo_vassal
						is_subject_of_type = daimyo_xiaken_vassal
					}
				}
			}
		}
	}
	selection_weight = {
		factor = 1
		modifier = {
			factor = 2
			NOT = { num_of_allies = 1 }
			NOT = { num_of_subjects = 1 }
		}
	}

					estate = estate_mages
					loyalty = 60
				}

				NOT = { is_rival = ROOT }
				NOT = { ROOT = { is_rival = PREV } }
				has_opinion = { who = ROOT value = -50 }
				NOT = { alliance_with = ROOT }
				NOT = {
					border_distance = {
						who = ROOT
						distance = 100
					}
				}
				OR = { 
					AND = {
						OR = {
							is_subject = no
							is_subject_of_type = tributary_state
						}
						ROOT = {
							AND = { 
								NOT = {	is_subject_of_type = great_daimyo_vassal }
								NOT = { is_subject_of_type = daimyo_vassal }
								NOT = { is_subject_of_type = daimyo_xiaken_vassal }
							}
						}
					}
					AND = {
						ROOT = {
							OR = {
								is_subject_of_type = great_daimyo_vassal
								is_subject_of_type = daimyo_vassal
								is_subject_of_type = daimyo_xiaken_vassal
							}
						}

							is_subject_of_type = great_daimyo_vassal

			}
		}

			NOT = { has_estate = estate_mages }
		}
	}

		alliance_with = event_target:agenda_country
	}

		}
