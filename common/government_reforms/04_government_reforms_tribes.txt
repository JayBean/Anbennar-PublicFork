	icon = "protection_of_foreign_traders"
	allow_normal_conversion = yes
	potential = {
		eligible_for_gov_tribal_reform_monster_non_horde = yes
		OR = {
			culture_group = goblin
			culture_group = harpy
			culture_group = kobold
		}
	}
	trigger = {
		NOT = { has_country_modifier = monstrous_nation }
	}
	modifiers = {
		trade_efficiency = 0.1
	}
	effect = {
		custom_tooltip = protection_of_foreign_traders_tt
		hidden_effect = {
			every_owned_province = {
				if = {
					limit = {
						province_has_center_of_trade_of_level = 1
					}
					add_province_modifier = { 
						name = protection_of_foreign_traders_reform_modifier
						duration = -1
					}
				}
			}
		}
	}
	removed_effect = {
		hidden_effect = {
			every_owned_province = {
				limit = {
					has_province_modifier = protection_of_foreign_traders_reform_modifier
				}
				remove_province_modifier = protection_of_foreign_traders_reform_modifier
			}
		}
	}
}
