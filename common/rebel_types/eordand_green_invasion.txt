eordand_green_rebels = {

	color = { 0 128 64 }

	area = nation					# Stays within the nation
	government = any
	defection = none
	independence = none
	gfx_type = culture_province
	
	resilient = yes
	reinforcing = yes
	general = yes
	smart = yes
	unit_transfer = no
	will_relocate = yes
	
	infantry = 0.9
	cavalry = 0.1
	artillery = 0
	
	morale = 1.5
	
	# Possible handle actions
	handle_action_negotiate = no
	handle_action_stability = no
	handle_action_build_core = no
	handle_action_send_missionary = no
	
	# The rebel type with the highest modifier for this province gets picked
	spawn_chance = {
		factor = 0
	}

 	# This is checked for EACH province in the Area of Operations
	movement_evaluation = {
		factor = 1
		modifier = {
			factor = 0.1
			units_in_province = 1
		}
		modifier = {
			factor = 0.001
			controlled_by = eordand_green_rebels
		}
		modifier = {
			factor = 10
			is_capital = yes
		}
		modifier = {
			factor = 1.5
			unrest = 2
		}
		modifier = {
			factor = 1.5
			unrest = 4
		}
		modifier = {
			factor = 1.5
			unrest = 6
		}
	}
 
 	# Province Scope
 	siege_won_trigger = {
		always = yes
 	}
	siege_won_effect = {
		add_local_autonomy = 25
		add_devastation = 25
		add_permanent_province_modifier = {
			name = devasted_province
			duration = 1825
		}
		if = {
			limit = { NOT = { is_core = H99 } }
			add_core = H99
			add_goblin_minority_size_effect = yes
			add_orcish_minority_size_effect = yes
		}
	}
	
	# Country scope
	can_negotiate_trigger = {
		always = no
	}
	
	# Country scope
	can_enforce_trigger = {
		always = yes
	}
	
	# Localisation for their demands
	demands_description = "eordand_green_rebels_demand"
	
	# Country Scope	
	demands_enforced_effect = {
		hidden_effect = {
			H21 = {
				add_country_modifier = {
					name = just_lost_to_rebels
					duration = 10
					hidden = yes
				}
				set_country_flag = eord_disaster_lost
			}
		}
		add_prestige = -10	
		every_owned_province = {
			limit = {
				has_rebel_faction = eordand_green_rebels
			}
			if = {
				limit = { NOT = { is_core = H99 } }
				add_core = H99
			}
			cede_province = H99
		}
		H99 = {
			change_primary_culture = common_goblin
			set_ruler_culture = common_goblin
			set_ruler_religion = great_dookan
			clear_fast_ruler_flags = yes
			set_ruler_flag = goblin_ruler

			set_heir_culture = common_goblin
			set_heir_religion = great_dookan
			clear_fast_heir_flags = yes
			set_heir_flag = goblin_heir

			set_consort_culture = common_goblin
			set_consort_religion = great_dookan
			clear_fast_consort_flags = yes
			set_consort_flag = goblin_consort
		}
		H99 = {
			every_owned_province = {
				limit = {
					has_goblin_minority_trigger = yes
				}
				change_culture = common_goblin
				add_ruinborn_minority_size_effect = yes
				remove_province_modifier = devasted_province
			}
		}
	}
}
