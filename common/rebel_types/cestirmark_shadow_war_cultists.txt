
######################################################################
# Crazed cultists with a few eldritch invocations; should be generic enough to be used elsewhere if needed
######################################################################
shadow_war_cultists = {

	color = { 50 85 210 }

	area = nation 					# Stays within the nation
	government = any	
	defection = none				# Does not defect
	defect_delay = 120				# Held provinces will auto-defect after 18 months, regardless of accept/enforce demands.
	independence = none				# Does not strive for independence
	gfx_type = culture_province
	
	resilient = yes
	reinforcing = yes
	general = yes
	smart = yes
	unit_transfer = yes
	will_relocate = yes
	
	artillery = 0.1
	infantry = 0.7
	cavalry = 0.2
	
	morale = 1.25					# Fanatics with a few eldritch abominations
	
	# Possible handle actions
	handle_action_negotiate = no
	handle_action_stability = no
	handle_action_build_core = no
	handle_action_send_missionary = no		
	handle_action_change_culture = no
	
	# The rebel type with the highest modifier for this province gets picked
	spawn_chance = {
		factor = 0
	}

	# This is checked for EACH province in the Area of Operations
	movement_evaluation = {
		factor = 1
		modifier = {
			factor = 0.1
			units_in_province = 1
		}
		modifier = {
			factor = 0.001
			controlled_by = REB
		}
		modifier = {
			factor = 2
			is_capital = yes
		}
		modifier = {
			factor = 1.5
			OR = {
				has_province_modifier = aw_eldritch_cultists_1
				has_province_modifier = aw_eldritch_cultists_2
				has_province_modifier = aw_eldritch_cultists_3
			}
		}
	}

	# Province Scope
	siege_won_trigger = {
		always = yes
	}
	siege_won_effect = {
		hidden_effect = { 
			add_devastation = 5
		}
		add_permanent_province_modifier = {
			name = devasted_province
			duration = 1095
		}
	}
	
	# Country scope
	can_negotiate_trigger = {
		always = no
	}
	
	# Country scope
	can_enforce_trigger = {
		always = yes
	}

	# Localisation for their demands
	demands_description = "shadow_war_cultists_demand"
	
	# Country Scope	
	demands_enforced_effect = {
		hidden_effect = {
			add_country_modifier = {
				name = just_lost_to_rebels
				duration = 10
				hidden = yes
			}
		}
		add_prestige = -50
		add_stability = -2
		every_owned_province = {
			limit = {
				has_rebel_faction = REB
			}
			add_local_autonomy = 20
		}
		capital_scope = {
			if = {
				limit = { has_province_modifier = aw_eldritch_cultists_1 }
				remove_province_modifier = aw_eldritch_cultists_1
			}
			if = {
				limit = { has_province_modifier = aw_eldritch_cultists_2 }
				remove_province_modifier = aw_eldritch_cultists_2
			}
			add_permanent_province_modifier = { 
				name = aw_eldritch_cultists_3
				duration = -1
			}
		}
	}
}
