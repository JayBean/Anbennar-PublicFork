
######################################################################
# Khugra rebels, spawned during war of colours
######################################################################
red_party = {
	
	color = { 190 55 55 }
	
	area = nation 					# Stays within the nation
	government = any				# Don't care about gov type				
	defection = none				# Will not defect
	independence = none				# Does not strive for independence
	gfx_type = culture_province
	will_relocate = yes	

	resilient = no					# Can retreat
	reinforcing = yes				# Can reinforce
	general = yes					# Does use generals
	smart = yes						# Do assault forts, tactically withdraw from battle, etc.
	unit_transfer = no				# Units do not remain after enforcing demands.
	
	artillery = 0
	infantry = 0.8
	cavalry = 0.2
	
	morale = 1.3
	
	# Possible handle actions
	handle_action_negotiate = no
	handle_action_stability = no
	handle_action_build_core = no
	handle_action_send_missionary = no
	
	# The rebel type with the highest modifier for this province gets picked
	spawn_chance = {
		factor = 0
	}

	# This is checked for EACH province in the Area of Operations
	movement_evaluation = {
		factor = 1
		modifier = {
			factor = 0.1
			units_in_province = 1
		}
		modifier = {
			factor = 0.001
			controlled_by = REB
		}
		modifier = {
			factor = 10
			is_capital = yes
		}
		modifier = {
			factor = 1.5
			unrest = 2
		}
		modifier = {
			factor = 1.5
			unrest = 4
		}
		modifier = {
			factor = 1.5
			unrest = 6
		}
	}
	
	# Province Scope
 	siege_won_trigger = {
 	}
	siege_won_effect = {
	}
	
	# Country scope
	can_negotiate_trigger = {
		OR = {
			is_subject_of = L96
			is_subject_of = L68
		}
	}
	
	# Country scope
	can_enforce_trigger = {
		always = yes
	}

	# Localisation for their demands
	demands_description = red_party_demands
	
	# Country scope
	demands_enforced_effect = {
		if = {
			limit = { 
				NOT = { has_country_flag = khugra_war_of_colours_karassk_rules_red_party }
			}
			L96 = {
				vassalize = L70
			}
		}
		if = {
			limit = { 
				has_country_flag = khugra_war_of_colours_karassk_rules_red_party
			}
			L68 = {
				vassalize = L70
			}
		}
		if = {
			limit = { 
				has_estate_privilege = estate_church_war_of_colours_imperial_green 
				exists = L95
			}
			every_owned_province = {
				limit = {
					is_permanent_claim = L95
				}
				cede_province = L95
			}
			remove_estate_privilege = estate_church_war_of_colours_imperial_green
		}
		if = {
			limit = { 
				has_estate_privilege = estate_burghers_war_of_colours_treacherous_gold 
				exists = L71
			}
			every_owned_province = {
				limit = {
					is_permanent_claim = L71
				}
				cede_province = L71
			}
			remove_estate_privilege = estate_burghers_war_of_colours_treacherous_gold
		}
		set_country_flag = khugra_war_of_colours_culminates
	}
}
