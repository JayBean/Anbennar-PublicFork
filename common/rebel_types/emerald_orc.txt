	color = { 50 150 50 }

	area = nation					# Stays within the nation
	government = any	
	defection = none				# Does not defect
	independence = none				# Does not strive for independence
	gfx_type = culture_province
	
	resilient = yes
	reinforcing = yes
	general = yes
	smart = yes
	unit_transfer = no
	will_relocate = yes
	
	infantry = 0.9
	cavalry = 0.1
	
	morale = 1.5
	
	# Possible handle actions
	handle_action_negotiate = no
	handle_action_stability = no
	handle_action_build_core = no
	handle_action_send_missionary = no
	
	# The rebel type with the highest modifier for this province gets picked
	spawn_chance = {
		factor = 0
		
	}
 
 	# This is checked for EACH province in the Area of Operations
	movement_evaluation = {
		factor = 1
		modifier = {
			factor = 0.1
			units_in_province = 1
		}
		modifier = {
			factor = 0.001
			controlled_by = REB
		}
		modifier = {
			factor = 10
			is_capital = yes
		}
		modifier = {
			factor = 1.5
			unrest = 2
		}
		modifier = {
			factor = 1.5
			unrest = 4
		}
		modifier = {
			factor = 1.5
			unrest = 6
		}
	}
 
 	# Province Scope
 	siege_won_trigger = {
		NOT = {
			has_province_modifier = deepwoods_fey_corruption
		}
 	}
	siege_won_effect = {
		add_local_autonomy = 25
		add_devastation = 25
		add_permanent_province_modifier = {
			name = deepwoods_fey_corruption
			duration = 9125
		}
	}
	
	# Country scope
	can_negotiate_trigger = {
		always = no
	}
	
	# Country scope
	can_enforce_trigger = {
		always = yes
	}
	
	# Localisation for their demands
	demands_description = "emerald_orc_rebel_demand"
	
	# Country Scope	
	demands_enforced_effect = {

			add_country_modifier = {
				name = just_lost_to_rebels
				duration = 10
				hidden = yes
			}
		}
		every_owned_province = {
			limit = {
				has_rebel_faction = REB
			}
			add_devastation = 25
		}
		add_prestige = -50
		reduce_legitimacy_medium_effect = yes
		add_stability = -2
		set_country_flag = green_orc_war_of_the_black_trees_lost
		country_event = { id = deepwoods.27 days = 1 }
	}
}
