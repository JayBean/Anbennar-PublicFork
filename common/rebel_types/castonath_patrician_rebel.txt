	color = { 255 255 255 }

	area = nation 					# Stays within the nation
	government = any	
	defection = none				# Does not defect
	independence = none				# Does not strive for independence
	gfx_type = culture_province
	
	resilient = yes
	reinforcing = yes
	general = yes
	smart = yes
	unit_transfer = no
	will_relocate = yes
	
	artillery = 0
	infantry = 0.8
	cavalry = 0.2
	
	morale = 1.1
	
	# Possible handle actions
	handle_action_negotiate = no
	handle_action_stability = no
	handle_action_build_core = no
	handle_action_send_missionary = no
	
	# The rebel type with the highest modifier for this province gets picked
	spawn_chance = {
		factor = 0
	}

	# This is checked for EACH province in the Area of Operations
	movement_evaluation = {
		factor = 1
		modifier = {
			factor = 0.1
			units_in_province = 1
		}
		modifier = {
			factor = 0.001
			controlled_by = REB
		}
		modifier = {
			factor = 10
			is_capital = yes
		}
		modifier = {
			factor = 1.5
			unrest = 2

		modifier = {
			factor = 1.5
			unrest = 4
		}
		modifier = {
			factor = 1.5
			unrest = 6
		}
	}
 
 	# Province Scope

		always = yes
 	}
	siege_won_effect = {
		add_permanent_province_modifier = {
			name = devasted_province
			duration = 1825
		}
	}
	
	# Country scope
	can_negotiate_trigger = {
		always = no
	}
	
	# Country scope
	can_enforce_trigger = {
		always = yes
	}
	
	# Localisation for their demands
	demands_description = castonath_patrician_rebel_demand
	
	# Country Scope	
	demands_enforced_effect = {
		hidden_effect = {
			add_country_modifier = {
				name = just_lost_to_rebels
				duration = 10
				hidden = yes
			}
		}
		every_owned_province = {
			limit = {
				has_rebel_faction = REB
			}
			if = {
				limit = {
					province_id = 831
				}
				remove_province_modifier = castonath_patrician_influence_some
				remove_province_modifier = castonath_patrician_influence_considerable

				add_permanent_province_modifier = {
					name = castonath_patrician_influence_full
					duration = -1
				}
			}
			else_if = {
				limit = {
					province_id = 832
				}
				remove_province_modifier = castonath_patrician_influence_some
				remove_province_modifier = castonath_patrician_influence_considerable
				remove_province_modifier = castonath_patrician_influence_significant
				add_permanent_province_modifier = {
					name = castonath_patrician_influence_full
					duration = -1
				}
			}
			else_if = {
				limit = {
					province_id = 833
				}
				remove_province_modifier = castonath_patrician_influence_some
				remove_province_modifier = castonath_patrician_influence_considerable
				remove_province_modifier = castonath_patrician_influence_significant
				add_permanent_province_modifier = {
					name = castonath_patrician_influence_full
					duration = -1
				}
			}
			add_devastation = 75
		}	
		add_prestige = -50
		reduce_legitimacy_medium_effect = yes
		add_stability = -2
	}
