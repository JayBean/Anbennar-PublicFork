#Open Artifice Inventions Menu
custom_button = {
	name = artifice_open_inventions
	potential = {
		artifice_invention_has_any_menu_open = no
	}

	trigger = {
		hidden_trigger = {
			has_country_flag = artificery_unlocked
			OR = {
				has_country_flag = magic_artificery_mixed
				has_country_flag = magic_artificery_artificery_only
			}
		}
	}
	
	effect = {
		set_country_flag = artificery_tab_generic_tier_1_open
		hidden_effect = {
			if = {
				limit = {
					OR = {
						artifice_has_cultural_invention = yes
						artifice_has_racial_military_invention = yes
					}
				}
				set_country_flag = has_cultural_special_invention
			}
			if = {
				limit = {
					artifice_has_unique_invention = yes
				}
				set_country_flag = has_unique_invention
			}
		}
	}

	tooltip = artifice_invention_open_button_tt
}

custom_button = {
	name = military_invention_1_1
	tooltip = military_invention_1_1_tt
	potential = { 
		has_country_flag = artificery_tab_generic_tier_1_open
	}
	frame = {
		number = 1
		trigger = {
			NOT = {
				has_country_flag = unlocked_artifice_invention_portable_turrets
			}
		}
	}
	frame = {
		number = 2
		trigger = {
			NOT = {
				has_estate_privilege = artifice_invention_portable_turrets
			}
			has_country_flag = unlocked_artifice_invention_portable_turrets
		}
	}
	frame = {
		number = 3
		trigger = {
			has_estate_privilege = artifice_invention_portable_turrets
		}
	}
	trigger = {
		hidden_trigger = {
			has_country_flag = unlocked_artifice_invention_portable_turrets
		}
	}
	effect = { 
		if = {
			limit = {
				artifice_can_select_invention_10 = yes
				NOT = {
					has_estate_privilege = artifice_invention_portable_turrets
				}
			}
			set_estate_privilege = artifice_invention_portable_turrets
		}
		else_if = {
			limit = {
				artifice_can_select_invention_10 = no
				NOT = {
					has_estate_privilege = artifice_invention_portable_turrets
				}
			}
			custom_tooltip = artifice_cannot_afford_invention
			tooltip = {
				set_estate_privilege = artifice_invention_portable_turrets
			}
		}
        else_if = {
			limit = {
				has_estate_privilege = artifice_invention_portable_turrets
			}
			remove_estate_privilege = artifice_invention_portable_turrets
		}
	}
}

		has_country_flag = unlocked_artifice_invention_synthetic
	}
	frame = {
		number = 1
		trigger = {
			NOT = {
				has_country_flag = unlocked_artifice_invention_synthetic
			}
		}
	}
	frame = {
		number = 2
		trigger = {
			NOT = {
				has_estate_privilege = artifice_invention_snotfinger_synthetic
			}
			has_country_flag = unlocked_artifice_invention_synthetic
		}
	}
	frame = {
		number = 3
		trigger = {
			has_estate_privilege = artifice_invention_snotfinger_synthetic
		}
	}
	trigger = {
		hidden_trigger = {
			has_country_flag = unlocked_artifice_invention_synthetic
		}
	}
	effect = {
		if = {
			limit = {
				has_country_flag = unlocked_artifice_invention_synthetic
			}
			if = {
				limit = {
					artifice_can_select_invention_30 = yes
					NOT = {
						has_estate_privilege = artifice_invention_snotfinger_synthetic
					}
				}
				set_estate_privilege = artifice_invention_snotfinger_synthetic
			}
			else_if = {
				limit = {
					artifice_can_select_invention_30 = no
					NOT = {
						has_estate_privilege = artifice_invention_snotfinger_synthetic
					}
				}
				custom_tooltip = artifice_cannot_afford_invention
				tooltip = {
					set_estate_privilege = artifice_invention_snotfinger_synthetic
				}
			}
			else_if = {
				limit = {
					has_estate_privilege = artifice_invention_snotfinger_synthetic
				}
				remove_estate_privilege = artifice_invention_snotfinger_synthetic
			}
		}
	}
}
