					owned_by = ROOT

					owner = { alliance_with = ROOT }
					owner = { ROOT = { truce_with = PREV } }
					has_terrain = cavern
					has_terrain = dwarven_road
					has_terrain = dwarven_hold
					has_terrain = dwarven_hold_surface
				}
				is_city = yes
				has_discovered = ROOT		
			}
			harpy_raid_effect = yes
			hidden_effect = {
				if = {
					limit = { owner = { NOT = { has_country_flag = informed_of_harpy_raid } } }
					owner = { # Raid and inform the victim.
						set_country_flag = informed_of_harpy_raid
						country_event = { id = the_hunt_events.$event_nr$ days = 1 } 
					}
				}
			}
		}
	}
}

return_all_uncored_provinces = { #Make this also release nations that do not exist
	custom_tooltip = return_all_uncored_provinces_tt
	hidden_effect = {
		[[release_nations]
			every_owned_province = {
				limit = {
					NOT = { is_core = ROOT }
					any_core_country = {
						# exists = no
						NOT = { tag = ROOT }
						PREV = { previous_owner = PREV }
					}
				}
				save_event_target_as = province_to_cede
				cede_province = previous_owner
				owner = {
					change_religion = event_target:province_to_cede
					remove_country_modifier = conversion_zeal
				}
				if = {
					limit = { ROOT = { overlord = { has_government_attribute = is_raja } } }
					owner = {
						random_country = {
							limit = { overlord_of = ROOT }
							vassalize = PREV
						}
					}
				}
			}
		]
		every_owned_province = {
			limit = {
				NOT = { is_core = ROOT }
				any_country = {
					NOT = { tag = ROOT }
					PREV = { previous_owner = PREV }
				}
			}
			random_country = {
				limit = {
					NOT = { tag = ROOT }
					PREV = { previous_owner = PREV }
				}
				PREV = { cede_province = PREV }
			}
		}
	}
}

remove_random_development = { #Unlike add_random_development = -1, this won't attempt to remove development at 1
	random_list = {
		1 = {
			trigger = { base_tax = 2 }
			add_base_tax = -1
		}
		1 = {
			trigger = { base_production = 2 }
			add_base_production = -1
		}
		1 = {
			trigger = { base_manpower = 2 }
			add_base_manpower = -1
		}
	}
}

define_heir_with_minimum_stats = { #This makes it so if your generated heir is below the amount, the stat will be increased to match
	export_to_variable = {
		variable_name = heir_adm_var
		value = heir_adm
		who = ROOT
	}
	set_variable = {
		which = heir_adm_sub
		value = $adm_amount$ 
	}
	export_to_variable = {
		variable_name = heir_dip_var
		value = heir_dip
		who = ROOT
	}
	set_variable = {
		which = heir_dip_sub
		value = $dip_amount$ 
	}
	export_to_variable = {
		variable_name = heir_mil_var
		value = heir_mil
		who = ROOT
	}
	set_variable = {
		which = heir_mil_sub
		value = $mil_amount$ 
	}
	if = {
		limit = {
			check_variable = {
				which = heir_adm_sub
				which = heir_adm_var
			}
		}
		subtract_variable = {
			which = heir_adm_sub
			which = heir_adm_var
		}
		while = {
			limit = {
				check_variable = {
					which = heir_adm_sub
					value = 1
				}
			}
			change_variable = {
				which = heir_adm_sub
				value = -1
			}
			change_heir_adm = 1
		}
	}
	if = {
		limit = {
			check_variable = {
				which = heir_dip_sub
				which = heir_dip_var
			}
		}
		subtract_variable = {
			which = heir_dip_sub
			which = heir_dip_var
		}
		while = {
			limit = {
				check_variable = {
					which = heir_dip_sub
					value = 1
				}
			}
			change_variable = {
				which = heir_dip_sub
				value = -1
			}
			change_heir_dip = 1
		}
	}
	if = {
		limit = {
			check_variable = {
				which = heir_mil_sub
				which = heir_mil_var
			}
		}
		subtract_variable = {
			which = heir_mil_sub
			which = heir_mil_var
		}
		while = {
			limit = {
				check_variable = {
					which = heir_mil_sub
					value = 1
				}
			}
			change_variable = {
				which = heir_mil_sub
				value = -1
			}
			change_heir_mil = 1
		}
	}
}

		if = {
			limit = {
				OR = {
					accepted_culture = $culture$
					primary_culture = $culture$
				}
			}
			add_dip_power = 100
		}
	}
	add_accepted_culture_or_dip_power = { culture = $culture$ [[dip_reward] dip_reward = $dip_reward$] [[free] free = $free$] }
	hidden_effect = {
		if = {
			limit = { NOT = { accepted_culture = $culture$ } }
			add_dip_power = 100
		}
	}
	[[dip_reward]
		custom_tooltip = ADD_ACCEPTED_CULTURE_SLOT_LIMITATION_TT #If the YCulture! is already an YAccepted Culture!, or if there are not enough YSlots!, gain G100! Diplomatic Power instead.
	]
}
