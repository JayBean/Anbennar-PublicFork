racial_tolerance_define_ruler = {
	if = {
		limit = { has_country_flag = racial_election_female_equality_flag }
		define_ruler = {
			mil = 2
			adm = 2
			dip = 2
			culture = $culture$
			random_gender = yes
		}
		
	}
	else = {
		define_ruler = {
			mil = 2
			adm = 2
			dip = 2
			culture = $culture$
		}
	}
}

#Half-Elves
get_ruler_from_majority_or_minority_half_elf = {
	if = {
		limit = { has_country_modifier = half_elven_administration }
		racial_tolerance_define_ruler = { culture = ROOT }
	}
	else_if = {
		limit = {
			any_owned_province = {
				has_half_elven_majority_trigger = yes
			}
		}
		hidden_effect = {
			random_owned_province = {
				limit = { has_half_elven_majority_trigger = yes }
				save_event_target_as = half_elf_majority_province
			}
		}
		ROOT = { racial_tolerance_define_ruler = { culture = event_target:half_elf_majority_province } }
	}
	else = {
		random_owned_province = {
			limit = { has_integrated_half_elven_minority_trigger = yes }
			if = {
				limit = { 
					superregion = bulwar_superregion
				}
				ROOT = { racial_tolerance_define_ruler = { culture = bulwari_half_elf } }
			}
			else = {
				ROOT = { racial_tolerance_define_ruler = { culture = half_elf } }
			}
		}
	}
	
}
