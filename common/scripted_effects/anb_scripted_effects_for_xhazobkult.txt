xhazobkult_terrain_razing = {
	if = {
		limit = {
			OR = {
				has_terrain = city_terrain
				province_group = hold_province
			}
		}
		owner = { add_patriarch_authority = 0.05 }
	}
	else_if = {
		limit = {
			has_terrain = farmlands
		}
		owner = { add_patriarch_authority = 0.03 }
	}
	else_if = {
		limit = {
			OR = {
				has_terrain = grasslands
				has_terrain = hills
				has_terrain = woods
				has_terrain = drylands
			}	
		}
		owner = { add_patriarch_authority = 0.025 }
	}
	else_if = {
		limit = {
			OR = {
				has_terrain = mountain
				has_terrain = highlands
				has_terrain = steppe
				has_terrain = forest
				has_terrain = ancient_forest
				has_terrain = jungle
			}	
		}
		owner = { add_patriarch_authority = 0.02 }
	}
	else = {
		owner = { add_patriarch_authority = 0.01 }
	}
}
