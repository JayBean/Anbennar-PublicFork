					has_building = university
					has_production_building_trigger = yes

				limit = {

						name = feiten_shuimu_engineering_workshop_no_tech_cost

					}
				}
			}
			else = {
				tooltip = {
					add_province_modifier = {
						name = feiten_shuimu_engineering_workshop
						duration = -1
					}
				}
			}
		}
	}

	benefits = { }

		random_owned_province = {
			limit = { has_province_flag = shuimu_workshop_build_location }
			clr_province_flag = shuimu_workshop_build_location
			if = {
				limit = {
					event_target:feiten_or_was_feiten_target = {
						check_variable = {
							which = feiten_number_of_shuimu_engineering_cadre_facilities_investments
							value = 5
						}
					}
				}
				add_permanent_province_modifier = {
					name = feiten_shuimu_engineering_workshop_no_tech_cost
					duration = -1
				}
			}
			else = {
				add_permanent_province_modifier = {
					name = feiten_shuimu_engineering_workshop
					duration = -1
				}
			}
