artifice_invention_research_privilege = {
	icon = privilege_Artf_Research
	max_absolutism = 0
	loyalty = 0.05
	influence = 0 #artificer privileges must never affect influence due to the fact influence affects points
	is_valid = {
		NOT = { has_country_flag = remove_artifice_invention_research }
		OR = {
			NOT = { has_estate_privilege = estate_artificers_organization_state }
			has_estate_privilege = artifice_invention_research_privilege
		}
	}
	can_select = {
		artifice_has_points_for_basic = yes
		is_year = 1445
		NOT = {
			has_estate_privilege = artifice_invention_directed_research
		}
	}
	can_revoke = {
		always = no
	}
	on_granted = {
		artifice_granted_basic = yes
		country_event = {
			id = artifice_events.1
		}
	}
	on_revoked = {
		artifice_remove_basic = yes
		#update_artifice_points = yes #is included in the above effect
		#custom_tooltip = decreases_our_artifice_upkeep_by_10
	}
	penalties = {
		prestige_decay = 0.02
		global_tax_modifier = -0.1
	}
	benefits = {
		#foo
	}
	cooldown_years = 10
	on_cooldown_expires = {
		country_event = {
			id = artifice_events.2
		}
	}
	on_invalid = {
		artifice_remove_basic = yes
		clr_country_flag = remove_artifice_invention_research
	}
	ai_will_do = {
		factor = 100 #ai shouldn't use this one. Just too much work getting them to understand it and making it not crash for linux. Their research will be done via decision? Doesnt look like it -Jay
		modifier = {
			factor = 0
			over_artifice_capacity = yes
		}
	}
}
