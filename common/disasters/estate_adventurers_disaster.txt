
estate_adventurers_disaster = {
	potential = {
		has_estate = estate_adventurers
		estate_influence = {
			estate = estate_adventurers
			influence = 80
		}
	}


	can_start = {
		has_any_disaster = no
		estate_influence = {
			estate = estate_adventurers
			influence = 100
		}
		is_at_war = no
	}
	
	can_stop = {
		OR = {
			has_any_disaster = yes
			NOT = {
				estate_influence = {
					estate = estate_adventurers
					influence = 100
				}		
			}
		}
	}
	
	down_progress = {
		factor = 0
		modifier = {
			factor = 1
			NOT = {
				estate_influence = {
					estate = estate_adventurers
					influence = 100
				}
			}
		}
	}
	
	progress = {
		modifier = {
			factor = 0.5
			NOT = {
				estate_loyalty = {
					estate = estate_adventurers
					loyalty = 50
				}
			}
			is_at_war = no
		}
		modifier = {
			factor = 0.5
			NOT = {
				estate_loyalty = {
					estate = estate_adventurers
					loyalty = 40
				}
			}
			is_at_war = no
		}
		modifier = {
			factor = 1
			NOT = {
				estate_loyalty = {
					estate = estate_adventurers
					loyalty = 30
				}
			}
			is_at_war = no
		}
		modifier = {
			factor = 1
			NOT = {
				estate_loyalty = {
					estate = estate_adventurers
					loyalty = 20
				}
			}
			is_at_war = no
		}
	}
	
	can_end = {
		custom_trigger_tooltip = {
			tooltip = EST_CRUSHED_ADVENTURERS
			NOT = { has_country_flag = estate_adventurers_in_power }
		}
	}
	
	modifier = {	
		global_manpower_modifier = -0.20
		land_forcelimit_modifier = -0.20
		global_tax_modifier = -0.20
		prestige = -0.5
	}

	on_start = estate_disasters.13
	on_end = estate_disasters.14
	
	on_monthly = {
	}
}
