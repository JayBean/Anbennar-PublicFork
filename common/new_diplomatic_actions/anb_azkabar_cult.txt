azkabar_cult_minor_action = {
	category = covert

	is_visible = {
		ai = no
		has_country_flag = F35_minor_eclipse_1_enabled
	}

	is_allowed = {
		ROOT = {
			has_spy_network_in = {
				who = FROM
				value = 40
			}
			if = {
				limit = { has_country_flag = F35_minor_eclipse_3_enabled }
				treasury = 100
			}
			else_if = {
				limit = { has_country_flag = F35_minor_eclipse_2_enabled }
				treasury = 50
			}
		}
		FROM = {
		}
	}

	on_accept = {
		ROOT = {
			add_spy_network_in = {
				who = FROM
				value = -40
			}
			if = {
				limit = { has_country_flag = F35_minor_eclipse_3_enabled }
				add_treasury = -100
			}
			else_if = {
				limit = { has_country_flag = F35_minor_eclipse_2_enabled }
				add_treasury = -50
			}
		}
		FROM = {
			country_event = { id = flavour_azkabar.101 }
			if = {
				limit = {
					check_variable = {
						which = minor_eclipse_operation_count
						value = 1
					}
				}
				change_variable = {
					which = minor_eclipse_operation_count
					value = 1
				}
			}
			else = {
				set_variable = {
					which = minor_eclipse_operation_count
					value = 1
				}
			}
		}
	}
	ai_will_do = {
		always = no
	}
}
