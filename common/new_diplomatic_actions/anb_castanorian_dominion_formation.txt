cancel_castanorian_dominion =  {
	category = influence
	require_acceptance = no
	is_visible = {
		overlord_of = FROM
		FROM = {
			is_subject_of_type = castanor_dominion
		}
	}
	
	is_allowed = {
		hidden_trigger = {
			overlord_of = FROM
			FROM = {
				is_subject_of_type = castanor_dominion
			}
		}
		FROM = {
			NOT = { liberty_desire = 50 }
		}
		reverse_has_opinion = {
			who = FROM
			value = 0
		}
		FROM = {
			is_subject_of_type = castanor_dominion
			is_subject_of = ROOT		
		}
		stability = 0
	}
	
	on_accept = {
		add_trust = {
			who = FROM
			value = -20
			mutual = yes
		}
		create_subject = {
			subject_type = vassal
			subject = FROM  # dk check
		}
		FROM = {
			add_liberty_desire = 65
		}
		add_stability = -1

	}
	
	ai_will_do = {
		always = no
	}
}
