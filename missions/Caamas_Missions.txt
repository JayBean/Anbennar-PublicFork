					any_province_in_state = {
						NOT = { province_has_center_of_trade_of_level = 1 }
						base_production = 6
						base_manpower = 4
					}
				}
			}
		}
		
		effect = {
			if = {
				limit = {

					}

					random_province_in_state = {
						limit = {
							province_has_center_of_trade_of_level = 1
							base_production = 6

							name = "caamas_trade_secured"
							duration = 7300
						}
					}
				}
			}
			else = {
				capital_scope = {
					random_province_in_state = {
						limit = {
