				}
				AND = {

							is_subject_of_type = great_daimyo_vassal
						}

					num_of_allies = 2

						owner ={
							has_spy_network_from = {
								who = ROOT
								value = 20
							}
						}
					}				
				}
			}
		}

		effect = {
			4362 = {
				add_permanent_claim = ROOT
			}
			4361 = {
				add_permanent_claim = ROOT
			}
			4363 = {
				add_permanent_claim = ROOT
			}
			if = {
				limit = {
					is_subject = yes
				}
				overlord = {
					ROOT = {
						add_favors = {
							who = PREV
							amount = 10
						}
					}	
				}
			}
			else = {
				#Find out who has the most development within scope.
				every_ally = {
					hidden_effect = {
						ROOT = {
							export_to_variable = {
								which = dev_new
								value = total_development 
								who = THIS
							}
						}
						if = {
							limit = {
								NOT = {
									has_saved_event_target = highestdev
								}
							}
							save_event_target_as = highestdev
							ROOT = {
								set_variable = {
									which = dev_old
									which = total_development 
								}
							}
						}
						else_if= {
							limit = {
								#checks if new dev value is higher than old dev valus 
								check_variable = {
									which = dev_new
									which = dev_old
								}
							}
							ROOT = {
								set_variable = {
									which = dev_old
									which = dev_new
								}
							}
							save_event_target_as = highestdev
						}
					}	
				}
				ROOT = {
					add_favors = {
						who = event_target:highestdev
						amount = 10
					}
				}
			}
		}
	}
