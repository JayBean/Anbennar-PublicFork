		icon = mission_fortify_rajputana
		position = 3
		required_missions = {
			cave_goblin_knives_in_the_dark
			g_our_hold
		}
		trigger = {
			any_owned_province = {
                AND = {
					OR = {
						has_terrain = dwarven_hold
						has_terrain = dwarven_hold_surface
					}
					OR = {
						culture_group = dwarven
						culture_group = orcish
					}
                }
            }
		}
		
		effect = {
			any_target_province = {
				add_base_manpower = -1
				add_base_production = -1
				add_base_tax = -1
			}
			capital_scope = {
				add_base_manpower = 1
				add_base_production = 1
				add_base_tax = 1
			}
		}
	}

			g_plunder_the_enemy_fortress

						has_terrain = dwarven_hold
						has_terrain = dwarven_hold_surface
					}
				}
				amount = 4
			}
		}
		
		effect = {
			if = {
				limit = {
					is_lacking_institutions = yes
				}
				capital_scope = {
					add_next_institution_embracement = 25
				}
			}
			else_if = {
				limit = {
					NOT = { has_dlc = "Rule Brittania" }
				}

			}
			else = {
				change_innovativeness = 10

                OR = {

			add_adm_power = 100

			}
		}
	}

	g_our_hold = {
		icon = mission_fortify_rajputana
		position = 1
		required_missions = {
			g_conquest_of_a_hold
		}
		trigger = {
			any_owned_province = {
				OR = {
					has_terrain = dwarven_hold
					has_terrain = dwarven_hold_surface
				}
				is_core = FROM
				NOT = {
					has_province_modifier = ruined_hold
					has_province_modifier = partially_ruined_hold
					has_province_modifier = damaged_hold
				}
			}
		}
		
		effect = {
			add_years_of_income = 0.25
			add_adm_power = 50
			random_owned_province = {
				limit = {
					OR = {
						has_terrain = dwarven_hold
						has_terrain = dwarven_hold_surface
					}
					is_core = FROM
					NOT = {
						has_province_modifier = ruined_hold
						has_province_modifier = partially_ruined_hold
						has_province_modifier = damaged_hold
					}
				} 
				owner = { 
					set_capital = PREV
				}
				change_culture = cave_goblin
				change_religion = goblinic_shamanism
			}
		}
	}
