			}
			custom_trigger_tooltip = {
				tooltip = azkare_citadel_tooltip
			employed_advisor = {
				type = fortification_expert
				}
			}
		}
		
		effect = {
			if = {  #If elf advisor, also gain 2 military development, otherwise just gain improved modifier
				limit = { 
					employed_advisor = {
						culture = sunrise_elf
						type = fortification_expert
					}
				}
					4789 = { #Azkare
						remove_province_modifier = azkare_modifier_citadel_of_dawn_azkare
						add_province_modifier = {
							name = azkare_modifier_citadel_of_dawn_azkare_2
							duration = -1
						}
						add_base_manpower = 2
					}
			}
			else = {
				4789 = { #Azkare
					remove_province_modifier = azkare_modifier_citadel_of_dawn_azkare
					add_province_modifier = {
						name = azkare_modifier_citadel_of_dawn_azkare_2
						duration = -1
					}
				}
			}
		}
