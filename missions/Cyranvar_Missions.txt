	slot = 3
	generic = no
	ai = yes
	has_country_shield = yes
	
	potential = {
		tag = I45
	}
	
	I45_demand_the_oak_oath = {
		icon = mission_unite_the_princes
		position = 1
		
		trigger = {
			has_legitimacy_equivalent = { VAL=50 }
			dip_power = 200
			NOT = { has_ruler_modifier = lich_ruler }
		}
		
		effect = {
			if = {
				limit = {
					any_known_country = {
						primary_culture = wood_elf 
						capital_scope = { superregion = deepwoods_superregion } 
						NOT = { tag = I45 } 
						NOT = { is_subject = yes }
					}
				}
				add_dip_power = -200
				every_known_country = {
					limit = { 
						primary_culture = wood_elf 
						capital_scope = { superregion = deepwoods_superregion } 
						NOT = { tag = I45 } 
						NOT = { is_subject = yes }
					}
					country_event = { id = cyranvar.4 days = 20 random = 10 }
				}
			}
			else = {
				add_stability_or_adm_power = yes
			}
		}
	}
	
	I45_a_capital_city = {
		icon = mission_conquer_novgorod
		position = 3
		
		required_missions = {
			I45_demand_the_oak_oath
		}
		
		provinces_to_highlight = {
			is_capital_of = ROOT
			NOT = { development = 20 }
		}
		
		trigger = {
			if = {
				limit = { ai = no }
				capital_scope = {
					development = 20
				}
			}
			else = {
				always = yes
			}
		}
		
		effect = {
			capital_scope = {
				if = {
					limit = { province_has_center_of_trade_of_level = 3 }
					add_base_manpower = 3
					add_base_production = 3
					add_base_tax = 3
				}
				else_if = {
					limit = { province_has_center_of_trade_of_level = 1 }
					add_center_of_trade_level = 1
				}
				else = {
					center_of_trade = 1
				}
			}
		}
	}
	
	I45_toward_the_future = {
