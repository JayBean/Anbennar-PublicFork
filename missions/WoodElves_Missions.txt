	}
	
	WOOD_the_verdant_pact = {

		position = 1
		
		trigger = {
			any_ally = {
				has_opinion = {
					who = ROOT
					value = 150
				}
			}
		}
		
		effect = {
			add_splendor = 200
			add_dip_power = 150
		}
	}
	
	WOOD_settle_the_refugees = {
		icon = mission_alliances
		position = 2
		required_missions = {

		}
		
		trigger = {
			treasury = 150
			stability = 0
		}
		
		effect = {
			add_treasury = -150
			if = {
				limit = { stability = 3 }
				capital_scope = {
					add_base_manpower = 1
					add_base_tax = 1
					add_base_production = 1
				}
			}
			else = {
				add_stability_or_adm_power = yes
			}
			capital_scope = {
				add_province_modifier = {
					name = deepwoods_settled_refugee
					duration = 18250
				}
			}
		}
	}
	
	WOOD_invest_in_capital = {
		icon = mission_early_game_buildings

		}

			stability = 3
			adm_power = 150
		}
		
		effect = {
			add_adm_power = -150
			capital_scope = {
				add_base_tax = 1
				add_base_production = 1
				add_base_manpower = 1
			}
		}
	}
	
	WOOD_prosperous_capital = {
		icon = mission_early_modern_university
		position = 5
		required_missions = {
			WOOD_invest_in_capital
		}
		
		trigger = {
			if = {
				limit = { has_dlc = "Mandate of Heaven" }

					tooltip = prosperous_capital_tooltip

			}

deepwoods_3_b = {
	slot = 3
	generic = no
	ai = yes

	potential = {
		primary_culture = wood_elf
		has_global_flag = end_verdant_pact
		NOT = { tag = I45 }
	}
	
	WOOD_end_of_the_green_menace = {
		icon = mission_steppe_warriors
		position = 4
		required_missions = {
			WOOD_retaking_our_land
		}
		
		trigger = {
			if = {
				limit = { ai = no }
				stability = 3
				religious_unity = 1
				prestige = 0
			}
			else = {
				stability = 1
			}
		}
		
		effect = {
			add_prestige = 20
			set_country_flag = permanent_safe_wood
			add_country_modifier = {
				name = deepwoods_safe_wood
				duration = -1
			}
		}
	}
	
	WOOD_unite_the_deepwoods = {
		icon = mission_monarch_in_throne_room
		position = 5
		required_missions = {
			WOOD_end_of_the_green_menace
		}
		
		trigger = {
			if = {
				limit = { ai = no }
				legitimacy = 95
				stability = 3
			}
			else = {
				stability = 1
			}
			is_subject = no
		}
		
		effect = {
			if = {
				limit = { ai = yes }
				add_stability = 2
			}
			deepwoods_superregion = {
				limit = {
					NOT = { is_core = ROOT }
					NOT = { is_permanent_claim = ROOT }
					NOT = { is_claim = ROOT }
					NOT = { owner = { is_subject_of = ROOT } }
				}
				add_permanent_claim = ROOT
			}
			custom_tooltip = unite_deepwoods_tooltip
			hidden_effect = {
				every_country = {
					limit = { 
						primary_culture = wood_elf
						capital_scope = { superregion = deepwoods_superregion }
						is_subject_of = ROOT
					}
					every_owned_province = {
						cede_province = ROOT
						add_core = ROOT
						change_culture = wood_elf
					}
				}
			}
		}
	}
}
