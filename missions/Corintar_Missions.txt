	corintar_aid_from_anbennar = {
		icon = mission_have_two_subjects
		position = 5
		required_missions = { corintar_envoy_to_the_borders }
		
		provinces_to_highlight = {
			owned_by = emperor
		}

		trigger = {
			# Case 1: the deluge has begun and the emperor is regent court -> skip mission
			if = {
				limit = {
					hre_size = 1
					is_religion_enabled = corinite
					emperor = { religion = regent_court }
					has_global_flag = crimson_deluge
				}
				prestige = 25
			}
			# Case 2: the emperor is regent court but corinite is enabled -> reduced emperor relations requirement
			if = {
				limit = {
					hre_size = 1
					is_religion_enabled = corinite
					emperor = {
						religion = regent_court
					}
				}
				has_spy_network_in = {
					who = emperor
					value = 50
				}
				OR = {
					emperor = {
						has_opinion = {
							who = ROOT
							value = 100
						}
					}
					calc_true_if = {
						all_elector = {
							has_opinion = {
								who = ROOT
								value = 150
							}
						}
						amount = 3
					}
				}
			}
			# Case 3: corinite isn't enabled -> full relations requirements
			else_if = {
				limit = { hre_size = 1 }
				has_spy_network_in = {
					who = emperor
					value = 50
				}
				OR = {
					emperor = {
						has_opinion = {
							who = ROOT
							value = 150
						}
					}
					calc_true_if = {
						all_elector = {
							has_opinion = {
								who = ROOT
								value = 150
							}
						}
						amount = 3
					}
				}
			}
			# Case 4: the empire doesn't exist -> skip mission
			else = {
				prestige = 25
			}
		}
		
		effect = {
			if = {
				limit = {
					hre_size = 1 
					NOT = {
						AND = {
							emperor = { religion = regent_court }
							has_global_flag = crimson_deluge
						}
					}
				}
				add_treasury = 400
			}
			else = {
				add_treasury = 100
			}
			if = {
				limit = {
					calc_true_if = {
						all_elector = {
							has_opinion = {
								who = ROOT
								value = 150
							}
						}
						amount = 3
					}
				}
				every_elector = {
					limit = {
						has_opinion = {
							who = ROOT
							value = 150
						}
					}
					add_trust = {
						who = ROOT
						value = 10
						mutual = yes
					}
					add_country_modifier = {
						name = corintar_corintar_chapter
						duration = -1  
					}
					hidden_effect = {
						set_country_flag = corintar_chapter
						#country_event = { id = flavor_corintar.17 }
					}
				}
				hidden_effect = {
					#country_event = { id = flavor_corintar.17 }
				}
			}
			else = {
				emperor = {
					add_trust = {
						who = ROOT
						value = 10
						mutual = yes
					}
					add_country_modifier = {
						name = corintar_corintar_chapter
