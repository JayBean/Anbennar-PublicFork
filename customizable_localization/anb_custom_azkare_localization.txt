defined_text = {
	name = GetAzkareRepresentation
	random = no
	text = {
		localisation_key = AzkareNotPresent
		trigger = {
			is_variable_equal = { which = azkareCultureDev value = 0 }
		}
	}
	text = {
		localisation_key = AzkareUnrepresented
		trigger = {
			check_variable = { which = azkareCultureDev value = 1 }
			NOT = { check_variable = { which = azkareRepresentation which = azkareMajorlyUnrep } }
		}
	}
	text = {
		localisation_key = AzkareMajorlyUnrepresented
		trigger = {
			check_variable = { which = azkareRepresentation which = azkareMajorlyUnrep }
			NOT = { check_variable = { which = azkareRepresentation which = azkareMinorlyUnrep } }
		}
	}
	text = {
		localisation_key = AzkareMinorlyUnrepresented
		trigger = {
			check_variable = { which = azkareRepresentation which = azkareMinorlyUnrep }
			NOT = { check_variable = { which = azkareRepresentation which = azkareAcceptableRep } }
		}
	}
	text = {
		localisation_key = AzkareRepresented
		trigger = {
			check_variable = { which = azkareRepresentation which = azkareAcceptableRep }
			NOT = { has_province_flag = azkare_most_represented }
		}
	}
	text = {
		localisation_key = AzkareMostRepresented
		trigger = {
			check_variable = { which = azkareRepresentation which = azkareAcceptableRep }
			has_province_flag = azkare_most_represented
		}
	}
}
