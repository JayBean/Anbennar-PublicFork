province_event = {
	id = flavour_arawkelin.11
	title = flavour_arawkelin.11.t
	desc = flavour_arawkelin.11.d

	picture = SCIENTISTS_eventPicture

	is_triggered_only = yes
	trigger = { always = yes }

	option = {
		name = flavour_arawkelin.11.a
		ai_chance = { factor = 1 }
		goto = ROOT
		#halving event
		custom_tooltip = dev_halving_tt
		hidden_effect = {
			if = {
				limit = {
					ROOT = {
						base_production = 2
					}
				}
				export_to_variable = {
					which = dev47661
					value = base_production
					who = ROOT
				}
				export_to_variable = {
					which = dev47662
					value = base_production
					who = ROOT
				}
				divide_variable = {
					which = dev47661
					value = 2
				}
				while = {
					limit = {
						NOT = {
							check_variable = {
								which = dev47661
								which = dev47662
							}
						}
					}
					ROOT = {
						add_base_production = -1
					}
					change_variable = {
						which = dev47662
						value = -1
					}
				}
			}
       		else = {
       			ROOT = {
       				add_devastation = 10
       			}
       		}
		}
	}
}
