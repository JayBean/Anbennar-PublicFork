	id = hoardcurse.1
	title = hoardcurse.1.t
	desc = hoardcurse.1.d
	picture = ARTWORK_eventPicture
	
	mean_time_to_happen = {
		years = 20

	}
	
	trigger = {
		culture_group = dwarven
		OR = {
			treasury = 5000
			monthly_income = 100
		}
		ruler_is_foreigner = no
		NOT = { has_country_flag = hoardcurse_happened }
		NOT = { ruler_has_personality = greedy_personality }
		num_of_owned_provinces_with = {
			value = 10
			OR = {
				has_terrain = dwarven_hold
				has_terrain = dwarven_hold_surface
			}
		}
	}
	
	immediate = {
		hidden_effect = {
			set_ruler_flag = greedy_personality_flag
			if = {
				limit = { ruler_has_personality = mage_personality }
				set_ruler_flag = mage_personality_flag
				clear_ruler_personalities = yes
			}
			apply_saved_ruler_personalities = yes
		}
	}
	
	option = {
        name = hoardcurse.1.a
        ai_chance = { factor = 100 }
		tooltip = { add_ruler_personality = greedy_personality }
	}
}
