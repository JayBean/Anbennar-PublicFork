	id = ynn_debate.249
	title = ynn_debate.249.t
	desc = ynn_debate.249.d
	picture = ACCUSATION_eventPicture
	
	is_triggered_only = yes
	trigger = {
		always = yes
	}
	
	option = {
		name = ynn_debate.249.a
		ai_chance = { factor = 100 }
		add_prestige = 10
		tooltip = { add_liberty_desire = -5 }
		if = {
			limit = {
				event_target:iosahar_a = {
					tag = ROOT
				}
			}
			add_opinion = {
				who = event_target:iosahar_b
				modifier = ynn_iosahar_infighting
			}
		}
		else = {
			add_opinion = {
				who = event_target:iosahar_a
				modifier = ynn_iosahar_infighting
			}
		}
		event_target:iosahar_a = {
			add_trust = {
				who = event_target:iosahar_b
				value = -5
				mutual = yes
			}
		}
	}
}

#replaces religious_reforms.4 for those without El Dorado
country_event = {
	id = ynn_debate.300
	title = ynn_debate.300.t
	desc = ynn_debate.300.d
	picture = ANGRY_MOB_eventPicture

	is_triggered_only = yes
	
	trigger = {
		NOT = { has_dlc = "El Dorado" }
		religion = ynn_river_worship
	}

	hidden = yes
	
	mean_time_to_happen = {
		days = 1
	}
	
	option = {
		name = ynn_debate.300.a
		if = {
			limit = { has_country_flag = yrw_debate_cheat }
		}
		else = {
			if = {
				limit = {
					OR = {
						has_country_flag = yrw_1a_reform
						has_country_flag = yrw_1b_reform
					}
					OR = {
						has_country_flag = yrw_2a_reform
						has_country_flag = yrw_2b_reform
					}
					OR = {
						has_country_flag = yrw_3a_reform
						has_country_flag = yrw_3b_reform
					}
					OR = {
						has_country_flag = yrw_4a_reform
						has_country_flag = yrw_4b_reform
					}
				}
				country_event = { id = ynn_debate.2 }
				every_subject_country = {
					limit = {
						is_subject_of_type = ynnic_iosahar
					}
					country_event = { id = ynn_debate.3 }
				}
			}
			else = {
				country_event = { id = ynn_debate.10 } #debate event
			}
		}
	}
}
