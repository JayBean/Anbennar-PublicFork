			limit = { 4361 = { NOT = { has_province_modifier = seghdihr_sgc_patrols } } } #khinana

		}

			hidden_effect = { set_country_flag = sgc_westgulf_mil_complete }
		}

		}
		ai_chance = { factor = 5 }
	}
