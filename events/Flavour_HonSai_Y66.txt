		has_country_flag = honsai_ghostemperor_alive
		OR = {
			tag = Y66
			was_tag = Y66
		}
	}
	
	option = { 
		name = "flavor_honsai.1.o"
		if = {
			limit = {
				has_church_aspect = honsai_the_ghostemperor 
			}
			subtract_variable = {
				which = honsai_lifespan
				value = 0.66
			}
			set_variable = { 
				which = honsai_lifespan_tooltip 
				which = honsai_lifespan
			}
			multiply_variable = {            
				which = honsai_lifespan_tooltip
				value = 1.5
			}
		}
		else = {
			subtract_variable = {
				which = honsai_lifespan
				value = 1
			}
			set_variable = { 
				which = honsai_lifespan_tooltip 
				which = honsai_lifespan
			}
		}
		if = {
			limit = {
				NOT = {
					check_variable = {
						which = honsai_lifespan
						value = 0.1
					}
				}
			}
			country_event = {
				id = flavor_honsai.4
			}
		}
		else_if = {
			limit = {
				NOT = {
					check_variable = {
						which = honsai_lifespan
						value = 3
					}
				}
				check_variable = {
					which = honsai_lifespan
					value = 2
				}
			}  
			country_event = {
				id = flavor_honsai.66
			}
			
			country_event = {
				id = flavor_honsai.3
				days = 365
			}
		}
		else = {
			country_event = {
				id = flavor_honsai.3
				days = 365
			}
		}		
	}
}
