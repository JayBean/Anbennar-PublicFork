#Resettling Far Bulwar
country_event = {
    id =  bulwar_flavour.0
    title = bulwar_flavour.0.t
    desc = bulwar_flavour.0.d
    picture = COLONIZATION_eventPicture
    
    is_triggered_only = yes
    
    hidden = yes
    
    trigger = {
        owns = 601    #province id doesn't matter, it just to make is so the event fire only once per yearly pulse
        NOT = { is_year = 1500 }
		bulwar_superregion = {
			OR = {
				has_province_modifier = surani_refugee_large
				has_province_modifier = surani_refugee_medium
				has_province_modifier = surani_refugee_small
			}
		}
    }
    
    option = {
        name = bulwar_flavour.0.a
        far_bulwar_region = {
            limit = {
                NOT = {
					area = xhurgrum_area
                    area = sareyand_area
                    area = garpix_zokka_area
                    area = avamezan_area
					area = eastern_krahway_area
                }
                OR = { 
					culture_group = gnollish
					AND = { 
						has_province_flag = bulwar_accepted_refugees
						culture_group = bulwari
					}
				}
                owner = {
                    NOT = { has_country_modifier = gnollish_administration }
                    OR = {        #so if you're not monstrous or semi monstrous this event can fire for you
                        NOT = { has_country_modifier = monstrous_nation }
                        has_country_flag = semi_monstrous
                    }
                }
                NOT = { has_province_flag = rejected_surani_settlers }
            }
			if = { 
				limit = { has_province_flag = bulwar_accepted_refugees }
				province_event = { id = bulwar_flavour.100 days = 180 random = 180 }	#Same event as .1, but hidden and without the reject option.
			}
			else = {
				province_event = { id = bulwar_flavour.1 days = 180 random = 180 }		#Option to accept the refugees or reject them.
			}
        }
    }
}

#Resettling Far Bulwar
province_event = {
    id =  bulwar_flavour.1
    title = bulwar_flavour.1.t
    desc = bulwar_flavour.1.d
    picture = COLONIZATION_eventPicture
    goto = ROOT
    
    is_triggered_only = yes
    
    trigger = {
        NOT = { has_province_flag = rejected_surani_settlers }
        owner = {
            NOT = { has_country_modifier = gnollish_administration }
            OR = {        #so if you're not monstrous or semi monstrous this event can fire for you
                NOT = { has_country_modifier = monstrous_nation }
                has_country_flag = semi_monstrous
            }
        }
        or = { 
			culture_group = gnollish
			and = { 
				has_province_flag = bulwar_accepted_refugees
				culture_group = bulwari
			}
		}
    }

    option = {
        name = bulwar_flavour.1.a
        ai_chance = { factor = 1 }
		set_province_flag = bulwar_accepted_refugees
		custom_tooltip = accepted_surani_refugees_tooltip
		hidden_effect = {
			random_list = {
					33 = { add_base_tax = 1 }
					33 = { add_base_production = 1 }
					33 = { add_base_manpower = 1 }
			}
			if = {
				limit = { 
					NOT = { culture_group = bulwari } 
					has_large_human_minority_trigger = no
				}
				add_human_minority_size_effect = yes
				add_devastation = 1
			}
			else_if = {
				limit = {
					NOT = { culture_group = bulwari }
					has_large_human_minority_trigger = yes
				}
				if = { 
					limit = { area = jikarzax_area }
					change_culture = masnsih
					add_core = F79
				}
				else = {
					change_culture = surani
				}
				add_province_modifier = { name = add_unrest_5_modifier duration = 730 }
				add_devastation = 10
				if = {
					limit = { owner = { religion = the_jadd } }
					change_religion = the_jadd
				}
				else_if = {
					limit = { owner = { religion = old_bulwari_sun_cult } }
					change_religion = old_bulwari_sun_cult
				}
				else = {
					change_religion = bulwari_sun_cult
				}
			}
			random_province = {
				limit = {
					superregion = bulwar_superregion
					or = {
						has_province_modifier = surani_refugee_large
						has_province_modifier = surani_refugee_medium
						has_province_modifier = surani_refugee_small
					}
				}
				province_event = { id = bulwar_flavour.29 days = 10 }
			}
			random_province = {
				limit = {
					culture_group = bulwari
					or = {
						area = azka_sur_area
						area = upper_suran_area
						area = eduz_vacyn_area
					}
				}
				random_list = {
					55 = { }
					15 = { add_base_tax = 1 }
					15 = { add_base_production = 1 }
					15 = { add_base_manpower = 1 }
				}
			}
		}
	}
	
	option = {
		name = bulwar_flavour.1.b
		ai_chance = {
			factor = 0
			modifier = {
				factor = 50
				owner = { low_tolerance_human_race_trigger = yes }
			}
		}
		custom_tooltip = rejected_surani_refugees_tooltip
		set_province_flag = rejected_surani_settlers
		owner = {
			add_prestige = -5
		}
		add_province_modifier = {
			name = rejected_surani_province_modifier
			duration = 3650
		}
	}
}
