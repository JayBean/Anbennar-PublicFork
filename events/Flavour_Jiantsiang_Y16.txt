	is_triggered_only = yes
	trigger = {
		always = yes
	}
# All Hail the Undying Flame!
	option = {
		name = flavor_jiantsiang.16.a
		ai_chance = { factor = 1 }
		add_government_reform = flame_custodian_reform
		if = {
			limit = { 
				NOT = {
					has_ruler = "Katouka"
					dynasty = "of the Undying Flame"
				}
			}
			define_ruler = {
				name = "Kabanda"
				dynasty = "of the Undying Flame"
				claim = 100
				culture = serene_harimari
				age = 19
				male = yes
				adm = 4
				dip = 1
				mil = 4
			}
			add_ruler_personality = mage_personality
			remove_heir = {}
		}
		else = {
			define_heir = {
				name = "Kabanda"
				dynasty = "of the Undying Flame"
				claim = 100
				culture = serene_harimari
				age = 19
				male = yes
				adm = 4
				dip = 1
				mil = 4
			}
			add_heir_personality = mage_personality
			hidden_effect = {
				set_heir_culture = serene_harimari
			}
		}
	}
}
