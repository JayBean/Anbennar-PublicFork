#Dagrin effect
country_event = {
	id = dwarven_pantheon.1
	title = dwarven_pantheon.1.t
	desc = dwarven_pantheon.1.d
	picture = COURT_eventPicture
	
	is_triggered_only = yes
	trigger = { religion = dwarven_pantheon }

	after = { clr_dwarven_pantheon_deities = yes }

	option = {
		name = dwarven_pantheon.1.a
		add_stability_or_adm_power = yes

		if = {
			limit = { has_dwarven_pantheon_deity = { deity = dolurazan } }
			custom_tooltip = dwarven_pantheon_ritual_has_dolurazan_tooltip
			add_adm_power = 50
		}

		if = {
			limit = { has_dwarven_pantheon_deity = { deity = derzobrazan } }
			custom_tooltip = dwarven_pantheon_ritual_has_derzobrazan_tooltip
			add_dip_power = 50
		}
		if = {
			limit = { has_dwarven_pantheon_deity = { deity = bervinazan } }
			custom_tooltip = dwarven_pantheon_ritual_has_bervinazan_tooltip
			add_mil_power = 50
		}
		if = {
			limit = { has_dwarven_pantheon_deity = { deity = karazlov } }
			custom_tooltip = dwarven_pantheon_ritual_has_karazlov_tooltip
			if = {
				limit = { any_ally = { NOT = { trust = { who = ROOT value = 99 } } } }
				random_ally = {
					limit = { NOT = { trust = { who = ROOT value = 99 } } }
					add_trust = { who = ROOT value = 5 }
				}
			}
			else = { add_church_power = 25 }
		}
		if = {
			limit = { has_dwarven_pantheon_deity = { deity = grimthar } }
			custom_tooltip = dwarven_pantheon_ritual_has_grimthar_tooltip
			add_mercantilism = 1
		}
		if = {
			limit = { has_dwarven_pantheon_deity = { deity = verdrik } }
			custom_tooltip = dwarven_pantheon_ritual_has_verdrik_tooltip
			add_prestige = 5
		}
		if = {
			limit = { has_dwarven_pantheon_deity = { deity = thyrfen } }
			custom_tooltip = dwarven_pantheon_ritual_has_thyrfen_tooltip
			add_yearly_manpower = 0.03
		}
		if = {
			limit = { has_dwarven_pantheon_deity = { deity = werdun } }
			custom_tooltip = dwarven_pantheon_ritual_has_werdun_tooltip
			add_inflation = -1
		}
		if = {
			limit = { has_dwarven_pantheon_deity = { deity = lorgrim } }
			custom_tooltip = dwarven_pantheon_ritual_has_lorgrim_tooltip
			if = {
				limit = { any_rival_country = { NOT = { has_spy_network_from = { who = ROOT value = 80 } } } }
				random_rival_country = {
					limit = { NOT = { has_spy_network_from = { who = ROOT value = 80 } } }
					add_spy_network_from = { who = ROOT value = 25 }
				}
			}
			else = { add_church_power = 25 }

	id = dwarven_pantheon.3
	title = dwarven_pantheon.3.t
	desc = dwarven_pantheon.3.d
	picture = CAVE_eventPicture
	
	is_triggered_only = yes
	trigger = { religion = dwarven_pantheon }

	after = { clr_dwarven_pantheon_deities = yes }

	immediate = {
		hidden_effect = {
			random_owned_province = {
				limit = { has_terrain = cavern NOT = { has_province_flag = dwarven_pantheon_has_durvorazir_effect } }
				save_event_target_as = durvorazir_target1
			}
			random_owned_province = {
				limit = { has_terrain = cavern NOT = { has_province_flag = dwarven_pantheon_has_durvorazir_effect province_id = event_target:durvorazir_target1 } }
				save_event_target_as = durvorazir_target2
			}
			random_owned_province = {
				limit = { has_terrain = cavern NOT = { has_province_flag = dwarven_pantheon_has_durvorazir_effect province_id = event_target:durvorazir_target1 province_id = event_target:durvorazir_target2 } }
				save_event_target_as = durvorazir_target3
			}
		}
	}

	option = {
		name = dwarven_pantheon.3.a
		if = {
			limit = { NOT = { has_saved_event_target = durvorazir_target1 } }
			custom_tooltip = dwarven_pantheon_no_target
			add_church_power = 100
		}
		else = {
			event_target:durvorazir_target1 = { add_base_production = 1 hidden_effect = { set_province_flag = dwarven_pantheon_has_durvorazir_effect } }

				limit = { has_saved_event_target = durvorazir_target2 }

		}

		####derzobrazan, intentionally placed first for this effect
		if = {
			limit = { has_dwarven_pantheon_deity = { deity = derzobrazan } }
			custom_tooltip = dwarven_pantheon_ritual_has_derzobrazan_tooltip

				limit = { NOT = { has_saved_event_target = durvorazir_target1 } }
				add_church_power = 25
			}
			else = {
				event_target:durvorazir_target1 = { 
					if = { 
						limit = { NOT = { has_province_flag = dwarven_pantheon_durvorazir_immune trade_goods = gold trade_goods = mithril trade_goods = gems } }

						if = {

country_event = {
	id = dwarven_pantheon.100
	title = dwarven_pantheon.100.t
	desc = dwarven_pantheon.100.d
	picture = GOLDEN_CITY_eventPicture
	
	trigger = {
		religion = ancestor_worship
		NOT = {
			current_age = age_of_discovery
			has_global_flag = dwarven_pantheon_rediscovered
			has_country_flag = dwarven_pantheon_in_rediscovery
			has_country_flag = dwarven_pantheon_refused_rediscovery
		}
		if = {
			limit = { ai = no }
			num_of_owned_provinces_with = {
				value = 4
				province_group = hold_province
				hold_condition = { Repaired = yes }
			}
		}
		else = {
			num_of_owned_provinces_with = {
				value = 2
				province_group = hold_province
			}
			num_of_cities = 8
		}
		
		has_any_disaster = no
	}

	mean_time_to_happen = {
		years = 20
		modifier = {
			factor = 0.8
			total_development = 500
		}
		modifier = {
			factor = 0.8
			full_idea_group = religious_ideas
		}
		modifier = {
			factor = 0.8
			capital_scope = { superregion = east_serpentspine_superregion }
		}
	}

	#cool
	option = {
		name = dwarven_pantheon.100.a
		set_country_flag = dwarven_pantheon_in_rediscovery
		country_event = { id = dwarven_pantheon.101 days = 365 }
		ai_chance = { factor = 5 }
	}
}
