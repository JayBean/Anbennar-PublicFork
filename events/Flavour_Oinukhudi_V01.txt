#talda'khudi starting event
country_event = {
	id = oinukhudi.10
	title = oinukhudi.10.t
	desc = oinukhudi.10.d
	picture = HORDE_ON_HORSEBACK_eventPicture
	
	is_triggered_only = yes

	immediate = {
		hidden_effect = {
			hidden_effect = {
				release = V15
				release = V16
				every_owned_province = {
					limit = {
						has_province_modifier = hill_troll_taishi
					}
					cede_province = V16
					add_core = V16
				}
				V15 = {
					add_country_modifier = {
						name = V15_anarkhudi_war_economy
						duration = 1825
					}
					capital_scope = {
						build_to_forcelimit = {
							infantry = 0.6
							cavalry = 0.4
						}
					}
					create_general = { tradition = 50 add_shock = 1 }
					declare_war_with_cb = {
						who = V01
						casus_belli = cb_independence_war
					}
				}
				V16 = {
					add_country_modifier = {
						name = V16_hill_troll_warriors
						duration = 1825
					}
					capital_scope = {
						build_to_forcelimit = {
							infantry = 0.95
							cavalry = 0.05
						}
					}
					create_general = { tradition = 50 add_shock = 2 }
					declare_war_with_cb = {
						who = V01
						casus_belli = cb_independence_war
					}
				}
			}
		}
	}
	
	option = {	#stay oinukhudi
		name = oinukhudi.10.a
		add_horde_unity = -25
	}

	option = {	#switch to anarkhudi
		name = oinukhudi.10.b
		switch_tag = V15
		tooltip = { add_horde_unity = -25 }
	}

	option = {	#we do a little trolling
		name = oinukhudi.10.c
		switch_tag = V16
		tooltip = { add_horde_unity = -25 }
	}
}
