#FROM Ask for War Mandat
country_event = {
	id = lake.20
	title = lake.20.t
	desc = lake.20.d
	picture = STREET_SPEECH_eventPicture
	goto = capital
	is_triggered_only = yes
	
	trigger = {
		has_country_modifier = lake_federation_member
	}
	
	immediate = {
		hidden_effect = {
			if = {
				limit = { ai = no }
				FROM = {
					capital_scope = {
						save_event_target_as = capital
					}
					if = {
						limit = { any_ally = { NOT = { has_country_flag = ally@FROM } } }
						random_ally = { limit = { NOT = { has_country_flag = ally@FROM } } set_country_flag = ally@FROM save_event_target_as = ally1 }
					}
					if = {
						limit = { any_ally = { NOT = { has_country_flag = ally@FROM } } }
						random_ally = { limit = { NOT = { has_country_flag = ally@FROM } } set_country_flag = ally@FROM save_event_target_as = ally2 }
					}
					if = {
						limit = { any_ally = { NOT = { has_country_flag = ally@FROM } } }
						random_ally = { limit = { NOT = { has_country_flag = ally@FROM } } set_country_flag = ally@FROM save_event_target_as = ally3 }
					}
					if = {
						limit = { any_ally = { NOT = { has_country_flag = ally@FROM } } }
						random_ally = { limit = { NOT = { has_country_flag = ally@FROM } } set_country_flag = ally@FROM save_event_target_as = ally4 }
					}
					if = {
						limit = { any_rival_country = { NOT = { has_country_flag = rival@FROM } } }
						random_rival_country = { limit = { NOT = { has_country_flag = rival@FROM } }  set_country_flag = rival@FROM  save_event_target_as = rival1 }
					}
					if = {
						limit = { any_rival_country = { NOT = { has_country_flag = rival@FROM } } }
						random_rival_country = { limit = { NOT = { has_country_flag = rival@FROM } }  set_country_flag = rival@FROM  save_event_target_as = rival2 }
					}
					if = {
						limit = { any_rival_country = { NOT = { has_country_flag = rival@FROM } } }
						random_rival_country = { limit = { NOT = { has_country_flag = rival@FROM } }  set_country_flag = rival@FROM  save_event_target_as = rival3 }
					}
				}
				set_country_flag = recently_asked_war_right
			}
		}
	}
	

	after = {
		hidden_effect = {
			if = {
				limit = { ai = no }
				FROM = {
					every_known_country = {
						limit = { OR = { has_country_flag = rival@FROM has_country_flag = ally@FROM } }
						clr_country_flag = rival@FROM
						clr_country_flag = rival1@FROM
						clr_country_flag = rival2@FROM
						clr_country_flag = rival3@FROM
						clr_country_flag = ally@FROM
						clr_country_flag = ally1@FROM
						clr_country_flag = ally2@FROM
						clr_country_flag = ally3@FROM
						clr_country_flag = ally4@FROM
					}
				}
			}
		}
	}
	#Give them
	option = {
		name = lake.20.a
		ai_chance = { 
			factor = 65
			
			modifier = {
				factor = 2
				ai_attitude = {
					who = FROM
					attitude = attitude_friendly
				}
			}
			
			modifier = {
				factor = 2
				ai_attitude = {
					who = FROM
					attitude = attitude_protective
				}
			}
			
			modifier = {
				factor = 5
				ai_attitude = {
					who = FROM
					attitude = attitude_allied
				}
			}
			
			modifier = {
				factor = 1.5
				has_opinion = {
					who = FROM
					value = 50
				}
			}
			
			modifier = {
				factor = 1.8
