country_event = {
	id = flavor_adenica.3

	desc = flavor_adenica.3.d
	picture = HORDE_ON_HORSEBACK_eventPicture

					name = adenica_injured

				}

		}

			limit = {
				mil = 5
				NOT = { mil = 6 }
			}

				chance = 5
				add_ruler_modifier = {
					name = adenica_injured
					duration = 1825
				}
				set_country_flag = hunt_boar_injured
			}
			set_country_flag = hunt_boar_success
		}
		else_if = {
			limit = { mil = 6 }
		}	
		else = {
			random = {
				chance = 25
				add_ruler_modifier = {
					name = adenica_injured
					duration = 1825
				}
				set_country_flag = hunt_boar_injured

	option = {
		name = flavor_adenica.3.c

			limit = { 
				NOT = { ruler_has_max_personalities = yes }
				NOT = { ruler_has_personality = inspiring_leader_personality }
				NOT = { ruler_has_personality = craven_personality }
			}
			random = {
				chance = 50
				add_ruler_personality = craven_personality
			}
		}
		set_country_flag = hunt_boar_ran
	}
	after = {
		 country_event = {
			 id = flavor_adenica.7
		 }
	}
}

country_event = {
	id = flavor_adenica.7
	title = flavor_adenica.7.t
	desc = flavor_adenica.7.d
	picture = IMPORTANT_STATUE_eventPicture
	
	is_triggered_only = yes
	trigger = {
		always = yes
	}
	
	option = {
		name = flavor_adenica.7.a
		# Brought family along
		if = { limit = { has_country_flag = hunt_family }
			if = {
				limit = { has_heir = yes }
				if = {
					limit = { heir_mil = 6 }
					add_mil_power = 100
				}
				else = {
					change_heir_mil = 1
				}
			}
			if = {
				limit = { has_consort = yes }
				add_country_modifier = {
					name = adenica_hunt_closer_consort_ties
					duration = 3650
				}
			}
		}
		# Brought mages along
		else_if = { limit = { has_country_flag = hunt_mages }
			add_estate_loyalty_modifier = {
				estate = estate_mages
				desc = adenica_mages_loyalty_hunt
				loyalty = 10
				duration = 7300
			}
			define_advisor = {
				type = court_mage
				skill = 2
				discount = yes
			}
		}
		# Brought nobles along
		else_if = { limit = { has_country_flag = hunt_nobles }
			add_estate_loyalty_modifier = {
				estate = estate_nobles
				desc = adenica_nobles_loyalty_hunt
				loyalty = 10
				duration = 7300
			}
			add_adm_power = 50
		}
		# Brought adventurers along
		else_if = { limit = { has_country_flag = hunt_adventurers }
			add_estate_loyalty_modifier = {
				estate = estate_adventurers
				desc = adenica_adventurers_loyalty_hunt
				loyalty = 10
				duration = 7300
			}
			add_mil_power = 50
		}

		# Hunted a cockatrice
		if = { limit = { has_country_flag = hunt_cockatrice }
			add_ruler_modifier = {
				name = adenica_hunted_cockatrice
				duration = -1
			}
		}
		# Hunted a werebear
		else_if = { limit = { has_country_flag = hunt_werebear }
			add_ruler_modifier = {
				name = adenica_hunted_werebear
				duration = -1
			}
		}
		# Hunted an elk
		else_if = { limit = { has_country_flag = hunt_elk }
			add_ruler_modifier = {
				name = adenica_hunted_elk
				duration = -1
			}
		}
		# Hunted a boar
		else_if = { limit = { has_country_flag = hunt_boar }
			add_ruler_modifier = {
				name = adenica_hunted_boar
				duration = -1
			}
		}
	}

	after = {
		clr_country_flag = hunt_boar
		clr_country_flag = hunt_cockatrice
		clr_country_flag = hunt_werebear
		clr_country_flag = hunt_elk
		clr_country_flag = hunt_family
		clr_country_flag = hunt_mages
		clr_country_flag = hunt_nobles
		clr_country_flag = hunt_adventurers
		clr_country_flag = hunt_boar_injured
		clr_country_flag = hunt_boar_horse_saved
		clr_country_flag = hunt_nobles_save
		clr_country_flag = hunt_boar_success
		clr_country_flag = hunt_werebear_adventurer_negotiated
		clr_country_flag = hunt_werebear_dead
		clr_country_flag = hunt_werebear_negotiated
		clr_country_flag = hunt_werebear_negotiations_failed
		clr_country_flag = hunt_cockatrice_mages_save
		clr_country_flag = hunt_cockatrice_fear
