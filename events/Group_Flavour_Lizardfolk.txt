    #Ravaged by Blood
    country_event = {
        id = lizard.1
        title = lizard.1.t
        desc = lizard.1.d
        picture = IMPALED_SOLDIERS_eventPicture
        is_triggered_only = yes

        trigger ={
            any_owned_province = {
                culture = ashhana_lizardfolk
                NOT = { has_province_modifier = curse_appeased}
            }

        }
        immediate = {
            hidden_effect = {
                random_owned_province ={
                    limit = {
                        NOT = { has_province_modifier = curse_appeased }
                    }
                    save_event_target_as = raging_crocs_province
                }
            }
        }
		#Kill them
		option = {
			name = lizard.1.a
			ai_chance = { factor = 33}
			add_yearly_manpower = -0.2
			}
		#Let the locals handle it
		option = {
				goto = soothing_sound_province
				name = lizard.1.b
				ai_chance = { factor = 33 }
				event_target:raging_crocs_province = {
					add_devastation = 10
				}
			}
		#Sooth them
		option = {
			goto = soothing_sound_province
			name = lizard.1.c
			ai_chance = { factor = 33 }
			event_target:raging_crocs_province = {
				add_province_modifier = { 
					name = curse_appeased
					duration = 3650
				}
			}
			add_adm_power = -25
		}
    }
