# Refugees of the Purge
country_event = {
	id = flavor_corintar.10
	title = flavor_corintar.10.t
	desc = {
		trigger = { has_country_flag = corintar_chapter }
		desc = flavor_corintar.10.da
	}
	desc = {
		trigger = {
			NOT = { has_country_flag = corintar_chapter }
		}
		desc = flavor_corintar.10.db
	}
	picture = SWORD_OF_OSMAN_eventPicture
	
	is_triggered_only = yes
		
	# Accept refugees
	option = {
		name = flavor_corintar.10.a
		ai_chance = {
			factor = 75
			modifier = {
				factor = 0
				OR = {
					has_personal_deity = adean
					is_rival = B02
					is_enemy = B02
				}
			}
			modifier = {
				factor = 0.3
				NOT = { religious_unity = 0.5 }
				religion = regent_court
			}
		}
		remove_country_modifier = corintar_corintar_chapter
		remove_country_modifier = corintar_northern_corintar_chapter
		add_adm_power = 20
		add_dip_power = 20
		add_mil_power = 20
		if = {
			limit = {
				NOT = { religion = corinite }
			}
			random_owned_province = {
				limit = {
					NOT = { religion = corinite }
				}
				change_religion = corinite
				remove_province_modifier = religious_zeal_at_conv
			}
		}
		if = {
			limit = {
				has_dlc = "Common Sense"
				religion = corinite
			}
			add_karma = 10
		}
	}
	
	# Reject refugees
	option = {
		name = flavor_corintar.10.b
		ai_chance = { factor = 25 }
		remove_country_modifier = corintar_corintar_chapter
		remove_country_modifier = corintar_northern_corintar_chapter
		if = {
			limit = {
				is_rival = B02 # Corintar
			}
			add_prestige = 20
		}
		else = {
			add_prestige = 10
		}
	}
}
