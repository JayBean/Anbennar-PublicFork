#Kuiika
country_event = {
	id = medasi.2
	title = medasi.2.t
	desc = medasi.2.d
	picture = DIPLOMACY_eventPicture

	is_triggered_only = yes

	immediate = {
		hidden_effect = {
			if = {
				limit = { has_ruler_flag = ruler_is_feeling_unwell_flag }
				add_stability = -1
			}
		}
	}
	#Medawor
	option = {
		name = medasi.2.a
		trigger = {
			OR = {
				NOT = { ruler_culture = horakheti }
				has_ruler_flag = ruler_is_feeling_unwell_flag
			}
			
		}
		if = {
			limit = { religion_group = khetist }
			define_ruler = {
				dynasty = "Ahmose"
				min_age = 20
				max_age = 40
				culture = horakheti
				random_gender = yes
				change_adm = -1
				change_dip = -1
				change_mil = -1
			}
		}
		else = {
			define_ruler = {
				dynasty = "Ahmose"
				min_age = 20
				max_age = 40
				culture = horakheti
				religion = elikhetist
				random_gender = yes
				change_adm = -1
				change_dip = -1
				change_mil = -1
			}
		}
		add_government_power = {
			mechanic_type = anb_medasi_power_share
			power_type = medasi_power_share
			value = 25
		}
	}

	#Efergezmu
	option = {
		name = medasi.2.b
		trigger = {
			OR = {
				ruler_culture = horakheti
				has_ruler_flag = ruler_is_feeling_unwell_flag
			}
		}

			max_age = 40

			change_dip = -1
			change_mil = -1
			random_gender = yes
		}
		add_government_power = {
			mechanic_type = anb_medasi_power_share
			power_type = medasi_power_share
			value = -25
		}
	}
	option = {
		name = medasi.2.c
		trigger = {
			NOT = {
				has_ruler_flag = ruler_is_feeling_unwell_flag
				ruler_culture = horakheti
			}
		} 
		change_adm = 1
		change_dip = 1
		change_mil = 1
		add_government_power = {
			mechanic_type = anb_medasi_power_share
			power_type = medasi_power_share
			value = -30
		}
		add_scaled_republican_tradition = -10
	}
	
	option = {
		name = medasi.2.e
		trigger = {
			NOT = { has_ruler_flag = ruler_is_feeling_unwell_flag }
			ruler_culture = horakheti
		}
