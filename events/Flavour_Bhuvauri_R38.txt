    trigger = {
        has_reform = free_slave_reform
        NOT = { has_country_flag = in_bhuvauri_election_flag }
    }
    
    immediate = {
        hidden_effect = {
            set_country_flag = in_bhuvauri_election_flag
			S27 = { save_event_target_as = dakinshi_origin }
			set_saved_name = {
				key = dakinshi_name
				type = advisor
				scope = event_target:dakinshi_origin
			}
			U20 = { save_event_target_as = bulwari_origin }
			set_saved_name = {
				key = bulwari_name
				type = advisor
				scope = event_target:bulwari_origin
			}
			L16 = { save_event_target_as = fangaulan_origin }
			set_saved_name = {
				key = fangaulan_name
				type = advisor
				scope = event_target:fangaulan_origin
			}
        }
    }
	
	after = {
		hidden_effect = {
            clr_country_flag = in_bhuvauri_election_flag
			clear_saved_name = bulwari_name
			clear_saved_name = dakinshi_name
			clear_saved_name = fangaulan_name
        }
	}
	
	option = { #Dakinshi - Admin Candidate
		name = flavor_bhuvauri.3.a
		ai_chance = { factor = 1 }
		trigger = { has_country_flag = bhuvauri_fangaulan_ruler } #Country flags instead of ruler flags--less prone to breaking

			name = dakinshi_name
			max_random_adm = 3
			max_random_dip = 1
			max_random_mil = 2
			change_adm = 2
			change_dip = 1
			change_mil = 2
			culture = dakinshi
			religion = beast_memory
			random_gender = yes #Dakinshi faction allows women
		}
		hidden_effect = { set_dynasty = Skyeye }
		clr_country_flag = bhuvauri_fangaulan_ruler
		set_country_flag = bhuvauri_dakinshi_ruler
		set_religious_school = {
			group = raheni
			school = starry_eye_school
		}
	}
	
	option = { #Bulwari - Diplomatic Candidate
		name = flavor_bhuvauri.3.b
