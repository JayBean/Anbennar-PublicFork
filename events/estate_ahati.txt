#Establishing the Ahati
country_event = {
	id = ahati_events.1
	title = ahati_events.1.t
	desc = ahati_events.1.d
	picture = DHIMMI_ESTATE_eventPicture
	
	is_triggered_only = yes
	
	trigger = {
		is_emperor_of_china = yes
	}
	goto = 466

	option = {
		name = ahati_events.1.a
		if = {
			limit = { religion_group = khetist }
			set_estate_privilege = estate_ahati_khetist_recruitment
		}
		else = { set_estate_privilege = estate_ahati_non_khet_steward }
		466 = {
			if = {
				limit = { NOT = { religion = aakhetist } }
				hidden_effect = { change_religion = aakhetist }
			}
			add_province_modifier = { name = ahati_aakheta_base duration = -1 }
		}
		
		change_estate_land_share = { estate = estate_ahati share = 10 }

		#AI should do this always but it dies with no mandate
		if = {
			limit = { ai = yes }
			add_mandate = 80
			change_estate_land_share = { estate = estate_nobles share = -3 }
			change_estate_land_share = { estate = estate_burghers share = -3 }
			change_estate_land_share = { estate = estate_church share = -3 }
		}
		if = {
			limit = { has_country_flag = free_ahati }
			clr_country_flag = free_ahati
			add_mandate = 70
			add_stability = 1
		}

		hidden_effect = { country_event = { id = ahati_events.2 } }
	}
}

			add_yearly_manpower = -0.15

		}
	}

	#let them
	option = {
		name = ahati_events.41.b
		trigger = { NOT = { culture_group = gnollish } }
