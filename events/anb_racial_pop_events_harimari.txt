#Demand Representation
country_event = {
	id = racial_pop_events_harimari.1
	title = racial_pop_events_harimari.1.t
	desc = racial_pop_events_harimari.1.d
	picture = DEBATE_REPUBLICAN_eventPicture
	goto = racial_pop_province_target
	
	trigger = {
		NOT = { has_country_modifier = harimari_administration }
		any_owned_province = {
			NOT = { local_autonomy = 50 }
			is_capital = no
			has_harimari_minority_trigger = yes
		}
	}
	
	is_triggered_only = yes
	
	mean_time_to_happen = {
		days = 1
	}
	immediate = {
		hidden_effect = {
			random_owned_province = {
				limit = {
					has_harimari_minority_trigger = yes
					is_capital = no
				}
				save_event_target_as = racial_pop_province_target
			}
			random_owned_province = {
				limit = {
					NOT = { local_autonomy = 50 }
					is_capital = no
					has_harimari_minority_trigger = yes
				}
				save_event_target_as = racial_pop_province_target
			}
			random_owned_province = {
				limit = {
					NOT = { local_autonomy = 50 }
					is_capital = no
					has_harimari_minority_trigger = yes
					development = 10
				}
				save_event_target_as = racial_pop_province_target
			}
		}
	}
	
	option = {		#Granted!
		name = racial_pop_events_harimari.1.a
		ai_chance = {
			factor = 50
			modifier = {
				factor = 1.5
				wants_to_increase_tolerance_harimari = yes
			}
		}
		
		add_adm_power = -40
		
		medium_increase_of_harimari_tolerance_effect = yes
		
		event_target:racial_pop_province_target = {
			add_local_autonomy = 25
		}
	}
	
	option = {		#Make some concessions
		name = racial_pop_events_harimari.1.b
		ai_chance = {
			factor = 50
			modifier = {
				factor = 1.5
				wants_to_maintain_tolerance_harimari = yes
			}
			modifier = {
				factor = 0.5
				average_autonomy = 10
			}
		}
		
		add_adm_power = -20
		
		event_target:racial_pop_province_target = {
			add_local_autonomy = 10
		}

	}
	
	option = {		#No. Haven't we given them enough?!
		name = racial_pop_events_harimari.1.c
		ai_chance = {
			factor = 50
			modifier = {
				factor = 1.5
				wants_to_decrease_tolerance_harimari = yes
			}
		}
		
		medium_decrease_of_harimari_tolerance_effect = yes
		
	}
	
	option = {		#Special: Make empty promises
		name = racial_pop_events_harimari.1.e
		trigger = { ruler_has_personality = charismatic_negotiator_personality }
		highlight = yes
		ai_chance = {
			factor = 100
			modifier = {
				factor = 0.25
				wants_to_decrease_tolerance_harimari = yes
			}
			modifier = {
				factor = 1.5
				average_autonomy = 10
			}
		}
		
		add_dip_power = -10
		
		small_increase_of_harimari_tolerance_effect = yes
		
		event_target:racial_pop_province_target = {
			add_local_autonomy = 1
		}
		
	}

					NOT = { has_country_modifier = racial_pop_harimari_purge }
