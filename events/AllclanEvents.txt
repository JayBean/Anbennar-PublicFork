	id = allclan.910
	title = allclan.910.t
	desc = allclan.910.d
	picture = TRADEGOODS_eventPicture
	
	major = no

	trigger = {
		has_reform = nezkaru_council_reform
	}

	is_triggered_only = yes
	
	option = { #resolve the crisis
		name = allclan.910.a
		ai_chance = { factor = 100 }
		#allclan
		trigger = {
			OR = {
				stability = 3
				republican_tradition = 75
				AND = {
					NOT = {
						faction_influence = {
							faction = allclan_reformists
							influence = 40
						}
					}
					AND = {
						faction_influence = {
							faction = allclan_cartels
							influence = 25
						}
						faction_influence = {
							faction = allclan_technocrats
							influence = 25
						}
					}
				}
				AND = {
					NOT = {
						faction_influence = {
							faction = allclan_cartels
							influence = 40
						}
					}
					AND = {
						faction_influence = {
							faction = allclan_reformists
							influence = 25
						}
						faction_influence = {
							faction = allclan_technocrats
							influence = 25
						}
					}
				}
				AND = {
					NOT = {
						faction_influence = {
							faction = allclan_technocrats
							influence = 40
						}
					}
					AND = {
						faction_influence = {
							faction = allclan_cartels
							influence = 25
						}
						faction_influence = {
							faction = allclan_reformists
							influence = 25
						}
					}
				}
			}
		}
		add_republican_tradition = -5
		random = {
			chance = 75

			custom_tooltip = allclan_resolve_the_crisis_tt
			hidden_effect = {
				set_country_flag = allclan_resolved_council_disturbance
			}
		}
	}

	option = { #step down to resolve issues
		name = allclan.910.b
		ai_chance = { factor = 2 }
		#allclan
		kill_ruler = yes
		random = { 
			chance = 50 
		
			add_stability = 1
			add_republican_tradition = 5
		}
		if = {
			limit = {
				has_ruler_modifier = allclan_technocrat_ruler
			}
			add_faction_influence = {
				faction = allclan_technocrats
				influence = -10
			}
		}
		if = {
			limit = {
				has_ruler_modifier = allclan_cartel_ruler
			}
			add_faction_influence = {
				faction = allclan_cartels
				influence = -10
			}
		}
		if = {
			limit = {
				has_ruler_modifier = allclan_reformist_ruler
			}
			add_faction_influence = {
				faction = allclan_reformists
				influence = -10
			}
		}
	}

	option = { #work on issues
		name = allclan.910.c
		ai_chance = { factor = 1 }
		#allclan
		random = { 
			chance = 33 
	
			add_republican_tradition = 5

		}

			has_dwarven_minority_trigger = yes

		months = 48

					culture_group = dwarven
				}
			}
		}
		modifier = {
			factor = 2
			owner = { is_at_war = yes }
		}
		modifier = {
			factor = 2

		}

	immediate = {

		}

		if = {

				}
			}
			add_nationalism = 5
		}
		if = {
			limit = {
				OR = {
					has_terrain = dwarven_hold
					has_terrain = dwarven_hold_surface
				}

			}

			hidden_effect = {
				add_dwarven_minority_size_effect = yes
			}
		}
	}

	option = {

	}
