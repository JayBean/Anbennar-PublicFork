country_event = {
	id = sthanan_events.4 #The Uses of the Past
	title = sthanan_events.4.t
	desc = sthanan_events.4.d
	picture = INVENTION_eventPicture

	is_triggered_only = yes
	fire_only_once = yes

	immediate = {
		hidden_effect = {
			set_variable = {
				which = g81ProductionOnRelics
				value = 0
			}
			set_variable = {
				which = g81ProductionOnRelicsAdd
				value = 0
			}
			while = {
				limit = {
					any_province = {
						country_or_non_sovereign_subject_holds = ROOT 
						superregion = taychend_superregion
						trade_goods = precursor_relics
						NOT = {
							has_province_flag = g81ProductionOnRelicsChecked
						}
					}
				}
				random_province = {
					limit = {
						country_or_non_sovereign_subject_holds = ROOT 
						superregion = taychend_superregion
						trade_goods = precursor_relics
						NOT = {
							has_province_flag = g81ProductionOnRelicsChecked
						}
					}
					set_province_flag = g81ProductionOnRelicsChecked
					ROOT = {
						export_to_variable = {
							which = g81ProductionOnRelicsAdd
							value = trigger_value:base_production
							who = PREV
						}
					}
				}
				change_variable = {
					which = g81ProductionOnRelics
					which = g81ProductionOnRelicsAdd
				}
				set_variable = {
					which = g81ProductionOnRelicsAdd
					value = 0
				}
			}
		}
	}

	option = {
		name = sthanan_events.4.a
		custom_tooltip = g81ProductionOnRelics_tt
		if = {
			limit = {
				NOT = {
					check_variable = {
						which = g81ProductionOnRelics
						value = 20
					}
				}
			}
			2541 = {
				add_province_modifier = {
					name = G81_vustarin_knowledge_1
					duration = 9125
				}
				add_base_tax = 1
				add_base_production = 1
				add_base_manpower = 1
			}
		}
		else_if = {
			limit = {
				NOT = {
					check_variable = {
						which = g81ProductionOnRelics
						value = 30
					}
				}
			}
			change_innovativeness = 4
			2541 = {
				add_province_modifier = {
					name = G81_vustarin_knowledge_2
					duration = 9125
				}
				add_base_tax = 1
				add_base_production = 1
				add_base_manpower = 1
			}
		}
		else_if = {
			limit = {
				NOT = {
					check_variable = {
						which = g81ProductionOnRelics
						value = 40
					}
				}
			}
			change_innovativeness = 6
			2541 = {
				add_province_modifier = {
					name = G81_vustarin_knowledge_3
					duration = 9125
				}
				add_base_tax = 2
				add_base_production = 2
				add_base_manpower = 2
			}
		}
		else = {
			change_innovativeness = 8
			2541 = {
				add_province_modifier = {
					name = G81_vustarin_knowledge_4
					duration = 9125
				}
				add_base_tax = 2
				add_base_production = 2
				add_base_manpower = 2
			}
		}
	}
}
