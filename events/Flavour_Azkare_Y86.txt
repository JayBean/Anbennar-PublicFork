#Wuhyun conversions fired from on_war_won
country_event = {
	id = azkare.240
	title = azkare.240.t
	desc = azkare.240.d
	picture = STREET_SPEECH_eventPicture
	
	is_triggered_only = yes
	
	option = { 
		name = azkare.240.a
		every_owned_province = {
			limit = {
				culture_is_wuhyun = yes
				continent = asia
			}
			if = {
				limit = {
					check_variable = {
						which = original_wuhyun_religion
						value = 1
					} #uninitialised = 0 = no religion	
				}
				change_religion = variable:original_wuhyun_religion
			} # only do this if religion also changed - doing it blindly = no religion

			if = {
				limit = {
					check_variable = {
						which = original_wuhyun_culture
						value = 1
					} #uninitialised = 0 = no culture
				}
				change_culture = variable:original_wuhyun_culture
			} # only do this if culture changed - it should always have this variable set but
			else = {
				change_culture = ROOT
			} # for any edge cases make it your primary culture, or you will be stuck in an infinite loop of this event firing
		}
	}
}
