country_event = {
	id = diggy_project.1
	title = diggy_project.1.t
	desc = diggy_project.1.d
	picture = ARTWORK_eventPicture
	
	is_triggered_only = yes
	major = yes
	
	trigger = {
		always = yes
	}
	
	option = {
        name = diggy_project.1.a
        ai_chance = { factor = 100 }
		add_prestige = 5
		custom_tooltip = start_build_magma_tooltip 
	}
}

country_event = {
	id = diggy_project.2
	title = diggy_project.2.t
	desc = diggy_project.2.d
	picture = ARTWORK_eventPicture
	
	mean_time_to_happen = {
		years = 40
		
		modifier = {
			factor = 0.9
			capital_scope = { has_coexisting_dwarven_pop_trigger = yes }
		}
		modifier = {
			factor = 0.8
			capital_scope = { has_integrated_dwarven_pop_trigger = yes }
		}
		
		modifier = {
			factor = 0.9
			capital_scope = { base_production = 25 }
		}
		
		modifier = {
			factor = 0.9
			capital_scope = { base_production = 30 }
		}
		
		modifier = {
			factor = 0.9
			capital_scope = { base_production = 35 }
		}
		
		modifier = {
			factor = 0.8
			capital_scope = { base_production = 40 }
		}
	}
	
	trigger = {
		has_country_flag = building_magma_forge
		capital_scope = { has_province_modifier = hold_foundry }
	}
	
	option = {
        name = diggy_project.2.a
        ai_chance = { factor = 100 }
		hidden_effect = { clr_country_flag = building_magma_forge }
		hidden_effect = { clr_country_flag = building_diggy }
		hidden_effect = { set_country_flag = builded_magma_forge }
		add_prestige = 15
		if = {
			limit = { NOT = { has_global_flag = magma_forge_build } }
			hidden_effect = { set_global_flag = magma_forge_build }
			custom_tooltip = first_magma_forge_tooltip
			capital_scope = {
				remove_province_modifier = hold_foundry
				add_permanent_province_modifier = {
					name = hold_foundry_2
					duration = -1
				}
				add_permanent_province_modifier = {
					name = magma_forge
					duration = -1
				}
			}
		}
		else = {
			custom_tooltip = normal_magma_forge_tooltip
			capital_scope = {
				remove_province_modifier = hold_foundry
				add_permanent_province_modifier = {
					name = hold_foundry_2
					duration = -1
				}
			}
		}
	}
}
