#Factions are formed at the start
country_event = {
	id = fed_magew.1
	title = fed_magew.1.t
	desc = fed_magew.1.d
	picture = STREET_SPEECH_eventPicture

	is_triggered_only = yes

	trigger = {
		has_country_modifier = lake_federation_member
	}

	immediate = {
		hidden_effect = {
			if = {
				limit = {
					NOT = {
						has_global_flag = federation_crisis
					}
				}
				set_global_flag = federation_crisis
				every_country = {
					limit = {
						has_country_modifier = lake_federation_member
					}
					country_event = {
						id = fed_magew.2
						days = 730
						random = 730
					}
				}
				5230 = {
					set_variable = {
						which = LF_mage_crisis_number_total
						value = 0
					}
					set_variable = {
						which = LF_mage_crisis_number_high
						value = 0
					}
					set_variable = {
						which = LF_mage_crisis_number_mid
						value = 0
					}
					set_variable = {
						which = LF_mage_crisis_number_low
						value = 0
					}
				}
			}
			5230 = {
				change_variable = {
					which = LF_mage_crisis_number_total
					value = 1
				}
			}
		}
	}

	option = {
		name = fed_magew.1.a
		trigger = {
			has_country_flag = LF_mages_normal
		}
		hidden_effect = {
			5230 = {
				change_variable = {
					which = LF_mage_crisis_number_high
					value = 1
				}
			}
		}
		custom_tooltip = LF_magew_actions
	}

	option = {
		name = fed_magew.1.b
		trigger = {
			has_country_flag = LF_mages_onyxguard
		}
		hidden_effect = {
			5230 = {
				change_variable = {
					which = LF_mage_crisis_number_mid
					value = 1
				}
			}
		}
		custom_tooltip = LF_magew_actions
	}

	option = {
		name = fed_magew.1.c
		trigger = {
			has_country_flag = LF_mages_banned
		}
		hidden_effect = {
			5230 = {
				change_variable = {
					which = LF_mage_crisis_number_low
					value = 1
				}
			}
		}
		custom_tooltip = LF_magew_actions
	}

	after = {
		if = {
			limit = {
				ai = no
			}
			set_country_flag = federation_magew_can_sway
		}
	}
}
