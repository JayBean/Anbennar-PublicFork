country_event = {
    id = mother_akan.1
    title = mother_akan.1.t
    desc = mother_akan.1.d
    picture = STREET_SPEECH_eventPicture

    hidden = yes
    mean_time_to_happen = {
        days = 1
    }
    
    trigger = {
        OR = {
            NOT = { has_global_flag = akan_cycle_timer }
            had_global_flag = {
                flag = akan_cycle_timer
		days = 4380
            }
        }
    }
    
    immediate = {
        set_global_flag = akan_cycle_timer
    }
    
    option = {
        name = mother_akan.1.a
        ai_chance = { factor = 100 }
        if = {
			limit = { has_global_flag = mother_akan_4 }
			clr_global_flag = mother_akan_4
			set_global_flag = mother_akan_1
		}
		else_if = {
			limit = { has_global_flag = mother_akan_1 }
			clr_global_flag = mother_akan_1
			set_global_flag = mother_akan_2
		}
		else_if = {
			limit = { has_global_flag = mother_akan_2 }
			clr_global_flag = mother_akan_2
			set_global_flag = mother_akan_3
		}
		else_if = {
			limit = { has_global_flag = mother_akan_3 }
			clr_global_flag = mother_akan_3
			set_global_flag = mother_akan_4
		}
		every_country = {
			limit = { religion = mother_akan }
			country_event = { id = mother_akan.2 }
       }
    }
}
