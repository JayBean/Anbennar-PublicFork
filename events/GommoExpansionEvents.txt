country_event = { #Gnomish settlers
	id = gommo_expansion_events.1
	title = gommo_expansion_events.1.t
	desc = gommo_expansion_events.1.d
	picture = EUROPEAN_REFUGEES_eventPicture

	trigger = {
		any_owned_province = {
			is_core = ROOT
			trade_goods = precursor_relics
			development = 30
			NOT = {
				has_any_gnomish_pop_trigger = yes
			}
		}
		NOT = { culture_group = gnomish }
		has_institution = global_trade
	}
	
	is_triggered_only = yes
	
	option = { #Welcome them
		name = "gommo_expansion_events.1.a"
		largest_increase_of_gnomish_tolerance_effect = yes
		random_owned_province = {
			limit = {
				trade_goods = precursor_relics
				development = 30
				NOT = {
					has_any_gnomish_pop_trigger = yes
				}
			}
			add_gnomish_minority_size_effect = yes
		}
		
		ai_chance = {
			factor = 75
			modifier = {
				factor = 0
				wants_to_decrease_tolerance_gnomish = yes
			}
		}
	}

	option = { #Refuse them
		name = "gommo_expansion_events.1.b"
		
		medium_decrease_of_gnomish_tolerance_effect = yes
		add_adm_power = 35
		
		ai_chance = {
			factor = 25
			modifier = {
				factor = 0
				wants_to_increase_tolerance_gnomish = yes
			}
		}
	}
}

		owner = {
			NOT = { culture_group = gnomish }
			has_institution = global_trade
			NOT = { has_country_flag = gommo_petition_refused_gommo }
			is_core = ROOT
		}
		
		OR = {
			OR = {		
				has_integrated_gnomish_pop_trigger = yes
				has_coexisting_gnomish_pop_trigger = yes

				owner = {
					ai = yes
				}

			}
		}
		
		NOT = { has_province_modifier = gnomish_institute_gommo }
		NOT = { has_province_flag = special_gnomish_quarter_gommo }

	}
	
	is_triggered_only = yes
	
	option = { #Grant them charter

		owner = { largest_increase_of_gnomish_tolerance_effect = yes }
		add_gnomish_minority_size_effect = yes
		
		add_permanent_province_modifier = {
			name = gnomish_institute_gommo

		}
		
		hidden_effect = {

					chance = 50

					chance = 50

				}
				random = {
					chance = 50
					country_event = { id = gommo_expansion_events.8 days = 120 random = 120 }
				}
				random = {
					chance = 50
					country_event = { id = gommo_expansion_events.9 days = 90 random = 120 }
				}
			}
        }
		
		ai_chance = {
			factor = 75
			modifier = {
				factor = 0
				owner = { wants_to_decrease_tolerance_gnomish = yes }
				NOT = {
					province_id = 4871

				factor = 400

			}

		owner = { medium_decrease_of_gnomish_tolerance_effect = yes }
		
		ai_chance = {
			factor = 20

				factor = 0
				OR = {
					owner = { wants_to_increase_tolerance_gnomish = yes }
					province_id = 4871
				}
			}
		}
	}
	
	option = {
		name = "gommo_expansion_events.3.c"
		
		owner = { 
			large_decrease_of_gnomish_tolerance_effect = yes 
			set_country_flag = gommo_petition_refused_gommo
		}
		
		ai_chance = {
