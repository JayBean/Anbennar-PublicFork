##A new acolyte take the throne
country_event = {
	id = black_demesne.1
	title = black_demesne.1.t
	desc = black_demesne.1.d
	picture = ACOLYTE_ANIMATED_eventPicture
	
	is_triggered_only = yes
	
	trigger = {
		tag = Z99
		NOT = { has_ruler_flag = well_of_majesty_ritual }
		any_subject_country = {
			is_subject_of_type = acolyte_dominion
			NOT = {
				calc_true_if = {
					all_known_country = {
						check_variable = {
							which = aInfluence
							which = PREV
						}
					}
					amount = 1
				}
			}
		}
	}
	
	immediate = {
		hidden_effect = {
			if = {
				limit = { NOT = { ruler_can_revive = yes } }
				random_subject_country = {
					limit = { 
						is_subject_of_type = acolyte_dominion
						NOT = {
							calc_true_if = {
								all_known_country = {
									check_variable = {
										which = aInfluence
										which = PREV
									}
								}
								amount = 1
							}
						}
					}
					save_event_target_as = country
					set_country_flag = old_dominion
					exile_ruler_as = { name = black_heir }
				}
			}
			clr_ruler_flag = magic_menu_opened
		}
	}
	
	after = {
		hidden_effect = { country_event = { id = black_demesne.100 } }
	}
	
	option = {
		name = black_demesne.1.a
		trigger = { NOT = { ruler_can_revive = yes } }
		ai_chance = { factor = 40 }
		set_ruler = black_heir
		hidden_effect = { clr_ruler_flag = mage_ruler_general }
		add_ruler_modifier = {
			name = new_sorcerer_king
			duration = 3650
		}
	}
	
	option = {
		name = "magic_project_lichdom.12.c"
		trigger = { ruler_can_revive = yes }
		ai_chance = {  factor = 1000	 }
		custom_tooltip = remains_ruler
		add_ruler_modifier = {
			name = black_weakened_lich
			duration = 5475
		}
		revival_stat_loss = yes
	}
}

#The Apostle
country_event = {
	id = black_demesne.3
	title = black_demesne.3.t
	desc = black_demesne.3.d
	picture = APOSTLE_eventPicture
	
	is_triggered_only = yes
	
	trigger = {
		always = yes
	}
	
	immediate = {
		hidden_effect = { set_country_flag = debug_menu }
		hidden_effect = {
			if = {
				limit = { ai = no }
				set_variable = { foo = 0 }
				export_to_variable = {
					which = foo
					value = years_of_income
				}
				multiply_variable = { foo = 0.03 }
				divide_variable = { foo = 1000 }
				multiply_variable = { foo = 1000 }
			}
		}
	}
	after = {
		hidden_effect = { clr_country_flag = debug_menu country_event = { id = black_demesne.100 } }
	}
	
	#Core
	option = {
		name = black_demesne.3.a
		ai_chance = { factor = 40 }
		custom_tooltip = the_apostle_tooltip
		while = {
			limit = {
				any_owned_province = {
					is_territory = yes
					NOT = { is_core = ROOT }
					NOT = { has_construction = core }
					NOT = { is_colony = yes }
				}
				adm_power = 50
			}
			random_owned_province = {
				limit = { 
					is_territory = yes
					NOT = { is_core = ROOT }
					NOT = { has_construction = core }
					NOT = { is_colony = yes }
				}
				if = {
					limit = { owner = { ai = yes } }
					if = {
						limit = { owner = { overextension_percentage = 1 } }
						add_core_construction = {
							speed = 0.25
							cost = 0.5
						}
					}
					else_if = {
						limit = { owner = { overextension_percentage = 0.5 } }
						add_core_construction = {
							speed = 0.5
							cost = 0.5
						}
					}
					else_if = {
						limit = { owner = { overextension_percentage = 0.5 } }
						add_core_construction = {
							speed = 0.75
							cost = 0.5
						}
					}
				}
				else = {
					add_core_construction = {
						speed = 2
						cost = 0.75
					}
				}
			}
			if = {
				limit = { ai = no }
				add_years_of_income = -0.03
			}
		}
	}
	
	#Go Back
	option = {
		name = black_demesne.3.b
		ai_chance = { factor = 0 }
	}
}

					}

					}
				}
			}
		}
	}

				}

				army_strength = {

				NOT = {

						who = event_target:country
						value = 50
					}
				}
			}
			
			modifier = {
				factor = 1.5
				
				NOT = {

					}
				}
			}
