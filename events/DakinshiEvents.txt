    fire_only_once = yes

	trigger = {
		tag = S28
	}

	}

		hidden_effect = {
			S27 = { country_event = { id = dakinshi.13 } }

			define_ruler_to_general = {
				fire = 0
				shock = 8
   				manuever = 3
				siege = 6
				trait = war_wizard_personality
			}
		}
    }

	option = {
		name = dakinshi.1.a
        ai_chance = { factor = 100 }
		declare_war_with_cb = {
		    who = S27
		    casus_belli = cb_dakinshi_civil
		}
		custom_tooltip = "§TWhile the Warlord Era is raging you can freely annex the other warlords and the Republic of Butuguzi by defeating them in a war.§!"
        hidden_effect = {
            if = {
                limit = {
                    ai = yes
                }
                add_manpower = 4
                add_treasury = 50
                add_army_tradition = 25
            }
        }
	}

    option = {
		name = dakinshi.1.b
        ai_chance = { factor = 0 }

		custom_tooltip = "§TWhile the Warlord Era is raging you can freely annex the other warlords and the Republic of Butuguzi by defeating them in a war.§!"
	}
