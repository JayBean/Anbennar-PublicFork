	trigger = {
		development = 25
		has_province_modifier = mulen_hanging_city_foundation
		owner = {
			OR = {
				AND = {
					OR = {
						was_tag = F50
						tag = F50
					}
					has_country_flag = mulen_hanging_cities
				}
				AND = {
					is_subject = yes
					overlord = {
						OR = {
							was_tag = F50
							tag = F50
						}
						has_country_flag = mulen_hanging_cities
					}
				}
            }
            has_country_modifier = harpy_administration
		}
	}

	option = {
		name = flavour_mulen.1.a
		remove_province_modifier = mulen_hanging_city_foundation
		add_permanent_province_modifier = {
			name = mulen_hanging_city
			duration = -1
		}
		if = {
			limit = {
				owner = {
					has_estate_privilege = estate_burghers_matriarch_autonomy
				}
			}
			add_province_modifier = {
				name = mulen_matriarch_autonomy
				duration = -1
			}
		}
		ai_chance = {
			factor = 1
		}
	}
}
