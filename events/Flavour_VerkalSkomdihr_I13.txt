	trigger = {
		always = yes
	}

		name = verkalskomdihr.7.a
		deepwoods_superregion = { discover_country = ROOT }
		add_manpower = 5
		add_army_tradition = 10
		add_prestige = 10
		lonfarch_area = {
			limit = {
				NOT = { owned_by = ROOT }
				NOT = { is_permanent_claim = ROOT }
			}
			add_permanent_claim = ROOT
		}
		ayeth_area = {
			limit = {
				NOT = { owned_by = ROOT }
				NOT = { is_permanent_claim = ROOT }
			}
			add_permanent_claim = ROOT
		}
	}
	
	after = {
		set_country_flag = verkalskomdihr_expedition_end
	}
}
