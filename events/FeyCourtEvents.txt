#improve faction relations events from church actions
country_event = {
	id = fey_court.1
	title = fey_court.1.t
	desc = fey_court.1.d
	picture = WOOD_ELVEN_THRONE_ROOM_eventPicture
	goto = fey_court_seelie_offering_target
	
	is_triggered_only = yes

	trigger = {
		religion = fey_court
	}

	immediate = {
		hidden_effect = {
			capital_scope = {
				save_event_target_as = fey_court_seelie_offering_target
			}
		}
	}

	option = {
		name = fey_court.1.a
		if = { 
			limit = { max_manpower = 12 }
			add_manpower = -10
		} else = {
			add_yearly_manpower = -0.5
			if = {
				limit = { 
					OR = {
						fey_court_unseelie_court_pleased_trigger = yes 
						fey_court_unseelie_court_delighted_trigger = yes
						has_reform = favor_the_unseelie
					}
				}
				add_yearly_manpower = -0.25
			}
			if = {
				limit = { 
					OR = {
						fey_court_unaligned_fey_pleased_trigger = yes
						fey_court_unaligned_fey_delighted_trigger = yes
					}
				}
				add_yearly_manpower = -0.25
			}
		}
		fey_court_slightly_improve_seelie_court_relations = yes
		fey_court_slightly_hurt_unseelie_court_relations = yes
		add_country_modifier = {
			name = fey_court_seelie_offering_cooldown
			duration = 1825
		}
	}
	option = {
		name = fey_court.1.b
		custom_tooltip = fey_court.1.b.tt
		dip_power_cost = 50
		fey_court_moderately_improve_seelie_court_relations = yes
		fey_court_slightly_hurt_unseelie_court_relations = yes
		add_country_modifier = {
			name = fey_court_seelie_offering_cooldown
			duration = 1825
		}
	}
	option = {
		name = fey_court.1.c
		custom_tooltip = fey_court.1.c.tt
		event_target:fey_court_seelie_offering_target = {
			add_devastation = 25
		}
		fey_court_greatly_improve_seelie_court_relations = yes
		fey_court_slightly_hurt_unseelie_court_relations = yes
		add_country_modifier = {
			name = fey_court_seelie_offering_cooldown
			duration = 1825
		}
	}
	option = {
		name = fey_court.1.e
		add_church_power = 75
	}
}
