# 2_ Notestock I mean symposium
country_event = {
	id = ravstate.2
	title = ravstate.2.t
	desc = ravstate.2.d
	picture = CULTURE_eventPicture
	
	is_triggered_only = yes
	trigger = {
		always = yes
	}
	
	# A_Displays of power
	option = {
		name = ravstate.2.a
		ai_chance = { factor = 50 }
		trigger = {
			has_global_flag = fourth_ravelian_council
			reform_desire = 25
		}
		add_years_of_income = -0.5
		add_reform_desire = -25
		every_country = {
			limit = {
				not = { tag = Z97}
				religion = ravelian
				ai = yes
			}
			add_country_modifier = {
				name = symposium_displays_of_strength
				duration = 1825
			}
		}
	}
	# B_Convince some of the visitors to join the army!
	option = {
		name = ravstate.2.b
		ai_chance = { factor = 0 }
		trigger = {
			prestige = 50 
			mission_completed = Z97_draft_more_internationals
		}
		if = { 
			limit = { 
				calc_true_if = {
					all_province = { religion = ravelian } 
					amount = 1250
				}
			}
			add_manpower = 40
		}
		else_if = { 
			limit = { 
				calc_true_if = {
					all_province = { religion = ravelian } 
					amount = 1000
				}
			}
			add_manpower = 30
		}
		else_if = { 
			limit = { 
				calc_true_if = {
					all_province = { religion = ravelian } 
					amount = 750
				}
			}
			add_manpower = 20
		}
		else = {
			add_manpower = 10
		}
	}
	# C_Organize mission trips
	option = {
		name = ravstate.2.c
		ai_chance = { factor = 1}
		
		add_years_of_income = -0.25
		every_country = {
			limit = {
				not = { tag = Z97}
				NOT = { war_with = Z97}
				not = { is_enemy = Z97}
				not = { is_rival = Z97}
				religion = ravelian
			}
			add_country_modifier = {
				name = symposium_mission_trips
				duration = 1825
			}
		}
	}
	# E_Suggest more propaganda music for our soldiers...
	option = {
		name = ravstate.2.e
		ai_chance = { factor = 0 }
		trigger = {	prestige = 80}
		add_country_modifier = {
			name = "A90_notestock_3"
			duration = 1825
		}
	}
	# F_Charge Admission
	option = {
		name = ravstate.2.f
		ai_chance = { factor = 50 }
		trigger = { prestige = 20 }
		add_prestige = -20
		add_years_of_income = 1
	}
	# G_Promote aggressive trade
	option = {
		name = ravstate.2.g
		ai_chance = { factor = 0 }
		trigger = {
			prestige = 50
		}
		add_mercantilism = 2
		every_country = {
			limit = {
				NOT = {war_with = Z97}
				NOT = {is_enemy = Z97}
				NOT = {is_rival = Z97}
				religion = ravelian
				#ai = yes
			}
			add_country_modifier = {
				name = symposium_trade_ventures
				duration = 1825
			}
			add_treasury = 100
		}
	}
	# Same time next half-decade?
	after = {
		hidden_effect = {
			country_event = {
				id = ravstate.2
				days = 1825 # 5 years
			}
		}
	}
}
