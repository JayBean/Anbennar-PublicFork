		is_year = 1630

		religion_group = cannorian

			NOT = { num_of_cities = 6 }

		}

		OR = {
			AND = {
				has_global_flag = allow_non_ideal_rectorate_candidate

				can_have_center_of_reformation_trigger = { RELIGION = ravelian }

			factor = 1.5
			religion = regent_court
		}
		modifier = {
			factor = 0.5
			tag = A38
		}
		modifier = {
			factor = 0.5
			num_of_owned_provinces_with = {
				value = 1
				development = 30
			}
		}
		
		modifier = {
			factor = 0.5

		}

			factor = 1.5
			NOT = { innovativeness_ideas = 3 }

			factor = 2
			NOT = { innovativeness_ideas = 2 }
		}
		modifier = {
			factor = 2
			NOT = { innovativeness_ideas = 1 }
		}

			factor = 0.8
			has_idea = humanist_tolerance

			factor = 0.5

		ai_chance = {
			factor = 70
		}
		enable_religion = ravelian

			province_type = random_owned_province
		}
		
		capital_scope = {
			change_religion = ravelian

			name = "conversion_zeal"
