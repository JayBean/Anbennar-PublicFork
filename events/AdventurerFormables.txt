	id = marcher.3

	desc = marcher.3.d

	trigger = {
		NOT = {current_age = age_of_discovery}
		NOT = {current_age = age_of_reformation}

			NOT = {
				capital_scope = {
					region = west_castanor_region
				}
				capital_scope = {
					region = south_castanor_region

					region = inner_castanor_region
				}

					area = whistlevale_area
				}

					area = cursewood_area

			any_owned_province = {
				OR = {
					region = west_castanor_region
					region = south_castanor_region
					region = inner_castanor_region

					area = cursewood_area
				}
				NOT = { culture = ROOT }
				OR = {
					culture = aldresian
					culture = anbenncoster
