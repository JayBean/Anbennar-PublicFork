country_event = {
	id =  flavour_reuyel.3
	title = flavour_reuyel.3.t
	desc = flavour_reuyel.3.d
	picture = NOBLE_ESTATE_eventPicture
	
	fire_only_once = yes
	
	trigger = { 
		tag = U18
	}
	mean_time_to_happen = {
		days = 180
	}
	
	#Jadd option
	option = {
		name = flavour_reuyel.3.a
		ai_chance = {
			factor = 0.33
			modifier = {
				factor = 2
				any_enemy_country = {
					primary_culture = sun_elf
				}
			}
		}
		define_general = {
			name = "Camnaril of the Sands"
			shock = 5
			fire = 2
			manuever = 5
			siege = 3
			trait = hardy_warrior_personality
		}
		set_country_flag = reuyel_jadd
		
		every_country = {
			limit = { 
				religion = the_jadd 
				NOT = { tag = ROOT }
			}
			add_opinion = { who = ROOT modifier = the_jadd_converted_to_jadd }
		}
		every_country = {
			limit = { 
				religion = bulwari_sun_cult 
				NOT = { tag = ROOT }
			}
			add_opinion = { who = ROOT modifier = bulwari_converted_to_jadd }
		}
		swap_non_generic_missions = yes
	}
	#OSC option
	option = {
		name = flavour_reuyel.3.b
		ai_chance = {
			factor = 0.33
			modifier = {
				factor = 2
				any_enemy_country = {
					primary_culture = sun_elf
				}
			}
			modifier = {
				factor = 2
				any_enemy_country = {
					primary_culture = exodus_goblin
				}
			}
		}
		define_advisor = {
			type = theologian
				name = "Sepeh of Kuzak"
				skill = 2
				discount = yes
				culture = bahari
				religion = old_bulwari_sun_cult
			}
		set_country_flag = reuyel_osc
		every_country = {
			limit = { 
				religion = old_bulwari_sun_cult 
				NOT = { tag = ROOT }
			}
			add_opinion = { who = ROOT modifier = osc_converted_to_jadd }
		}
		every_country = {
			limit = { 
				religion = bulwari_sun_cult 
				NOT = { tag = ROOT }
			}
			add_opinion = { who = ROOT modifier = bulwari_converted_to_jadd }
		}
		swap_non_generic_missions = yes
	}
	#NSC option
	option = {
		name = flavour_reuyel.3.c
		ai_chance = {
			factor = 0.33
			modifier = {
				factor = 2
				any_ally = {
					primary_culture = sun_elf
				}
			}
		}
		set_country_flag = reuyel_nsc
		every_country = {
			limit = {

			}
