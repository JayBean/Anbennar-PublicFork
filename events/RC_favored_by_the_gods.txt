# Favored of the Gods - Castellos
country_event = {
	id = favored_RC.1
	title = "favored_RC.1.t"
	desc = "favored_RC.1.d"
	picture = RELIGION_eventPicture

	trigger = {
		has_personal_deity = castellos
		OR = {
			ruler_has_personality = righteous_personality
			ruler_has_personality = just_personality
			ruler_has_personality = well_advised_personality
		}
		has_regency = no
		NOT = { has_ruler_modifier = RC_withdrawn_meditation }
		NOT = { has_ruler_modifier = favored_RC_castellos_t_a }
		NOT = { has_ruler_modifier = favored_RC_castellos_t_b }
		NOT = { has_ruler_flag = RC_emulant_rejected_castellos }
		OR = {
			NOT = { has_ruler_modifier = favored_RC_adean_t_a has_ruler_modifier = favored_RC_adean_t_b }
			ruler_has_personality = just_personality
			ruler_has_personality = well_advised_personality
			ruler_has_personality = martial_educator_personality
			ruler_has_personality = strict_personality
		}
		OR = {
			NOT = { has_ruler_modifier = favored_RC_nerat_t_a has_ruler_modifier = favored_RC_nerat_t_b }
			ruler_has_personality = righteous_personality
			ruler_has_personality = well_advised_personality
			ruler_has_personality = lawgiver_personality
			ruler_has_personality = incorruptible_personality
		}
	}

	is_triggered_only = yes

	# Emulate aspect 1
	option = {
		name = "favored_RC.1.a"
		add_ruler_modifier = {
			name = favored_RC_castellos_t_a
			hidden = yes
			duration = -1
		}
		add_ruler_modifier = {
			name = RC_withdrawn_meditation
			duration = 1825
		}
		custom_tooltip = favored_tt
		tooltip = {
			add_ruler_modifier = {
				name = RC_emulant_castellos_1
				duration = -1
			}
		}
		hidden_effect = {
			country_event = {
				id = favored_RC.2
				days = 1826
			}
		}
	}

	# Emulate aspect 2
	option = {
		name = "favored_RC.1.b"
		add_ruler_modifier = {
			name = favored_RC_castellos_t_b
			hidden = yes
			duration = -1 }
		add_ruler_modifier = {
			name = RC_withdrawn_meditation
			duration = 1825
		}
		custom_tooltip = favored_tt
		tooltip = {
			add_ruler_modifier = {
				name = RC_emulant_castellos_2
				duration = -1
			}
		}
		hidden_effect = {
			country_event = {
				id = favored_RC.3
				days = 1826
			}
		}
	}

	# Eschew emulation
	option = {
		name = "favored_RC.1.c"
		add_prestige = 2
		set_ruler_flag = RC_emulant_rejected_castellos
	}
}
