namespace = anb_piracy

country_event = { #Rhasail
	id = anb_piracy.1
	title = "anb_piracy.1.t"
	picture = PIRATE_CAPTAINS_eventPicture
	desc = "anb_piracy.1.d"

	fire_only_once = yes

	trigger = {
		has_dlc = "Golden Century"
		NOT = { exists = J95 }
		has_global_flag = golden_age_of_piracy
		NOT = { has_reform = pirate_republic_reform }
		ruined_isles_region = {
			type = all
			is_city = yes
		}
		OR = {
			AND = {
				owns_core_province = 1100
				is_subject = no
				num_of_cities = 2
			}
			any_subject_country = {
				owns_core_province = 1100
				num_of_cities = 2
			}
		}
		1100 = { # must be privateering in node
			privateer_power = {
				country = ROOT
				share = 5
			}
		}
		OR = {
			1100 = {
				recent_treasure_ship_passage = yes
			}
			1100 = {
				trade_share = {
					country = ROOT
					share = 10
				}
			}
			any_country = {
				capital_scope = {
					region = ruined_isles_region
				}
				war_exhaustion = 5
			}
		}
	}

	immediate = {
		hidden_effect = {
			1100 = {
				discover_country = J95
				add_core = J95
			}
			random_country = {
				limit = {
					is_rival = ROOT
					any_subject_country = {
						any_owned_province = {
							region = ruined_isles_region
						}
					}
				}
			}
		}
	}

	mean_time_to_happen = {
		months = 12
		# modifiers
	}

	option = { # pirates get released
		name = "anb_piracy.1.a"
		# play as Rhasail
		trigger = {
			ai = no
		}
		1100 = {
			owner = {
				release = J95
			}
			change_province_name = "Rhasáil"
			rename_capital = "Rhasáil"
		}
		J95 = {
			change_technology_group = ROOT
			change_government = republic
			add_government_reform = pirate_republic_reform
			adopt_reform_progress = ROOT
			change_religion = ROOT
			change_primary_culture = ROOT
			discover_provinces = ROOT
			add_republican_tradition = 50
			capital_scope = {
				build_to_forcelimit = {
					infantry = 0.8
					artillery = 0.2
					heavy_ship = 0.3
					light_ship = 0.3
					transport = 0.3
				}
			}
			define_ruler = {
				name = "Thirendil Windsail"
				age = 126
				adm = 3
				dip = 4
				mil = 4
				culture = sea_elf
				religion = elven_forebears
			}
			swap_non_generic_missions = yes
		#	set_ruler_flag = historical_pirate
			set_ruler_flag = long_lived_ruler
			add_ruler_personality = legendary_pirate_personality			
			add_ruler_personality = kind_hearted_personality
		}
		hidden_effect = {
			set_ruler = Thirendil
			country_event = { id = anb_miscevents.22 } # new country setup
		}
	}
	option = { # play as the pirates
		name = "anb_piracy.1.b"
		1100 = {
			owner = {
				release = J95
			}
			change_province_name = "Rhasáil"
			rename_capital = "Rhasáil"
		}
		J95 = {
			change_technology_group = ROOT
			change_government = republic
			add_government_reform = pirate_republic_reform
			adopt_reform_progress = ROOT
			change_religion = ROOT
			change_primary_culture = ROOT
			add_republican_tradition = 50
			capital_scope = {
				build_to_forcelimit = {
					infantry = 0.8
					artillery = 0.2
					heavy_ship = 0.3
					light_ship = 0.3
					transport = 0.3
				}
			}
			define_ruler = {
				name = "Thirendil Windsail"
				age = 126
				adm = 3
				dip = 4
				mil = 4
				culture = sea_elf
				religion = elven_forebears
			}
			swap_non_generic_missions = yes
		#	set_ruler_flag = historical_pirate
			set_ruler_flag = long_lived_ruler
			add_ruler_personality = legendary_pirate_personality			
			add_ruler_personality = kind_hearted_personality
			
			hidden_effect = {
				set_ruler = Thirendil
				country_event = { id = anb_miscevents.22 } # new country setup
			}
		}
		switch_tag = J95
	}
}
