#Removing Damestear
province_event = {
	id = damestear.2
	title = damestear.2.t
	desc = damestear.2.d
	picture = TRADE_GOODS_METALS_AND_MINING_EQUIPENT_eventPicture
	goto = root
	
	mean_time_to_happen  = {
		years = 40
		
		modifier = {
			factor = 0.7
			REB = {
				check_variable = {
					which = num_damestear
					value = 15
				}
			}
		}
	}
	
	trigger = {
		trade_goods = damestear
		NOT = { has_province_modifier = skyfall_damestear } # We don't want the Damestear of Ardail to deplete
		NOT = { has_province_flag = permanent_damestear }
	}
	
	option = {
        name = damestear.2.a
        ai_chance = { factor = 100 }
		ROOT = {
			change_trade_goods = unknown
			remove_province_modifier = damestear_rich_deposit
		}
		hidden_effect = {
			clr_province_flag = specialized_hold
			remove_province_modifier = hold_foundry
			remove_province_modifier = hold_foundry_2
			
		}
		hidden_effect = {
			REB = {
				subtract_variable = {
					which = num_damestear
					value = 1
				}
			}
			set_rem_damestear_count = yes
		}
	}
}


country_event = {
	id = damestear.3
	title = damestear.3.t
	desc = damestear.3.d
	picture = FEAST_eventPicture

	mean_time_to_happen  = {
		years = 100
	}
	
	trigger = {
		damestear_can_spawn = yes
	}
	
	option = {
        name = damestear.3.a
        ai_chance = { factor = 100 }
		random_owned_province = {
			limit = {
				is_city = yes
				NOT = {
					trade_goods = precursor_relics
					trade_goods = damestear
					trade_goods = gold
					trade_goods = gems
					has_province_flag = damestear_no_spawn
					has_province_flag = has_damestear
					has_terrain = dwarven_hold
					has_terrain = dwarven_hold_surface
				}
			}
			change_trade_goods = damestear
			set_province_flag = has_damestear
		}
		set_add_damestear_count = yes
	}
}

country_event = {
	id = damestear.4
	title = damestear.4.t
	desc = damestear.4.d
	picture = COMET_SIGHTED_eventPicture
	goto = damestear_province

	mean_time_to_happen  = {
		years = 300
	}
	
	trigger = {
		damestear_can_spawn = yes
	}
	
	immediate = {
		hidden_effect = {
			random_owned_province = {
				limit = {
					is_city = yes
					NOT = {
						trade_goods = precursor_relics
						trade_goods = damestear
						trade_goods = gold
						has_province_flag = has_damestear
						has_terrain = dwarven_hold
						has_terrain = dwarven_hold_surface
						has_terrain = cavern
						has_terrain = dwarven_road
					}
				}
				save_event_target_as = damestear_province
			}
		}
	}
	
	option = {
        name = damestear.4.a
        ai_chance = { factor = 100 }
		event_target:damestear_province = {
			add_base_tax = -2
			add_base_production = -2
			add_base_manpower = -2
			add_permanent_province_modifier = {
				name = damestear_rich_deposit
				duration = -1
			}
			change_trade_goods = damestear
			set_province_flag = has_damestear
		}
		set_add_damestear_count = yes
	}
}
