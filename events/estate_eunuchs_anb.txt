#Eunuchs encroach on Government Lands
country_event = {
	id = eunuchs_estate_anb_events.3
	title = eunuchs_estate_anb_events.3.t
	desc = eunuchs_estate_anb_events.3.d
	picture = NOBLE_ESTATE_eventPicture
	goto = eunuch_encroachment
	
	trigger = {
		has_estate = estate_eunuchs_anb
		num_of_cities = 15
		any_owned_province = {
			is_overseas = no
			is_capital = no
			is_colony = no
			has_seat_in_parliament = no
			is_territory = no
		}
	}
	
	is_triggered_only = yes
	
	immediate = {
		hidden_effect = {
			random_owned_province = {
				limit = {
					is_capital = no
					is_overseas = no
					is_colony = no
					has_seat_in_parliament = no
					is_territory = no
				}
				save_event_target_as = eunuch_encroachment
			}
		}
	}
	
	mean_time_to_happen = {
		days = 1
		modifier = {
			factor = 2.0
			estate_influence = {
				estate = estate_eunuchs_anb
				influence = 80
			}
			NOT = {
				estate_loyalty = {
					estate = estate_eunuchs_anb
					loyalty = 80
				}
			}
		}
	}
	
	option = {
		name = eunuchs_estate_anb_events.3.a #Allow it.
		ai_chance = {
			factor = 50
			modifier = {
				factor = 0.25
				estate_territory = {
					estate = estate_eunuchs_anb
					territory = 12
				}
			}
			modifier = {
				factor = 0.25
				estate_influence = {
					estate = estate_eunuchs_anb
					influence = 60
				}
			}
			modifier = {
				factor = 1.5
				NOT = {
					estate_loyalty = {
						estate = estate_eunuchs_anb
						loyalty = 30
					}
				}
			}
		}
		event_target:eunuch_encroachment = {
			add_local_autonomy = 25
		}
		give_estate_land_share_small = { estate = estate_eunuchs_anb }
	}
	option = {
		name = eunuchs_estate_anb_events.3.b #Take it Back.
		ai_chance = {
			factor = 50
			modifier = {
				factor = 1.1
				NOT = {
					estate_territory = {
						estate = estate_eunuchs_anb
						territory = 15
					}
				}
			}
			modifier = {
				factor = 0.25
				NOT = {
					estate_loyalty = {
						estate = estate_eunuchs_anb
						loyalty = 35
					}
				}
			}
		}
		event_target:eunuch_encroachment = {
			random_list = {
				50 = {
					spawn_rebels = {
						type = noble_rebels
						size = 1
					}
				}
				50 = {  }
			}
		}
		add_estate_loyalty = {
			estate = estate_eunuchs_anb
			loyalty = -15
		}
	}
}

	trigger = {
		has_estate = estate_eunuchs_anb
		estate_influence = {
			estate = estate_eunuchs_anb
			influence = 70
		}
		any_owned_province = {
			OR = {
				development = 15
				province_trade_power = 8
			}
			is_city = yes
			is_capital = no
		}
		NOT = {
			has_estate_influence_modifier = {
				estate = estate_eunuchs_anb
				modifier = EST_VAL_EUNUCH_RIGHTS_EXTENDED
			}
		}
		NOT = {
			has_estate_influence_modifier = {
				estate = estate_eunuchs_anb
				modifier = EST_VAL_EUNUCH_RIGHTS_REFUSED
			}
		}
	}

	mean_time_to_happen = {
		days = 1
	}

		hidden_effect = {
			while = {
				limit = {
					NOT = { 
						development_in_provinces = {
							has_province_flag = extended_eunuch_privileges
							value = estate
							estate = estate_eunuchs_anb
						}

						OR = {
							development = 15
							province_trade_power = 8
						}

			influence = 10
			duration = 5475
		}

			limit = {
				has_province_flag = extended_eunuch_privileges

		}
	}

		name = eunuchs_estate_anb_events.7.b #
