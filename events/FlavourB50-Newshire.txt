#Culture Shift
country_event = {
	id = flavor_newshire.2
	title = flavor_luciande.2.t
	desc = flavor_luciande.2.d
	picture = EUROPEAN_REFUGEES_eventPicture
	
	fire_only_once = yes
	
	trigger = {
		tag = B34
		NOT = { has_country_modifier = legacy_of_adventurers_modifier }
		
		num_of_cities = 5
	}
	
	mean_time_to_happen = {
		months = 800
		modifier = {
			factor = 0.7
			stability = 1
		}
		modifier = {
			factor = 0.5
			num_of_cities = 10
		}
		modifier = {
			factor = 0.2
			in_golden_age = yes
		}
	}
	
	immediate = {
		hidden_effect = {
			if = {
				limit = {
					culture = ROOT
					any_owned_province = {
						superregion = escann_superregion
						region = forlorn_vale_region
					}
				}
				random_owned_province = {
					limit = {
						OR = {
							superregion = escann_superregion
							region = forlorn_vale_region
						}
						culture = ROOT
					}
					save_event_target_as = culture_shift_province
				}
			}
		}
	}
	
	# Birth of a New People
	option = {		
		name = flavor_luciande.1.a
		ai_chance = { factor = 65 }	
		
		event_target:culture_shift_province = {
			area = {
				limit = {
					owned_by = ROOT
					culture = ROOT
				}
				change_culture = newfoot_halfling
			}
		}
		
		add_country_modifier = {
			name = birth_of_a_new_people
			duration = 3650
		}
		
		change_primary_culture = newfoot_halfling
		
		if = {
			limit = { ruler_culture = culture_shift_province }
			set_ruler_culture = newfoot_halfling
		}
		if = {
			limit = { heir_culture = culture_shift_province }
			set_heir_culture = newfoot_halfling
		}
		if = {
			limit = { consort_culture = culture_shift_province }
			set_consort_culture = newfoot_halfling
		}
	}
	
	option = { # Reject Freedom
		name = flavor_luciande.1.b
		trigger = {
			culture = ROOT
			ai = no
		}
		event_target:culture_shift_province = {
			change_culture = newfoot_halfling
		}
	}
}
