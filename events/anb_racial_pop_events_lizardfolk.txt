#Event Manager
country_event = {
	id = racial_pop_events_lizardfolk.0
	title = racial_pop_events_lizardfolk.0.t
	desc = racial_pop_events_lizardfolk.0.d
	picture = DEBATE_REPUBLICAN_eventPicture
	
	hidden = yes
	
	is_triggered_only = yes

	trigger = {
		if = {
			limit = { ai = no }
			OR = {
				fast_owns_num_pops_of_race = { race = lizardfolk amount = 1 }
				any_neighbor_country = { fast_owns_num_pops_of_race = { race = lizardfolk amount = 1 } }
			}
		}
		else = {
			fast_owns_num_pops_of_race = { race = lizardfolk amount = 1 }
		}
	}
	
	mean_time_to_happen = {
		days = 1
		modifier = {
			factor = 10
			has_country_modifier = racial_focus_lizardfolk
		}
	}
	
	option = {
		name = racial_pop_events_lizardfolk.0.a
		random_list = {
			2 = { }
			# ensure there's an empty option first in case none of the others are available
			# and also because this one should be picked rarely even if the others are available

			#ai external migration simulation
			10 = {
				trigger = {
					ai_external_migration_trigger = { race = lizardfolk }
				}
				modifier = {
					factor = 0.5
					all_neighbor_country = {
						low_tolerance_lizardfolk_race_trigger = yes
					}
				}
				ai_pop_migration_leave = { race = lizardfolk }
			}
			#events 1-3 are reserved for making external migration stuff function
			15 = { #random pop growth
				trigger = {
					any_owned_province = {
						has_small_lizardfolk_minority_trigger = yes
						OR = {
							is_prosperous = yes
							development = 20
						}
					}
				}
				random_owned_province = {
					limit = {
						has_small_lizardfolk_minority_trigger = yes
						OR = {
							is_prosperous = yes
							development = 20
						}
					}
					province_event = { id = racial_pop_events_lizardfolk.4 }
				}
			}
			15 = { #random pop shrinkage
				trigger = {
					if = { limit = { ai = yes } fast_owns_num_pops_of_race = { race = lizardfolk amount = 2 } }
					any_owned_province = {
						has_lizardfolk_minority_trigger = yes
						OR = {
							unrest = 10
							devastation = 25
							has_oppressed_lizardfolk_minority_trigger = yes
						}
					}
				}
				modifier = {
					factor = 0.67
					ai = yes
				}
				modifier = {
					factor = 2
					OR = {
						has_country_modifier = racial_pop_lizardfolk_purge
						has_country_modifier = racial_pop_lizardfolk_expulsion
					}
				}
				random_owned_province = {
					limit = {
						has_lizardfolk_minority_trigger = yes
						OR = {
							unrest = 10
							devastation = 25
							has_oppressed_lizardfolk_minority_trigger = yes
						}
					}
					province_event = { id = racial_pop_events_lizardfolk.5 }
				}
			}
			5 = { #Internal migration occurs
				trigger = {
					pop_internal_migration_trigger = { race = lizardfolk }
				}
				modifier = {
					factor = 2
					fast_owns_num_pops_of_race = { race = lizardfolk amount = 4 }
				}
				country_event = { id = racial_pop_events_lizardfolk.6 }
			}
			1 = { #Migration from neighboring country
				trigger = {
					pop_external_migration_trigger = { neighbor = yes race = lizardfolk }
				}
				modifier = {
					factor = 2
					has_country_modifier = racial_focus_lizardfolk
				}
				modifier = {
					factor = 2
					AND = {
						has_country_modifier = racial_focus_lizardfolk
						NOT = { fast_owns_num_pops_of_race = { race = lizardfolk amount = 1 } }
					}
				}
				modifier = {
					factor = 1.5
					num_of_cities = 10
					NOT = { fast_owns_num_pops_of_race = { race = lizardfolk amount = 5 } }
				}
				modifier = {
					factor = 1.5
					num_of_cities = 25
					NOT = { fast_owns_num_pops_of_race = { race = lizardfolk amount = 10 } }
				}
				modifier = {
					factor = 1.5
					num_of_cities = 50
					NOT = { fast_owns_num_pops_of_race = { race = lizardfolk amount = 25 } }
				}
				modifier = {
					factor = 1.5
					num_of_cities = 100
					NOT = { fast_owns_num_pops_of_race = { race = lizardfolk amount = 50 } }
				}
				modifier = {
					factor = 0.5
					low_tolerance_lizardfolk_race_trigger = yes
				}
				modifier = {
					factor = 2
					culture_is_lizardfolk = yes
				}
				
				country_event = { id = racial_pop_events_lizardfolk.7 }
			}
			#10 = { #Migration from overlord
			#	trigger = {
			#		pop_external_migration_trigger = { overlord = yes race = lizardfolk }
			#	}
			#	choose_pop_destination_province = { race = lizardfolk }
			#	event_target:destination_province = { province_event = { id = racial_pop_events_lizardfolk.8 } }
			#}
			#10 = { #Migration from vassal
			#	trigger = {
			#		pop_external_migration_trigger = { vassal = yes race = lizardfolk }
			#	}
			#	choose_pop_destination_province = { race = lizardfolk }
			#	event_target:destination_province = { province_event = { id = racial_pop_events_lizardfolk.9 } }
			#}
			#events 100-200 are reserved for custom migration events, put their triggers here
		}
	}
}
