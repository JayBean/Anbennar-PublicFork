### Venail receive demand
country_event = {
	id = venail.6
	title = venail.6.t
	desc = venail.6.d
	picture = DIPLOMACY_eventPicture
	
	is_triggered_only = yes
	
	trigger = {
		always = yes
	}
	
	### Pay now in full
	option = {
        name = venail.6.a
        ai_chance = { factor = 25 }
		add_treasury = -3000
		A25 = { add_treasury = 3000 }
		A25 = { country_event = { id = venail.7 } }
	}
	
	###Slow repayment
	option = {
		name = venail.6.b
        ai_chance = { factor = 65 }
		add_country_modifier = {
			name = venail_repay_loan
			duration = 27375
		}
		A25 = {
			add_country_modifier = {
				name = venail_damescrown_receive_loan
				duration = 27357
			}
			country_event = { id = venail.7 }
		}
	}
	###Give them big port
	option = {
		name = venail.6.c
        ai_chance = { factor = 0 }
		trigger = {
			any_owned_province = {
				has_province_modifier = venail_cestircel
			}
		}
		random_province = {
			limit = {
				has_province_modifier = venail_cestircel
			}
			cede_province = A25
			add_core = A25
			remove_core = ROOT
		}
		A25 = {
			country_event = { id = venail.7 }
			set_country_flag = repayment_with_port
		}
	}
	### Refuse
	option = {
		name = venail.6.e
        ai_chance = { factor = 10 }
		every_country = {
			limit = {
				capital_scope = { superregion = western_cannor_superregion }
			}
			add_trust = {
				who = ROOT
				value = -45
			}
		}
		add_historical_rival = A25
		A25 = {
			set_country_flag = repayment_denied
			country_event = {id = venail.7}
		}
	}
}

### Damescrown response
country_event = {
	id = venail.7
	title = venail.7.t
	desc = {
		trigger = { NOT = { has_country_flag = repayment_denied } }
		desc = venail.7.d.a
	}
	desc = {
		trigger = { has_country_flag = repayment_denied }
		desc = venail.7.d.b
	}
	picture = GREAT_BUILDING_eventPicture
	
	is_triggered_only = yes
	
	trigger = {
		always = yes
	}
	
	### Payment done
	option = {
		trigger = {
			NOT = { has_country_flag = repayment_denied }
		}
        name = venail.7.a
        ai_chance = { factor = 0 }
		add_prestige = 25
	}
	### Investing options
	# Develop
	option = {
		trigger = {
			NOT = { has_country_flag = repayment_denied }
			NOT = { has_country_flag = repayment_with_port }
		}
		if = {
			limit = { has_country_modifier = venail_damescrown_receive_loan }
			remove_country_modifier = venail_damescrown_receive_loan
		}
		else = { add_treasury = -3000 }
        name = venail.7.b
        ai_chance = { factor = 5 }
		add_prestige = 25
		capital_scope = {
			add_base_manpower = 2
			add_base_production = 2
			add_base_tax = 2
			area = {
				add_base_manpower = 1
				add_base_production = 1
				add_base_tax = 1
			}
		}
	}
	# Gain Exclusive trade
	option = {
		trigger = {
			NOT = { has_country_flag = repayment_denied }
			NOT = { has_country_flag = repayment_with_port }
		}
		if = {
			limit = { has_country_modifier = venail_damescrown_receive_loan }
			remove_country_modifier = venail_damescrown_receive_loan
		}
		else = { add_treasury = -3000 }
        name = venail.7.c
        ai_chance = { factor = 95 }
		add_prestige = 25
		capital_scope = {
			add_base_manpower = 2
			add_base_production = 2
			add_base_tax = 2
			change_trade_goods = precursor_relics
			set_province_flag = permanent_precursor_relics
			add_province_modifier = {
				name = aelnar_exclusive_trade
				duration = -1
			}
		}
	}
	### Refused payment
	option = {
		trigger = {
			has_country_flag = repayment_denied
		}
		name = venail.7.e
        ai_chance = { factor = 100 }
		clr_country_flag = repayment_denied
		add_prestige = -25
		random_province = {
			limit = {
				has_province_modifier = venail_cestircel
			}
			add_permanent_claim = ROOT
		}
		add_historical_rival = A21
		add_historical_rival = Z43
		Z43 = { 
			add_country_modifier =  {
				name = mischievious_elf
				duration = -1
			}
		}
		A21 = { 
			add_country_modifier =  {
				name = mischievious_elf
				duration = -1
			}
		}
	}
}
