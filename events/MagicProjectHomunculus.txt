country_event = {
	id = magic_project_homunculus.6
	title = magic_project_homunculus.6.t
	desc = magic_project_homunculus.6.d
	picture = wihgfx_FEMALES_eventPicture
	
	trigger = {
		has_ruler_flag = magic_project_homunculus_5
		NOT = { has_ruler_flag = magic_project_homunculus_complete }
		is_lesser_in_union = no
		NOT = { has_country_flag = anti_homunculus_cheese } #Anti homunculus cheese with magical heir
	}
	
	mean_time_to_happen = {
		months = 12
		modifier = {
			factor = 2
			has_ruler_flag = homunculus_current_humanlike
		}
		modifier = {
			factor = 0.05
			had_ruler_flag = {
				flag = magic_project_homunculus_5
				days = 1095
			}
		}
	}

	immediate = { #Anti homunculus cheese with magical heir
		hidden_effect = {
			set_country_flag = anti_homunculus_cheese
		}
	}

	option = {	#A familiar to aid in my magical research
		name = magic_project_homunculus.6.a
		trigger = {
			NOT = { has_country_flag = ruler_studying_homunculus_familiar }
		}
		ai_chance = {
			factor = 50
			modifier = {
				factor = 2
				OR = {
					ruler_has_personality = scholar_personality
					ruler_has_personality = careful_personality
					ruler_has_personality = cruel_personality
				}
			}
		}
		
		set_country_flag = ruler_studying_homunculus_familiar
		custom_tooltip = tooltip_homunculus_familiar
		
		set_ruler_flag = magic_project_homunculus_complete
		custom_tooltip = tooltip_magic_project_complete
		
		magic_project_clear_homunculus_flags = yes
	}
	
	option = {	#General
		name = magic_project_homunculus.6.b
		ai_chance = {
			factor = 50
			modifier = {
				factor = 2
				is_at_war = yes
			}
			modifier = {
				factor = 2
				personality = ai_militarist
			}
			modifier = {
				factor = 0.25
				num_of_generals = 3
			}
		}
		
		create_general = {
			tradition = 75
		}
		
		set_ruler_flag = magic_project_homunculus_complete
		custom_tooltip = tooltip_magic_project_complete
		
		magic_project_clear_homunculus_flags = yes
		
	}
	
	option = {	#Admiral
		name = magic_project_homunculus.6.c
		ai_chance = {
			factor = 50
			modifier = {
				factor = 2
				is_at_war = yes
			}
			modifier = {
				factor = 2
				personality = ai_militarist
			}
			modifier = {
				factor = 0	
				NOT = { navy_size = 10 }
			}
			modifier = {
				factor = 0.25
				num_of_admirals = 3
			}
		}
		
		create_admiral = {
			tradition = 75
		}
		
		magic_project_clear_homunculus_flags = yes
		
		set_ruler_flag = magic_project_homunculus_complete
		custom_tooltip = tooltip_magic_project_complete
	}
	
	option = {	#Masquerade them as an heir of my house
		name = magic_project_homunculus.6.dd
		trigger = {
			has_ruler_flag = homunculus_current_humanlike
			has_government_attribute = heir
			NOT = { has_reform = magocracy_reform }
			NOT = { has_reform = magisterium_reform }
			NOT = { has_reform = states_general_reform }	#mamluk and iqta also apply here
			NOT = { has_ruler_flag = xhazob_patron }
		}
		ai_chance = {
			factor = 50
			modifier = {
				factor = 0.25
				OR = {
					heir_adm = 3
					heir_dip = 3
					heir_mil = 3
				}
			}
		}
		remove_heir = {  }
		
		if = {
			limit = {
				OR = {
					has_ruler_flag = homunculus_current_male
					has_ruler_flag = homunculus_current_androgyne
				}
			}
			define_heir = {
				dynasty = ROOT
				age = 18
				change_adm = 4
				change_dip = 4
				change_mil = 4
				claim = 25
				no_consort_with_heir = yes
				male = yes
			}
		}
		if = {
			limit = {
				has_ruler_flag = homunculus_current_female
			}
			define_heir = {
				dynasty = ROOT
				age = 18
				change_adm = 4
				change_dip = 4
				change_mil = 4
				claim = 25
				no_consort_with_heir = yes
				female = yes	
			}
		}
		hidden_effect = { set_heir_flag = set_immortality }
		
		set_heir_flag = is_a_secret_homunculus
		
		set_ruler_flag = magic_project_homunculus_complete
		custom_tooltip = tooltip_magic_project_complete
		
		magic_project_clear_homunculus_flags = yes
	}
	
	
	option = {	#Make em my consort
		name = magic_project_homunculus.6.e
		trigger = {
			has_ruler_flag = homunculus_current_humanlike
			has_consort = no
			OR = {
				government = monarchy
				has_government_attribute = royal_marriage
			}
			NOT = { has_reform = states_general_reform }	#mamluk and iqta also apply here
			
			OR = {
				AND = {
					is_female = yes
					OR = {
						has_ruler_flag = homunculus_current_male
						has_ruler_flag = homunculus_current_androgyne
					}
				}
				AND = {
					is_female = no
					OR = {
						has_ruler_flag = homunculus_current_female
						has_ruler_flag = homunculus_current_androgyne
					}
				}
			}
		}
		ai_chance = {
			factor = 50
			modifier = {
				factor = 0.25
				OR = {
					consort_adm = 3
					consort_dip = 3
					consort_mil = 3
				}
			}
		}

		remove_consort_effect = yes
		
		if = {
			limit = {
				is_female = no
			}
			define_consort = {
				age = 18
				adm = 4
				dip = 4
				mil = 4
				female = yes
			}
		}
		else_if = {
			limit = {
				is_female = yes
			}
			define_consort = {
				dynasty = ROOT
				age = 18
				adm = 4
				dip = 4
				mil = 4
				male = yes	
			}
		}
		hidden_effect = { set_consort_flag = set_immortality }
		
		set_consort_flag = is_a_secret_homunculus
		
		set_ruler_flag = magic_project_homunculus_complete
		custom_tooltip = tooltip_magic_project_complete
		
		magic_project_clear_homunculus_flags = yes
	}
	
	option = {	#Advisor
		name = magic_project_homunculus.6.f
		ai_chance = {
			factor = 50
		}
		
		country_event = { id = magic_project_homunculus.7 days = 1 }
		
		set_ruler_flag = magic_project_homunculus_complete
		custom_tooltip = tooltip_magic_project_complete
	}

	after = { #Anti homunculus cheese with magical heir
		clr_country_flag = anti_homunculus_cheese
		set_ruler_flag = magic_project_homunculus_complete_achievement
	}
}
