# Oren Nayiru Intro/Warning Event
country_event = {
	id = oren_nayiru.1
	title = oren_nayiru.1.t #First Contact
	desc = oren_nayiru.1.d
	picture = SIEGE_eventPicture #TODO
	
	fire_only_once = yes
	trigger = {
		NOT = {
			OR = {
				has_global_flag = oren_nayiru_intro
				has_global_flag = oren_nayiru_intro_early
			}
		}
		OR = {
			exists = H66 #Jaherian Exemplars
			exists = H67 #Rezankand
			is_year = 1550
		}
		any_owned_province = { superregion = taychend_superregion }
	}

	mean_time_to_happen = { days = 1 }
	
	immediate = {
		hidden_effect = {
			every_country = {
				limit = {
					any_owned_province = { superregion = taychend_superregion }
					NOT = { tag = ROOT }
				}
				country_event = { id = oren_nayiru.910 }
			}
			if = {
				limit = { is_year = 1500 }
				set_global_flag = oren_nayiru_intro
			}
			else = {
				set_global_flag = oren_nayiru_intro_early # prevents oren nayiru from spawning in the 1400s (too early)
			}
		}
	}
	
	option = {
		name = oren_nayiru.1.a
	}
}

# The Calsithara Revolt
country_event = {
	id = oren_nayiru.4
	title = oren_nayiru.4.t #The Calsithara Revolt
	desc = oren_nayiru.4.d
	picture = SIEGE_eventPicture #TODO
	
	mean_time_to_happen = { days = 1 }
	fire_only_once = yes
	
	trigger = {
		is_religion_enabled = oren_nayiru
		NOT = { OR = { religion = oren_nayiru religion = oren_nayiru_reformed } }
		any_owned_province = { superregion = taychend_superregion }
		2502 = { #Uesrennu
			check_variable = {
				which = taychend_oren_nayiru_provinces
				value = 35
			}
		}
	}
	
	immediate = {
		hidden_effect = {
			every_country = {
				limit = {
					any_owned_province = { superregion = taychend_superregion }
					NOT = { OR = { religion = oren_nayiru religion = oren_nayiru_reformed } }
					NOT = { tag = ROOT }
				}
				country_event = {
					id = oren_nayiru.940
				}
			}
		}
	}
	
	option = {
		name = oren_nayiru.4.a
		taychend_superregion = {
			limit = { 
				religion = oren_nayiru
				is_reformation_center = no
                owner = { NOT = { religion = oren_nayiru } }
				NOT = { province_id = 2502 } #Uesrennu
			}
			if = {
				limit = { 2502 = { check_variable = { which = onVariable value = 2 } } }
				oren_nayiru_rebels = 1
				remove_province_modifier = religious_zeal_at_conv
				set_province_flag = calsithara_revolt_province_flag
				2502 = { set_variable = { which = onVariable value = 0 } }
			}
			else = {
				2502 = { change_variable = { which = onVariable value = 1 } }
			}
		}
		if = {
			limit = {
				2502 = { owner = { NOT = { religion = oren_nayiru } } }
			}
			2502 = { #Uesrennu
				spawn_rebels = {
					type = oren_nayiru_rebels
					size = 3
					leader = "Calsithara Flamespeaker"
				}
				set_province_flag = calsithara_revolt_leader_province_flag
			}
		}
		else = {
			random_province = {
				limit = {
					superregion = taychend_superregion
					religion = oren_nayiru
					owner = { NOT = { religion = oren_nayiru } }
					NOT = { has_province_flag = calsithara_revolt_province_flag }
				}
				spawn_rebels = {
					type = oren_nayiru_rebels
					size = 3
					leader = "Calsithara Flamespeaker"
				}
				set_province_flag = calsithara_revolt_leader_province_flag
			}
		}
	}
}
