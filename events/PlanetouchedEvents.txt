# Turning ruler into planetouched commander.

	id = planetouched.100
	title = planetouched.100.t
	desc = planetouched.100.d
	picture = BIG_BOOK_eventPicture

	is_triggered_only = yes

		}

				fire = 3
				shock = 6
				manuever = 5
				siege = 2

			}
		}
		else_if = {
			limit = { mil = 3 }

				fire = 2
				shock = 5
				manuever = 4
				siege = 2
				trait = planetouched_commander_personality

		}

				fire = 2

# Young Heir turns to be planetouched
country_event = {
	id = planetouched.101
	title = planetouched.101.t
    desc = planetouched.101.d
    picture = DIPLOMACY_eventPicture

	trigger = {
		heir_can_be_planetouched_monarchy = yes
	}

	mean_time_to_happen = {
		#years = 100
		years = 75 #once we have more stuff interacting with these modifiers we'll go back to 100

		modifier = {
			factor = 0.5
			planetouched_high_affinity = yes
		}
		modifier = {
			factor = 0.75
			planetouched_medium_affinity = yes
		}
		modifier = {
			factor = 1.25
			planetouched_low_affinity = yes
		}
		modifier = {
			factor = 0.8
				has_dlc = "Leviathan"
				owns = 383
				383 = { has_great_project = { type = aur_kes_akasik tier = 3 } }
		}
	}

	option = {
		name = planetouched.101.a
		ai_chance = { factor = 100 }

		set_heir_random_planetouched_effect = yes
	}
}

# Planetouched commander
country_event = {
	id = planetouched.103
	title = planetouched.103.t
    desc = planetouched.103.d
    picture = DIPLOMACY_eventPicture

	trigger = {
		capital_scope = { continent = africa }
	}

	mean_time_to_happen = {
		years = 75

		modifier = {
			factor = 0.5
			planetouched_high_affinity = yes
		}
		modifier = {
			factor = 0.75
			planetouched_medium_affinity = yes
		}
		modifier = {
			factor = 1.25
			planetouched_low_affinity = yes
		}
		modifier = {
			factor = 0.9
			has_country_flag = increased_planetouched_commander_chance_high_affinity
		}
	}

	immediate = {
		hidden_effect = {
			set_saved_name = {
				key = root_name_male
				type = advisor
				scope = ROOT
			}
		}
	}

	option = {
		name = planetouched.103.a
		ai_chance = { factor = 100 }

        #create_planetouched_general_effect = yes
		if = {
			limit = { army_tradition = 80 }
			define_general = {
				fire = 5
				shock = 5
				manuever = 5
				siege = 3
				trait = planetouched_commander_personality
				name = root_name_male
			}
		}
		else_if = {
			limit = { army_tradition = 60 }
			define_general = {
				fire = 4
				shock = 5
				manuever = 5
				siege = 2
				trait = planetouched_commander_personality
				name = root_name_male
			}
		}
		else_if = {
			limit = { army_tradition = 40 }
			define_general = {
				fire = 3
				shock = 5
				manuever = 4
				siege = 1
				trait = planetouched_commander_personality
				name = root_name_male
			}
		}
		else_if = {
			limit = { army_tradition = 20 }
			define_general = {
				fire = 3
				shock = 4
				manuever = 3
				siege = 1
				trait = planetouched_commander_personality
				name = root_name_male
			}
		}
		else = {
			define_general = {
				fire = 2
				shock = 3
				manuever = 3
				siege = 1
				trait = planetouched_commander_personality
				name = root_name_male
			}
		}
		hidden_effect = { clear_saved_name = root_name_male }
	}
}
