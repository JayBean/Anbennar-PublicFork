country_event = { #migration logic
	id = cardesti_events.2
	title = cardesti_events.2.t
	desc = cardesti_events.2.d
	picture = EUROPEAN_REFUGEES_eventPicture

	is_triggered_only = yes

	hidden = yes

	option = {
		name = cardesti_events.2.a
		random_province = {
			limit = {
				continent = europe
				NOT = {
					is_island = yes
				}
				if = {
					limit = {
						NOT = { is_year = 1494 }
					}
					NOT = { superregion = escann_superregion }
				}
				NOT = {
					has_province_modifier = cardesti_minority_modifier
					superregion = deepwoods_superregion
					superregion = deepwoods_portal_superregion
					owner = {
						OR = {
							has_country_modifier = monstrous_nation
							has_country_modifier = racial_pop_human_purge
							has_country_modifier = racial_pop_human_expulsion
							has_country_flag = cardesti_migrating_out
						}
					}
				}
				is_city = yes
			}
			add_province_modifier = {
				name = cardesti_minority_modifier
				duration = -1
			}
			owner = {
				if = {
					limit = {
						NOT = { has_country_flag = cardesti_no_notice }
					}
					country_event = { id = cardesti_events.6 }
				}
			}
		}
	}
}

					owner = {
						OR = {
							has_country_modifier = monstrous_nation
							has_country_modifier = racial_pop_human_purge
							has_country_modifier = racial_pop_human_expulsion
						}

				}
				is_city = yes
			}
			add_province_modifier = {
				name = cardesti_minority_modifier
				duration = -1
			}
			owner = {
				if = {
					limit = {
						NOT = { has_country_flag = cardesti_no_notice }
					}
					country_event = { id = cardesti_events.6 }
				}
			}
			clear_global_event_target = cardesti_core_province
			save_global_event_target_as = cardesti_core_province

		europe = {
			limit = { NOT = { province_id = event_target:cardesti_core_province } }
			remove_province_modifier = cardesti_minority_modifier
			cardesti_remove_event_modifiers_scripted_effect = yes
		}
		country_event = { id = cardesti_events.1 }
	}
}
