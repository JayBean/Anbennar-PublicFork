country_event = {
	id = flavor_azka_sur.7 #Gnolls in the Sad Sur
	title = flavor_azka_sur.7.t
	desc = flavor_azka_sur.7.d
	picture = CULTURE_eventPicture
	
	fire_only_once = yes
	is_triggered_only = yes
	
	option = {
		name = flavor_azka_sur.7.a #
		ai_chance = {
			factor = 50
		}
		every_owned_province = {
			limit = {
				culture = east_sandfang_gnoll
				OR = {
					area = harklum_area
					area = yrkur_area
					area = garpix_tluukt_area
					area = grixekyr_area
				}
				
			}
			change_culture = zanite
			change_religion = bulwari_sun_cult
			add_nationalism = -20
			add_devastation = 50
		}
		every_owned_province = {
			limit = {
				culture = east_sandfang_gnoll
				OR = {
					area = zansap_area
					area = sur_mountains_area
					area = garpix_ukraz_area
				}
			}
			change_culture = brasanni
			change_religion = bulwari_sun_cult
			add_nationalism = -20
			add_devastation = 50
		}
	}
	
	option = {
		name = flavor_azka_sur.7.a #
		ai_chance = {
			factor = 50
		}
		add_accepted_culture = east_sandfang_gnoll 
		add_dip_power = 150 
	}
}
