country_event = {
	id = flavor_kalsyto.1011
	title = flavor_kalsyto.1011.t
	desc = flavor_kalsyto.1011.d
	picture = DIPLOMACY_eventPicture
	
	hidden = yes
	
	trigger = {
		tag = J33
		any_subject_country = {
			OR = {
				AND = {
					is_subject_of_type = democratic_sphere_of_influence_subject
					NOT = { government = republic }
				}
				AND = {
					has_country_flag = J33_CDS_flag
					kalsyto_not_CDS = yes
				}
			}
		}
	}
	
	mean_time_to_happen = {
		months = 1
		modifier = {
			factor = 0.1
			stability = 1
		}
	}
	
	option = {
		name = flavor_kalsyto.1011.a
		every_subject_country = {
			limit = {
				is_subject_of_type = democratic_sphere_of_influence_subject
				NOT = { government = republic }
			}
			change_government = republic
		}
		every_subject_country = {
			limit = {
				has_country_flag = J33_CDS_flag
				kalsyto_not_CDS = yes
			}
			add_government_reform = lake_consolidated_state
		}
	}
}
