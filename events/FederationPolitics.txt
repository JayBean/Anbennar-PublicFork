#PoliticalMenu
country_event = {
	id = fed_politics.1
	title = fed_politics.1.t
	desc = fed_politics.1.d
	picture = DECLARATION_INDEPENDENCE_eventPicture
	
	is_triggered_only = yes
	
	trigger = {
		has_country_modifier = lake_federation_member
	}
	
	immediate = {
		hidden_effect = { set_country_flag = federation_politics_menu_open }
	}
	
	after = {
		hidden_effect = {
			clr_country_flag = federation_politics_menu_open
		}
	}
	
	#Request Federation Aid
	option = {
		name = fed_politics.1.a
		ai_chance = { 
			factor = 50
			
			modifier = {
				factor = 0
				NOT = { federation_has_high_standing = yes }
			}
			modifier = {
				factor = 0
				stability = 2
			}
		}
		lake_federation_lose_2_points = yes
		lake_federation_lose_1_politics = yes
		add_adm_power = 25
		add_dip_power = 25
		add_mil_power = 25
		tooltip = {
			random = {
				chance = 20
				add_stability_or_adm_power = yes
			}
		}
		hidden_effect = { country_event = { id = fed_politics.100 } }
		if = {
			limit = { ai = yes }
			add_stability_or_adm_power = yes
		}
		hidden_effect = { set_country_flag = recent_federation_politic }
	}
	#Share Granaries
	option = {
		name = fed_politics.1.b
		ai_chance = {
			factor = 100
		}
		trigger = { 
			NOT = { has_country_modifier = federation_shared_granaries }
			any_neighbor_country = {
				has_country_modifier = lake_federation_member
				NOT = { is_rival = ROOT }
			}
			
		}
		lake_federation_lose_1_politics = yes
		lake_federation_gain_1_points = yes
		add_country_modifier = {
			name = federation_shared_granaries
			duration = 1825
		}
		random_neighbor_country = {
			limit = { 
				has_country_modifier = lake_federation_member
				NOT = { is_rival = ROOT }
			}
			hidden_effect = { add_yearly_manpower = 1 }
			add_opinion = {
				who = ROOT
				modifier = federation_pos_intrigue
			}
		}
		hidden_effect = { set_country_flag = recent_federation_politic }
	}
	
	#Scholarly Cooperation
	option = {
		name = fed_politics.1.c
		ai_chance = {
			factor = 1000
		}
		trigger = { 
			any_neighbor_country = {
				has_country_modifier = lake_federation_member
				NOT = { has_country_modifier = federation_shared_institution }
				ROOT = {
					institution_difference = {
						who = PREV
						value = 1
					}
				}
			}
		}
		lake_federation_lose_1_politics = yes
		lake_federation_gain_2_points = yes
		add_dip_power = -75
		random_neighbor_country = {
			limit = {
				has_country_modifier = lake_federation_member
				NOT = { has_country_modifier = federation_shared_institution }
				ROOT = {
					institution_difference = {
						who = PREV
						value = 1
					}
				}
			}
			add_country_modifier = {
				name = federation_shared_institution
				duration = 3650
			}
			capital_scope = {
				add_next_institution_embracement = 50
			}
			add_opinion = {
				who = ROOT
				modifier = federation_pos_intrigue
			}
		}
		hidden_effect = { set_country_flag = recent_federation_politic }
	}
	
	#Exile troublemaker
	option = {
		name = fed_politics.1.e
		ai_chance = {
			factor = 50

			modifier = {
				factor = 0
				stability = 2
			}
			modifier = {
				factor = 0
				NOT = { manpower_percentage = 0.5 }
			}
		}
		trigger = {
			OR = {
				average_unrest = 1
				NOT = { stability = 0 }
			}
			manpower = 4
			NOT = { has_country_modifier = federation_exiled_troublemaker }
			5284 = { owner = { has_country_modifier = lake_federation_member } }
		}
		lake_federation_lose_1_politics = yes
		add_country_modifier = {
			name = federation_exiled_troublemaker
			duration = 3650
		}
		hidden_effect = { country_event = { id = fed_politics.101 } }
		tooltip = {
			random_list = {
				40 = {
					add_manpower = -1
					hidden_effect = { set_country_flag = manpower_1 }
					5284 = { 
						owner = {
							country_event = { id = lake.51 days = 10 random = 60 }
						}
					}
				}
				30 = {
					add_manpower = -2
					hidden_effect = { set_country_flag = manpower_2 }
					5284 = { 
						owner = {
							country_event = { id = lake.51 days = 10 random = 60 }
						}
					}
				}
				20 = {
					add_manpower = -3
					hidden_effect = { set_country_flag = manpower_3 }
					5284 = { 
						owner = {
							country_event = { id = lake.51 days = 10 random = 60 }
						}
					}
				}
				10 = {
					add_manpower = -4
					hidden_effect = { set_country_flag = manpower_4 }
					5284 = { 
						owner = {
							country_event = { id = lake.51 days = 10 random = 60 }
						}
					}
				}
			}
		}
		hidden_effect = { country_event = { id = fed_politics.102 } }
		tooltip = {
			random_list = {
				40 = {
					add_stability_or_adm_power = yes
				}
				55 = {}
				5 = {
					trigger = { ai = no }
					reduce_stability_or_adm_power = yes
				}
			}
		}
		hidden_effect = { set_country_flag = recent_federation_politic }
	}
	
	#Triumph Over Centaurs
	option = {
		name = fed_politics.1.f
		ai_chance = {
			factor = 50
			modifier = {
				factor = 0
				NOT = { need_7_politics = yes }
			}
		}
		highlight = yes
		trigger = {
			has_country_flag = federation_cohesion_triumph
			NOT = { has_country_flag = federation_cohesion_triumphed }
		}
		if = {
			limit = { need_7_politics = yes }
			lake_federation_lose_7_politics = yes
			lake_federation_gain_10_points = yes
			custom_tooltip = lake_federation_cohesion_up
			country_event = { id = lake.106 }
			hidden_effect = { set_country_flag = recent_federation_politic }
			set_country_flag = federation_cohesion_triumphed
		}
		else = {
			custom_tooltip = federation_politic_point_7_tooltip
			tooltip = {
				lake_federation_lose_7_politics = yes
				lake_federation_gain_10_points = yes
				custom_tooltip = lake_federation_cohesion_up
				country_event = { id = lake.106 }
			}
		}
	}
	
	#Doomsay Over Centaurs
	option = {
		name = fed_politics.1.g
		ai_chance = {
			factor = 50
			modifier = {
				factor = 0
				NOT = { need_7_politics = yes }
			}
		}
		highlight = yes
		trigger = {
			any_core_province = { owner = { has_country_modifier = centaur_administration } }
			NOT = { has_country_flag = federation_cohesion_doomsayd }
		}
		if = {
			limit = { need_7_politics = yes }
			lake_federation_lose_7_politics = yes
			lake_federation_lose_10_points = yes
			custom_tooltip = lake_federation_cohesion_down
			country_event = { id = lake.107 }
			hidden_effect = { set_country_flag = recent_federation_politic }
			set_country_flag = federation_cohesion_doomsayd
		}
		else = {
			custom_tooltip = federation_politic_point_7_tooltip
			tooltip = {
				lake_federation_lose_7_politics = yes
				lake_federation_lose_10_points = yes
				custom_tooltip = lake_federation_cohesion_down
				country_event = { id = lake.107 }
			}
		}
	}
	
	#Go Back
	option = {
		name = fed_politics.1.h
		ai_chance = { factor = 1 }
		if = {
			limit = { ai = yes }
			add_country_modifier = {
				name = fed_assembly_house_cooldown
				duration = 180
			}
		}
	}
}

#FederationReunion
country_event = {
	id = fed_politics.3
	title = fed_politics.3.t
	desc = fed_politics.3.d
	picture = FORT_eventPicture
	
	hidden = yes
	
	immediate = {
		set_global_flag = federation_assembly_reunion
	}
	
	trigger = {
		has_country_modifier = lake_federation_member
		has_global_flag = federation_assembly_house_build
		NOT = { has_disaster = lake_federation_crisis }
		OR = {
			NOT = { has_global_flag = federation_assembly_reunion }
			had_global_flag = {
				flag = federation_assembly_reunion
				days = 730
			}
		}
		any_owned_province = {
			has_province_modifier = federation_assembly_house
		}
	}
	
	option = {
		name = fed_politics.3.a
		ai_chance = { factor = 50 }
		every_known_country = {
			limit = { has_country_modifier = lake_federation_member }
			country_event = { id = fed_politics.4 }
		}
		country_event = { id = fed_politics.4 }
	}
}
