##Colonist stuck on settlement growth fix
country_event = {
	id = anb_debug.2
	title = anb_debug.2.t
	desc = anb_debug.2.d
	picture = HUIZTILOPOCHTLI_eventPicture
	
	hidden = yes
	
	mean_time_to_happen = {
		months = 6
	}
	
	trigger = {
		has_country_flag = avoids_colonist_settlement_growth
		any_owned_province = {
			has_colonist = yes
			is_city = yes
		}
		ai = yes
	}
	
	option = {
		name = debug.6.a
		
		every_owned_province = {
			limit = { has_colonist = yes is_city = yes }
			cancel_construction = yes
		}
	}
}

##Cleaning province
province_event = {
	id = anb_debug.3
	title = anb_debug.3.t
	desc = anb_debug.3.d
	picture = HUIZTILOPOCHTLI_eventPicture
	
	hidden = yes
	is_triggered_only = yes
	
	trigger = {
		has_province_flag = cant_colonize
	}
	
	option = {
		name = debug.6.a
		
		if = {
			limit = { 
				owner = { ai = no }
			}
			clr_province_flag = cant_colonize
		}
		else = {
			add_colonysize = -1000
			clr_province_flag = cant_colonize
		}
	}
}
