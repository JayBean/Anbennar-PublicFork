namespace = anb_debug

country_event = {
	id = anb_debug.100
	title = anb_debug.100.t
	desc = anb_debug.100.d
	picture = HUIZTILOPOCHTLI_eventPicture

	is_triggered_only = yes

	immediate = {
		if = {
			limit = {
				has_government_attribute = is_technocracy
			}
			custom_tooltip = "+5 - Has a Technocratic government"
		}
		else_if = {
			limit = {
				has_government_attribute = is_magocracy
			}
			custom_tooltip = "-5 - Has a Magocratic government"
		}


		#artificer admin/mil/religion
		if = {
			limit = {
				OR = {
					has_country_modifier = gnomish_administration
					has_country_modifier = goblin_administration
					has_country_modifier = kobold_administration
				}
			}
			custom_tooltip = "+3 - Has gnomish, goblin, or kobold administration"
		}
		if = {
			limit = {
				OR = {
					has_country_modifier = gnomish_military
					has_country_modifier = goblin_military
					has_country_modifier = kobold_military
				}
			}
			custom_tooltip = "+2 - Has gnomish, goblin, or kobold military"
		}
		if = {
			limit = {
				OR = {
					religion = ravelian
					religion = the_thought
				}
			}
			custom_tooltip = "+2 - Has a pro-Artificer religion (Ravelian, The Thought)"
		}
	
		#trolls get a penalty because they are stupid. elves, harimari and dwarves get a penalty because they are traditionalist. Emerald orcs because they hate technology
	
		if = {
			limit = {
				OR = {
					has_country_modifier = troll_administration
					has_country_modifier = elven_administration
					has_country_modifier = dwarven_administration
					has_country_modifier = harimari_administration
				}
			}
			custom_tooltip = "-2 - Has troll, elf, harimari, or dwarf administration."
		}
	
		if = {
			limit = {
				OR = {
					has_country_modifier = troll_military
					has_country_modifier = elven_military
					has_country_modifier = dwarven_military
					has_country_modifier = harimari_military
				}
			}
			custom_tooltip = "-1 - Has troll, elf, harimari, or dwarf military."
		}
		#deepwoods fey hate tech
		if = {
			limit = {
				OR = {
					religion = fey_court
					religion = ashentree_pact
				}
			}
			custom_tooltip = "-2 - Has a anti-Artificer religion (Fey Court, Ashentree Pact)"
		}
		#liches and witch kings
		if = {
			limit = {
				has_ruler_flag = lich_ruler
			}
			custom_tooltip = "-10 - Liches have no need for artifice, and see artificers as a threat."
		}
		if = {
			limit = {
				OR = {
					has_ruler_modifier = witch_king_modifier
					has_country_flag = witch_king_flag
				}
				NOT = {
					has_ruler_flag = lich_ruler
				}
			}
			custom_tooltip = "-5 - Witch Kings have no need for artifice and see it as a threat to their power"
		}

		#institutions
		if = {
			limit = {
				has_institution = manufactories
			}
			custom_tooltip = "+2 - Embraced the manufactories institution"
		}
		if = {
			limit = {
				has_institution = industrialization
			}
			custom_tooltip = "+5 - Embraced the industrialization institution"
		}

		#estates
		if = {
			limit = {
				has_estate = estate_mages
				NOT = {
					estate_influence = {
						estate = estate_mages
						influence = 20
					}	
				}
			}
			custom_tooltip = "-2 - Have the Mages estate in your country"
		}
		else_if = {
			limit = {
				estate_influence = {
					estate = estate_mages
					influence = 60
				}
			}
			custom_tooltip = "-12 - 60% Mages influence"
		}
		else_if = {
			limit = {
				estate_influence = {
					estate = estate_mages
					influence = 80
				}
			}
			custom_tooltip = "-8 - 40% Mages influence"
		}
		else_if = {
			limit = {
				estate_influence = {
					estate = estate_mages
					influence = 60
				}
			}
			custom_tooltip = "-4 - 20% Mages influence"
		}
	
		if = {
			limit = {
				has_estate = estate_mages
				NOT = {
					estate_influence = {
						estate = estate_artificers
						influence = 20
					}	
				}
			}
			custom_tooltip = "+2 - Has the Artificers estate in your country"
		}
		else_if = {
			limit = {
				estate_influence = {
					estate = estate_artificers
					influence = 60
				}
			}
			custom_tooltip = "+12 - 60% Artificers influence"
		}
		else_if = {
			limit = {
				estate_influence = {
					estate = estate_artificers
					influence = 40
				}
			}
			custom_tooltip = "+8 - 40% Artificers influence"
		}
		else_if = {
			limit = {
				estate_influence = {
					estate = estate_artificers
					influence = 20
				}
			}
			custom_tooltip = "+4 - 20% Artificers Influence"
		}

			
		#innovativeness
		if = {
			limit = {
				innovativeness = 100
			}
			custom_tooltip = "+10 - 100 innovativeness"
		}
		else_if = {
			limit = {
				innovativeness = 80
			}
			custom_tooltip = "+8 - 80 innovativeness"
		}
		else_if = {
			limit = {
				innovativeness = 60
			}
			custom_tooltip = "+6 - 60 innovativeness"
		}
		else_if = {
			limit = {
				innovativeness = 40
			}
			custom_tooltip = "+4 - 40 innovativeness"
		}
		else_if = {
			limit = {
				innovativeness = 20
			}
			custom_tooltip = "+2 - 20 innovativeness"
		}

		#techs
		if = {
			limit = {
				adm_tech = 32
			}
			custom_tooltip = "+2 - Admin Tech 32"
		}
		if = {
			limit = {
				dip_tech = 32
			}
			custom_tooltip = "+2 - Dip Tech 32"
		}
		if = {
			limit = {
				mil_tech = 32
			}
			custom_tooltip = "+1 - Mil Tech 32"
		}
		#setting up the TT
		hidden_effect = {
			set_variable = {
				which = DamestearArtificePoints
				value = 0
			}
			if = {
				limit = {
					damestear = 1
					NOT = {
						is_colonial_nation = yes
					}
				}
				every_owned_province = {
					limit = {
						trade_goods = damestear
					}
					owner = {
						change_variable = {
							which = DamestearArtificePoints
							value = 10
						}
					}
				}
			}
			#resources owned by CNs
			if = {
				limit = {
					any_subject_country = {
						damestear = 1
					}
				}
				every_subject_country = {
					limit = {
						is_colonial_nation = yes
					}
					every_owned_province = {
						limit = {
							trade_goods = damestear
						}
						owner = {
							overlord = {
								change_variable = {
									which = DamestearArtificePoints
									value = 10
								}
							}
						}
					}
				}
			}
		}
		#resources
		custom_tooltip = damestear_points_tt
		hidden_effect = {
			set_variable = {
				which = RelicsArtificePoints
				value = 0
			}
			if = {
				limit = {
					precursor_relics = 1
					NOT = {
						is_colonial_nation = yes
					}
				}
				every_owned_province = {
					limit = {
						trade_goods = precursor_relics
					}
					owner = {
						change_variable = {
							which = RelicsArtificePoints
							value = 5
						}
					}
				}
			}
			if = {
				limit = {
					any_subject_country = {
						precursor_relics = 1
					}
				}
				every_subject_country = {
					limit = {
						is_colonial_nation = yes
					}
					every_owned_province = {
						limit = {
							trade_goods = precursor_relics
						}
						owner = {
							overlord = {
								change_variable = {
									which = RelicsArtificePoints
									value = 5
								}
							}
						}
					}
				}
			}
		}
		custom_tooltip = relics_points_tt
		hidden_effect = {
			set_variable = {
				which = CoalArtificePoints
				value = 0
			}
			if = {
				limit = {
					coal = 1
					NOT = {
						is_colonial_nation = yes
					}
				}
				every_owned_province = {
					limit = {
						trade_goods = coal
					}
					owner = {
						change_variable = {
							which = CoalArtificePoints
							value = 2  #coal is already valuable as fuck. And it isn't really magical
						}
					}
				}
			}
			if = {
				limit = {
					any_subject_country = {
						coal = 1
					}
				}
				every_subject_country = {
					limit = {
						is_colonial_nation = yes
					}
					every_owned_province = {
						limit = {
							trade_goods = coal
						}
						owner = {
							overlord = {
								change_variable = {
									which = CoalArtificePoints
									value = 2
								}
							}
						}
					}
				}
			}
		}
		custom_tooltip = coal_points_tt

		custom_tooltip = artificer_points_static_tt
		custom_tooltip = artificer_points_upkeep_tt
		hidden_effect = {
			set_up_artifice_system = yes
			update_artifice_points = yes
		}
	}
	
	option = {
		name = debug.100.a
	}
}
