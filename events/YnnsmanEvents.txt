country_event = {
	id = ynnsman_events.103
	title = ynnsman_events.103.t
	desc = ynnsman_events.103.d
	picture = ADVISOR_eventPicture

	is_triggered_only = yes

	trigger = {
		capital_scope = { superregion = ynn_superregion }
		culture_group = ynnic_ruinborn_elf
		NOT = { government = republic }
		OR = {
			has_human_advisor = yes
			has_elven_advisor = yes
			has_dwarven_advisor = yes
			has_half_orcish_advisor = yes
			has_halfling_advisor = yes
		}
	}

	immediate = {
		hidden_effect = {
			random_list = {
				1 = {
					trigger = {
						OR = { #there's a lot of people in the Ynn
							has_human_advisor_of_category = { category = ADM }
							has_elven_advisor_of_category = { category = ADM }
							has_dwarven_advisor_of_category = { category = ADM }
							has_half_orcish_advisor_of_category = { category = ADM }
							has_halfling_advisor_of_category = { category = ADM }
						}
					}
					set_country_flag = advisor_events_adm
				}
				1 = {
					trigger = {
						OR = {
							has_human_advisor_of_category = { category = DIP }
							has_elven_advisor_of_category = { category = DIP }
							has_dwarven_advisor_of_category = { category = DIP }
							has_half_orcish_advisor_of_category = { category = DIP }
							has_halfling_advisor_of_category = { category = DIP }
						}
					}
					set_country_flag = advisor_events_dip
				}
				1 = {
					trigger = {
						OR = {
							has_human_advisor_of_category = { category = MIL }
							has_elven_advisor_of_category = { category = MIL }
							has_dwarven_advisor_of_category = { category = MIL }
							has_half_orcish_advisor_of_category = { category = MIL }
							has_halfling_advisor_of_category = { category = MIL }
						}
					}
					set_country_flag = advisor_events_mil
				}
			}
		}
	}

	after = {
		hidden_effect = {
			trigger_switch = {
				on_trigger = has_country_flag
				advisor_events_adm = { clr_country_flag = advisor_events_adm }
				advisor_events_dip = { clr_country_flag = advisor_events_dip }
				advisor_events_mil = { clr_country_flag = advisor_events_mil }
			}
		}
	}

	option = { #favor skill
		name = ynnsman_events.103.a
		ai_chance = { factor = 100 }
		add_adm_power = 15
		add_dip_power = 15
		add_mil_power = 15
		reduce_estate_nobles_loyalty_effect = yes
		add_estate_loyalty_modifier = {
			estate = estate_nobles
			loyalty = -10
			desc = REFUSED_PRIVILEGE
			duration = 7300
		}
	}

	option = { #favor status
		name = ynnsman_events.103.b
		ai_chance = { factor = 100 }
		add_corruption = 0.5
		trigger_switch = { #no scripted effect to kill advisors so he will be allowed to live
			on_trigger = has_country_flag
			advisor_events_adm = { remove_advisor_by_category = ADM }
			advisor_events_dip = { remove_advisor_by_category = DIP }
			advisor_events_mil = { remove_advisor_by_category = MIL }
		}
	}
}
