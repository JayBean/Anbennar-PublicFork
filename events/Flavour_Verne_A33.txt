		months = 48

			factor = 0.5

			capital_scope = {

						estate_influence = {
							estate = estate_adventurers

						}
					}
				}
				set_province_flag = corinite_verne
				change_religion = corinite
			}

					NOT = { has_province_flag = corinite_verne }

						}
					}
				}

				change_religion = corinite
			}
			random_owned_province = {
				limit = {
					NOT = { has_province_flag = corinite_verne }

						estate_influence = {
							estate = estate_adventurers
							influence = 75
						}
					}
				}

				change_religion = corinite

					NOT = { has_province_flag = corinite_verne }

						estate_influence = {
							estate = estate_adventurers
							influence = 100
						}
					}
				}
				set_province_flag = corinite_verne
				change_religion = corinite
			}
			every_owned_province = {
				limit = {
					has_province_flag = corinite_verne
				}
				clr_province_flag = corinite_verne
			}
		}
	}
	
	#Convert to Corinite
	option = {
		name = verne.12.a
		ai_chance = {
			factor = 80
			modifier = {
				factor = 3
				any_owned_province = {
				religion = corinite
				}
			}
			modifier = {
				factor = 0.5
				is_emperor = yes
				hre_religion_treaty = no
			}
			modifier = {
				factor = 3
				any_owned_province = {
				is_reformation_center = yes
				}
			}
			modifier = {
				factor = 1.5
				has_personal_deity = corin
			}
			modifier = {
				factor = 1.5
				mil = 4
			}
			modifier = {
				factor = 1.5
				personality = ai_militarist
			}
			modifier = {
				factor = 1.25
				ruler_has_personality = inspiring_leader_personality
			}
			modifier = {
				factor = 1.25
				ruler_has_personality = bold_fighter_personality
			}
		}
		capital_scope = {
			change_religion = corinite
			add_reform_center = corinite
			add_permanent_province_modifier = {
				name = "religious_zeal_at_conv"
				duration = 9000
			}
		}
		change_religion = corinite
		add_country_modifier = {
			name = "conversion_zeal"
			duration = 3650
		}
	}
	
	#Stay Regent Court
	option = {
		name = verne.12.b

			factor = 20

				factor = 1.25

			}

				factor = 1.25

			}
			modifier = {
				factor = 1.25
				ruler_has_personality = careful_personality
			}
			modifier = {
				factor = 2
				is_defender_of_faith = yes
			}
		}
		reduce_stability_or_adm_power = yes
	}
