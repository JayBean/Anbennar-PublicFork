#milgame
country_event = {
	id = mengimonuments_events.2
	title = "mengimonuments_events.2.t"
	desc = "mengimonuments_events.2.d"
	picture = PIETY_eventPicture

	fire_only_once = no

	mean_time_to_happen = {
		months = 300
		modifier = {
			factor = 0.4
			any_owned_province = {
				has_great_project = {
					type = e_arenaginku_adras
					tier = 3
				}
			}
		}
	}	
	
	trigger = {
		OR = {
			culture_group = mengi
			dominant_culture = naleni_harpy
		}
		any_owned_province = {
			has_great_project = {
				type = e_arenaginku_adras
				tier = 2
				}
			}	
		NOT = {
			has_country_modifier = e_arenaginku_adras_cooldown_mod
		}
	}

	option = {
		name = "mengimonuments_events.2.a"
		add_army_tradition = 10
		add_army_professionalism = 0.025
		hidden_effect = {
			if = {
				limit = { 
					any_owned_province = {
						has_great_project = {
							type = e_arenaginku_adras
							tier = 3
						}
					}
				}
				add_country_modifier = {
					name = e_arenaginku_adras_cooldown_mod
					duration = 1825
				}
			}	
			else_if = {
				limit = { 
					any_owned_province = {
						has_great_project = {
							type = e_arenaginku_adras
							tier = 2
						}
					}
				}
				add_country_modifier = {
					name = e_arenaginku_adras_cooldown_mod
					duration = 5475
				}
			}
		}
	}
}
