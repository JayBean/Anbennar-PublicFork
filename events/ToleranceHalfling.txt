	is_triggered_only = yes
		
	immediate = {
		hidden_effect = {

				}

			}
			random_owned_province = {
				limit = {
					has_large_halfling_minority_trigger = yes
					development = 12
				}
				save_event_target_as = halfling_overpopulated_province
			}
			random_owned_province = {
				limit = {
					has_halfling_majority_trigger = yes
				}
				save_event_target_as = halfling_overpopulated_province
			}
			random_owned_province = {
				limit = {
					has_halfling_majority_trigger = yes
					development = 12
				}
				save_event_target_as = halfling_overpopulated_province
			}
			event_target:halfling_overpopulated_province = {
				random_neighbor_province = {
					save_event_target_as = halfling_move_to_province
				}
				random_neighbor_province = {
					limit = { owned_by = ROOT }
					save_event_target_as = halfling_move_to_province
				}
				random_neighbor_province = {
					limit = {
						NOT = { has_halfling_majority_trigger = yes }
						owned_by = ROOT
					}
					save_event_target_as = halfling_move_to_province
				}
				random_neighbor_province = {
					limit = {
						NOT = { has_halfling_majority_trigger = yes }
						NOT = { has_large_halfling_minority_trigger = yes }
						owned_by = ROOT
					}
					save_event_target_as = halfling_move_to_province
				}
				random_neighbor_province = {
					limit = {
						NOT = {
							has_any_halfling_pop_trigger = yes
						}
						owned_by = ROOT
					}
					save_event_target_as = halfling_move_to_province
				}
				random_neighbor_province = {
					limit = {
						NOT = { has_any_halfling_pop_trigger = yes }
						NOT = { development = 12 }
						owned_by = ROOT
					}
					save_event_target_as = halfling_move_to_province
				}
			}
		}
	}

	# Help them move
	option = {
		name = halfling_tolerance_events.7.a
		ai_chance = {
			factor = 50
		}
		add_years_of_income = -0.4
		medium_increase_of_halfling_tolerance_effect = yes
		event_target:halfling_overpopulated_province = {
			if = {
				limit = { base_tax = 2 }
				random = {
					chance = 50
					add_base_tax = -1
				}
				event_target:halfling_move_to_province = {
					add_base_tax = 1
				}
			}
			if = {
				limit = { base_manpower = 2 }
				random = {
					chance = 50
					add_base_manpower = -1
				}
				event_target:halfling_move_to_province = {
					add_base_manpower = 1
				}
			}
		}
		event_target:halfling_move_to_province = {
			add_halfling_minority_size_effect = yes
		}

		}
