### Event to change culture

	id = aelnar.1
	title = aelnar.1.t
	desc = aelnar.1.d

		always = yes
	}

        name = aelnar.1.a

			name = aelnar_citizenship
			duration = -1
		}
	}
}

##Kayd Revolt
country_event = {
	id = aelnar.45
	title = aelnar.45.t
	desc = aelnar.45.d
	picture = REVOLT_eventPicture
	goto = province
	mean_time_to_happen = {
		months = 3
	}
	
	trigger = {
		tag = Z43
		has_disaster = aelnar_civil_war
		check_variable = { kayd_value = 100 }
		NOT = { has_country_flag = kayd_revolted }
		any_owned_province = {
			is_capital = no
			is_in_capital_area = yes
			is_city = yes
			owned_by = ROOT
		}
	}
	
	immediate = {
		hidden_effect = {
			set_country_flag = kayd_revolted
			random_owned_province = {
				limit = {
					is_capital = no
					is_in_capital_area = yes
					is_city = yes
				}
				save_event_target_as = province
				add_core = Z70
				ROOT = { release_with_religion_and_culture = { tag = Z70 religion = elven_forebears culture = moon_elf } }
				area = {
					limit = { owned_by = ROOT is_capital = no }
					add_core = Z70
				}
			}
			every_owned_province = {
				limit = { is_core = Z70 }
				cede_province = Z70
			}
			Z70 = {
				change_technology_group = tech_elven
				change_unit_type = tech_elven
				define_ruler = {
					name = Kayd
					dynasty = Obaith
					adm = 2
					dip = 1
					mil = 6
					culture = moon_elf
					religion = elven_forebears
					min_age = 232
					male = yes
				}
				hidden_effect = { set_ruler_flag = long_lived_ruler }
				define_ruler_to_general = {
					fire = 5
					shock = 5
					manuever = 5
					siege = 2
				}
				kill_heir = { allow_new_heir = no }
				add_country_modifier = {
					name = aelnar_kayd_revolt
					duration = 3650
				}
				rianvisa_starting_army = yes
				rianvisa_declare_war = yes
			}
		}
	}
	
	option = {
        name = aelnar.45.a
        ai_chance = { factor = 100 }
	}
	
	option = { #Play as Them
        name = aelnar.45.b
        ai_chance = { factor = 0 }
		switch_tag = Z70
		custom_tooltip = aelnar_occupy_capital_rianvisa_tooltip
	}
}
