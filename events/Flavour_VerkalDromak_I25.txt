country_event = {
	id = verkaldromak.19
	title = verkaldromak.19.t
	desc = verkaldromak.19.d
	picture = WAR_OF_THE_ROSES_eventPicture
	
	# fire_only_once = yes
	# hidden = yes
	is_triggered_only = yes

	immediate = {
		hidden_effect = {
			if = {
				limit = {
					any_known_country = { 
						NOT = { tag = I25 }
						has_country_modifier = I25_friend_in_dwarovar
					}
				}
				random_country = {
					limit = {
						has_country_modifier = I25_friend_in_dwarovar
						NOT = { tag = I25 }
					}
					save_event_target_as = verkal_dromak_enemy
				}
			}
			else = { 
				random_neighbor_country = {
					limit = { culture_group = dwarven }
					save_event_target_as = verkal_dromak_enemy
				}
			}
		}
	}
	
	option = {
		name = verkaldromak.19.a

		remove_country_modifier = I25_friend_in_dwarovar
		event_target:verkal_dromak_enemy = {
			every_owned_province = {
				add_claim = ROOT
			}
			if = {
				limit = { has_country_modifier = I25_friend_in_dwarovar }
				remove_country_modifier = I25_friend_in_dwarovar
			}
		}
		if = {
			limit = { 
				event_target:verkal_dromak_enemy = { is_subject_of = ROOT }
			}
			add_opinion = {
				who = event_target:verkal_dromak_enemy
				modifier = scornfully_insulted
			}
			event_target:verkal_dromak_enemy = {
				add_opinion = {
					who = ROOT
					modifier = scornfully_insulted
				}
				add_liberty_desire = 50
			}
		}
		else = {
			declare_war_with_cb = {
				who = event_target:verkal_dromak_enemy
				casus_belli = cb_vassalize_mission
			}
		}
	}
}
