country_event = { # vassal warlord conspires to steal land
	id = taychend_flavour.6
	title = taychend_flavour.6.t
	desc = taychend_flavour.6.d
	picture = CONQUEST_eventPicture
	
	is_triggered_only = yes
	
	trigger = {
		religion = gods_of_the_taychend
		any_subject_country = {
			liberty_desire = 40
		}
		num_of_cities = 10
		OR = {
			has_personal_deity = taychend_glory_1
			has_personal_deity = taychend_glory_2
			has_personal_deity = taychend_glory_3
		}
	}
	
	immediate = {
		random_subject_country = {
			limit = {
				liberty_desire = 40
			}
			save_event_target_as = tf_conspiring_disloyal_vassal
		}
		
		random_owned_province = {
			limit = {
				is_neighbor_of = event_target:tf_conspiring_disloyal_vassal
				NOT = {
					development = 20
					is_capital = yes
				}
			}
			save_event_target_as = tf_disloyalvassalprov_1
		}
		
		random_owned_province = {
			limit = {
				is_neighbor_of = event_target:tf_conspiring_disloyal_vassal
				NOT = {
					development = 20
					is_capital = yes
					province_id = event_target:tf_disloyalvassalprov_1
				}
			}
			save_event_target_as = tf_disloyalvassalprov_2
		}
		
		random_owned_province = {
			limit = {
				is_neighbor_of = event_target:tf_conspiring_disloyal_vassal
				NOT = {
					development = 20
					is_capital = yes
					province_id = event_target:tf_disloyalvassalprov_1
					province_id = event_target:tf_disloyalvassalprov_2
				}
			}
			save_event_target_as = tf_disloyalvassalprov_3
		}
	}

	option = { # cede provinces to vassal
		name = taychend_flavour.6.a
		
		add_prestige = -10
		add_legitimacy = -10
		
		event_target:tf_conspiring_disloyal_vassal = {
			event_target:tf_disloyalvassalprov_1 = {
				cede_province = PREV
			}
			event_target:tf_disloyalvassalprov_2 = {
				cede_province = PREV
			}
			event_target:tf_disloyalvassalprov_3 = {
				cede_province = PREV
			}
		}
	}
	
	option = { # prevent the conquest
		name = taychend_flavour.6.b
		
		add_legitimacy = 10
		gain_taychend_glory_effect = { amount = 25 }
		
		add_opinion = {
			who = event_target:tf_conspiring_disloyal_vassal
			modifier = tf_stopped_vassal_conspiracy
		}
		
		if = {
			limit = {
				event_target:tf_conspiring_disloyal_vassal = {
					liberty_desire = 75
				}
			}
			
			declare_war_with_cb = {
				who = ROOT
				casus_belli = cb_independence_war
			}
		}
		
		else = {
			event_target:tf_conspiring_disloyal_vassal = {
				add_liberty_desire = 15
			}
		}	
	}
}
