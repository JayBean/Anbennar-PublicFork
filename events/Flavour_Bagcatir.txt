country_event = {
	id = bagcatir.3
	title = bagcatir.3.t
	desc = bagcatir.3.d
	picture = DIPLOMACY_eventPicture
	
	is_triggered_only = yes
	
	immediate = {
		hidden_effect = {
			fograc_provinces = {
				type = all
				limit = { 
					owned_by = H07
					is_city = yes
				}
				add_core = U43
				change_culture = fograc
				change_original_culture = fograc
				change_religion = raeldaeriach
			}
			fograc_provinces = {
				type = all
				limit = { 
					owned_by = H07
					is_city = no
				}
				change_culture = fograc
				change_original_culture = fograc
				change_religion = raeldaeriach
			}
			release_with_religion_and_culture = {
				tag = U43
				religion = raeldaeriach
				culture = fograc
			}
			U43 = {
				change_government = republic
				add_government_reform = merchants_reform
			}
			if = {
				limit = { has_dlc = "Leviathan" }
				create_subject = {
					subject_type = self_governing_colony
					subject = U43
				}
			}
			else = {
				create_subject = {
					subject_type = colony
					subject = U43
				}
			}
			
		}
	}
	
	after = {
		hidden_effect = { #Sanity check
			U43 = {
				change_religion = raeldaeriach
				change_culture = fograc
			}
		}
	}
	
	option = { 
		name = bagcatir.3.a
	}
	option = { 
		name = bagcatir.3.b
		switch_tag = U43
	}
}
