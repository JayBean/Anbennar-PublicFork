		}
		else = {
			# lose 2% existing knowledge
			multiply_variable = {
				which = vazhatun_astral_knowledge
				value = 0.98
			}
			# truncate to tenths place
			divide_variable = {
				which = vazhatun_astral_knowledge
				value = 100
			}
			multiply_variable = {
				which = vazhatun_astral_knowledge
				value = 100
			}
			# clamp to 0
			if = {
				limit = {
					NOT = {
						check_variable = {
							which = vazhatun_astral_knowledge
							value = 0
						}
					}
				}
				set_variable = {
					which = vazhatun_astral_knowledge
					value = 0
				}
			}
		}
	}
