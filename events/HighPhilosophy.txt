	title = high_philosophy_events.1.t
	desc = high_philosophy_events.1.d
	picture = MUSLIM_SCHOLAR_WRITING_eventPicture
	
	trigger = {
		has_dlc = "Cradle of Civilization"
		has_religious_school = yes
		total_development = 50
		num_of_cities = 5
		any_known_country = {
			has_religious_school = yes
			total_development = 50
			religion = ROOT
			num_of_cities = 5
			NOT = { has_religious_school_of = ROOT }
			school_opinion = { #0 or more
				who = ROOT
				opinion = 0
			}
			NOT = {
				school_opinion = { #Below 1
					who = ROOT
					opinion = 1
				}
			}
			trust = {
				who = ROOT
				value = 70
			}
		}
	}
	
	immediate = {
		hidden_effect = {
			random_known_country = {
				limit = {
					has_religious_school = yes
					total_development = 50
					num_of_cities = 5
					religion = ROOT
					NOT = { has_religious_school_of = ROOT }
					school_opinion = { #0 or more
						who = ROOT
						opinion = 0
					}

						school_opinion = { #Below 1
							who = ROOT
							opinion = 1
						}
					}
					trust = {
						who = ROOT
						value = 70
					}
				}
				save_event_target_as = school_opinion_increase_country
			}
		}
	}

	mean_time_to_happen = {
		months = 360
	}
	
	option = {
		name = high_philosophy_events.1.a #
		event_target:school_opinion_increase_country = {
			set_school_opinion = {
				who = ROOT
				opinion = 1
			}
		}
	}
}

country_event = {
	id = high_philosophy_events.3
	title = high_philosophy_events.3.t
	desc = high_philosophy_events.3.d
	picture = MUSLIM_SCHOLAR_WRITING_eventPicture
	
	trigger = {
		has_dlc = "Cradle of Civilization"
		has_religious_school = yes
		total_development = 50
		num_of_cities = 5
		any_known_country = {
			defensive_war_with = ROOT
			has_religious_school = yes
			total_development = 50
			religion = ROOT
			num_of_cities = 5
			NOT = { has_religious_school_of = ROOT }
			school_opinion = {
				who = ROOT
				opinion = 0
			}
			is_in_war = {
				defender_leader = THIS
				attacker_leader = ROOT
				duration = 1095
			}
		}
	}
	
	immediate = {
		hidden_effect = {
			random_known_country = {
				limit = {
					has_religious_school = yes
					total_development = 50
					num_of_cities = 5
					war_with = ROOT
					religion = ROOT
					NOT = { has_religious_school_of = ROOT }
					school_opinion = {
						who = ROOT
						opinion = 0
					}
					defensive_war_with = ROOT
					is_in_war = {
						defender_leader = THIS
						attacker_leader = ROOT
						duration = 1095
					}
				}
				save_event_target_as = school_opinion_decrease_country
			}
		}
	}

	mean_time_to_happen = {
		months = 30
		modifier = {
			factor = 0.75
			any_known_country = {
				defensive_war_with = ROOT
				has_religious_school = yes
				total_development = 50
				religion = ROOT
				num_of_cities = 5
				NOT = { has_religious_school_of = ROOT }
				school_opinion = {
					who = ROOT
					opinion = 0
				}
				is_in_war = {
					defender_leader = THIS
					attacker_leader = ROOT
					duration = 1825 #5 years
				}
			}
		}
	}
	
	option = {
		name = high_philosophy_events.3.a #
		if = { # 0 is changed to -1
			limit = {
				event_target:school_opinion_decrease_country = {
					NOT = {
						school_opinion = {
							who = ROOT
							opinion = 1
						}
					}
				}
			}
			event_target:school_opinion_decrease_country = {
				set_school_opinion = {
					who = ROOT
					opinion = -1
				}
			}
		}
		else = { #1 is changed to 0

			event_target:school_opinion_decrease_country = {
				set_school_opinion = {
					who = ROOT
					opinion = 0
				}
			}
		}
	}
}
