		hidden_effect = {
			
			random_province = {
				limit = {
					area = drabs_cambann_area

				}

			}

			}
			
		}

		name = aelantirspawnables.1.a
		
		H30 = {
			event_target:spawn_province = {
				create_colony = 1000
				add_core = H30
			}
		}

			if = {
				limit = {
					NOT = { event_target:spawn_province = { province_id = 2088 } }

					country_event = { 
						id = nativebuyout.1
						days = 10
					}

					id = anb_miscevents.7
					days = 7

					dip = 5
					mil = 4
