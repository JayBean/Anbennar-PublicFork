						add_permanent_province_modifier = {

							duration = -1

						add_permanent_province_modifier = { 	
							name = surface_mine0b
							duration = -1
						}
					}
					10 = {
						add_permanent_province_modifier = { 	
							name = surface_mine0c
							duration = -1
						}
					}
					10 = {
						add_permanent_province_modifier = { 	
							name = surface_mine0d
							duration = -1
						}
					}
					10 = {
						add_permanent_province_modifier = { 	
							name = surface_mine0e
							duration = -1
						}
					}
					10 = {
						add_permanent_province_modifier = { 	
							name = surface_mine0f
							duration = -1
						}
					}
					10 = {
						add_permanent_province_modifier = { 	
							name = surface_mine0g
							duration = -1
						}
					}
				}
				if = {
					limit = {
						NOT = {
							OR = {
								trade_goods = copper
								trade_goods = iron
								trade_goods = gems
								trade_goods = gold
								trade_goods = coal
								trade_goods = damestear

						}

						25 = {
							change_trade_goods = copper
						}
						40 = {
							change_trade_goods = iron
						}
						25 = {
							change_trade_goods = gems
						}
						5 = {
							change_trade_goods = gold
						}
						2 = {
							if = {
								limit = { is_year = 1700 }
								change_trade_goods = coal
							}
							else = {
								change_trade_goods = gold
							}
						}

							change_trade_goods = damestear

								name = damestear_rich_deposit

		trade_goods = damestear

		}
	}
	
	option = {
        	name = diggy_surface_mine.7.a
        	ai_chance = { factor = 100 }
		hidden_effect = {
			ROOT = {
				random_list = {
					25 = {
						change_trade_goods = copper
					}
					40 = {
						change_trade_goods = iron
					}
					26 = {
						change_trade_goods = gems
					}
					6 = {
						change_trade_goods = gold
					}
					2 = {
						if = {
							limit = { is_year = 1700 }
							change_trade_goods = coal
						}
						else = {
							change_trade_goods = gold

					}
					1 = {
						change_trade_goods = precursor_relics

				}

		}
	}
}

	desc = diggy_surface_mine.8.d
	picture = ARTWORK_eventPicture
	goto = root
	
	mean_time_to_happen = {
		years = 40
		
		modifier = {
			factor = 1.5
			
			REB = {
				NOT = {
					check_variable = {
						which = num_precursor_relics
						value = 20
					}
				}
				check_variable = {
					which = num_precursor_relics
					value = 10
				}
			}
		}
		
		modifier = {
			factor = 2
			
			REB = {
				NOT = {
					check_variable = {
						which = num_precursor_relics
						value = 10
					}
				}
			}
		}
		modifier = {
			factor = 3
			
			has_province_modifier = eordan_expedition_site
		}
	}

			check_variable = {
				which = num_precursor_relics

	}

						change_trade_goods = gems
					}

						change_trade_goods = gold
					}

						if = {
							limit = { is_year = 1700 }
							change_trade_goods = coal
						}
						else = {
							change_trade_goods = gold
						}
					}

						change_trade_goods = damestear
					}
				}
			}

				limit = {
