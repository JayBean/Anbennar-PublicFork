		major = yes
		potential = {

			adventurer_legacy_effect = yes
			set_country_flag = adventurer_derived_government
			set_country_flag = formed_diggy_formable
			clr_country_flag = knightly_order_adventurer

			if = {
				limit = { has_custom_ideas = no }
				country_event = { id = ideagroups.1 } #Swap Ideas
			}
			every_owned_province = {
				limit = { culture = ROOT }
				change_culture = kronium_dwarf
			}
			if = {
				limit = { ruler_culture = ROOT }
				set_ruler_culture = kronium_dwarf
			}
			if = {
				limit = { has_heir = yes heir_culture = ROOT }
				set_heir_culture = kronium_dwarf
			}
			if = {
				limit = { has_consort = yes consort_culture = ROOT }
				set_consort_culture = kronium_dwarf
			}
			change_primary_culture = kronium_dwarf
		}
		ai_will_do = {

		}
