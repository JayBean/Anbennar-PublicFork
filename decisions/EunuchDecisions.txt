
country_decisions ={
	suppress_eunuch_caste = {
		major = no
		potential = {
			has_estate = estate_eunuchs
			NOT = {
				has_reform = eunuch_republic
				has_reform = eunuch_dynasty_reform
			}
		}
		allow = {
			has_estate = estate_eunuchs
			NOT = {
				has_reform = eunuch_republic
				has_reform = eunuch_dynasty_reform
			}
		}
		effect = {
			add_ruler_modifier = { 
				name = eunuchs_purged
				duration = -1
			}
			custom_tooltip = eunuch_purge_tt
			if = {
				limit = {
					estate_influence = {
						estate = estate_eunuchs
						influence = 80
					}
				}
				spawn_rebels = {
					type = noble_rebels
					size = 3
				}
			}
			else_if = {
				limit = {
					estate_influence = {
						estate = estate_eunuchs
						influence = 60
					}
				}
				spawn_rebels = {
					type = noble_rebels
					size = 2
				}
			}
			else = {
				spawn_rebels = {
					type = noble_rebels
					size = 1
				}
			}
			set_country_flag = anti_eunuch_country
		}
		
		ai_will_do = {
			factor = 0	
		}
	}
	reinstate_eunuch_caste = {
		major = no
		potential = {
			NOT = { tag = Y05 }
			has_country_flag = anti_eunuch_country
			NOT = { has_ruler_modifier = eunuchs_purged }
		}
		allow = {
			has_country_flag = anti_eunuch_country
			NOT = { has_ruler_modifier = eunuchs_purged }
		}
		effect = {
			clr_country_flag = anti_eunuch_country
			custom_tooltip = eunuchs_return_tt
		}
		ai_will_do = {
			factor = 0
		}
	}
}
